43 /// \class StrideIterator
44 /// \tparam T value type
45 /// \brief Row-wise iterator
46 /// \warning This is not efficient, just for the sake of completeness
47 /// \ingroup ds
48 template <typename T>
49 class StrideIterator {
50  public:
51  // NOTE that we cannot use random iterator
52  // thus, using this iterator in some STL routines may not be efficient
53 
54  /// \name STL iterator requirements
55  ///@{
56 
57  typedef T* pointer; ///< pointer type
58  typedef std::bidirectional_iterator_tag iterator_category; ///< iter tag
59  typedef T value_type; ///< value
60  typedef T& reference; ///< reference
61  typedef std::ptrdiff_t difference_type; ///< difference
62 
63  ///@}
64 
65  /// \brief default constructor
66  StrideIterator() : _ptr(nullptr), _stride(0) {}
67 
68  /// \brief constructor with pointer
69  /// \param[in] ptr data pointer
70  /// \param[in] stride jump size (optional)
71  explicit StrideIterator(pointer ptr, const difference_type stride = 1)
72  : _ptr(ptr), _stride(stride) {}
73 
74  /// \brief constructor with another iterator
75  /// \tparam V another value type
76  /// \param[in] other another iterator
77  template <typename V>
78  explicit StrideIterator(const StrideIterator<V>& other)
79  : _ptr(&(*other)), _stride(other.stride()) {}
80 
81  StrideIterator(const StrideIterator&) = default;
82  StrideIterator& operator=(const StrideIterator&) = default;
83 
84  /// \brief assign to another iterator
85  /// \tparam V another value type
86  /// \param[in] rhs right-hand side iterator
87  template <typename V>
88  StrideIterator& operator=(const StrideIterator<V>& rhs) {
89  _ptr = &(*rhs);
90  _stride = rhs.stride();
91  return *this;
92  }
93 
94  // iterator interfaces
95  inline pointer operator->() const { return _ptr; }
96  inline reference operator*() const { return *_ptr; }
97  inline StrideIterator& operator++() {
98  _ptr += _stride;
99  return *this;
100  }
101  inline StrideIterator& operator--() {
102  _ptr -= _stride;
103  return *this;
104  }
105  inline StrideIterator operator++(int) {
106  StrideIterator tmp(*this);
107  ++(*this);
108  return tmp;
109  }
110  inline StrideIterator operator--(int) {
111  StrideIterator tmp(*this);
112  --(*this);
113  return tmp;
114  }
115 
116  // mostly for internal use
117  inline difference_type stride() const { return _stride; }
118 
119  /// \brief get reference
120  /// \param[in] i i-th index
121  /// \note stride is applied here
122  inline reference operator[](const difference_type i) const {
123  return _ptr[i * _stride];
124  }
125 
126  /// \brief compare equal
127  /// \param[in] rhs right-hand side iterator
128  inline bool operator==(const StrideIterator& rhs) const {
129  return _ptr == rhs._ptr;
130  }
131 
132  /// \brief compare not equal
133  /// \param[in] rhs right-hand side iterator
134  inline bool operator!=(const StrideIterator& rhs) const {
135  return !(*this == rhs);
136  }
137 
138  /// \brief compare equal with a foreign iterator
139  /// \tparam V another value type
140  /// \param rhs right-hand side iterator
141  template <typename V>
142  inline bool operator==(const StrideIterator<V>& rhs) const {
143  return _ptr == &(*rhs);
144  }
145 
146  /// \brief compare not equal with a foreign iterator
147  /// \tparam V another value type
148  /// \param rhs right-hand side iterator
149  template <typename V>
150  inline bool operator!=(const StrideIterator<V>& rhs) const {
151  return !(*this == rhs);
152  }
153 
154  private:
155  pointer _ptr; ///< pointer
156  difference_type _stride; ///< stride size

352  /// \brief convert CRS to dense storage
353  /// \tparam Cs compressed storage
354  /// \param[in] crs CRS matrix
355  /// \note SFINAE for CCS case
356  template <class Cs>
357  inline typename std::enable_if<Cs::ROW_MAJOR>::type copy_sparse(
358  const Cs& crs) {
359  if (crs.status() == DATA_UNDEF && (crs.nrows() || crs.ncols()))
360  hif_warning("input CRS ia a all-zero matrix.");
361  const size_type nrows(crs.nrows());
362  resize(nrows, crs.ncols());
363  // first step, set all values to zero
364  std::fill(_data.begin(), _data.end(), value_type());
365  // not efficient for CRS to dense
366  for (size_type r = 0u; r < nrows; ++r) {
367  auto i_itr = crs.col_ind_cbegin(r), i_end = crs.col_ind_cend(r);
368  for (auto v_itr = crs.val_cbegin(r); i_itr != i_end; ++i_itr, ++v_itr)
369  (*this)(r, *i_itr) = *v_itr;
370  }
371  }
372 
373  /// \brief convert CCS to dense storage
374  /// \tparam Cs compressed storage
375  /// \param[in] ccs CCS matrix
376  /// \note SFINAE for CRS case
377  template <class Cs>
378  inline typename std::enable_if<!Cs::ROW_MAJOR>::type copy_sparse(
379  const Cs& ccs) {
380  if (ccs.status() == DATA_UNDEF && (ccs.nrows() || ccs.ncols()))
381  hif_warning("input CCS ia a all-zero matrix.");
382  const size_type ncols(ccs.ncols());
383  resize(ccs.nrows(), ncols);
384  // first step, set all values to zero
385  std::fill(_data.begin(), _data.end(), value_type());
386  // efficient for CCS to dense
387  for (size_type c = 0u; c < ncols; ++c) {
388  auto i_itr = ccs.row_ind_cbegin(c), i_end = ccs.row_ind_cend(c);
389  const size_type ld = c * _nrows;
390  for (auto v_itr = ccs.val_cbegin(c); i_itr != i_end; ++i_itr, ++v_itr)
391  _data[ld + *i_itr] = *v_itr;
392  }
393  }
