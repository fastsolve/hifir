45 /// \brief drop \a L_E matrix or \a U_F matrix
46 /// \tparam IndPtrArray index pointer array
47 /// \tparam IntArray integer array, see \ref Array
48 /// \tparam ValueArray value array, see \ref Array
49 /// \tparam BufArray value buffer array type
50 /// \tparam IntBufArray integer buffer array type
51 /// \param[in] ref_indptr reference starting position array
52 /// \param[in] alpha drop space limiter threshold
53 /// \param[in,out] indptr in and out put index starting array
54 /// \param[in,out] indices in and out put index array
55 /// \param[in,out] vals in and out put values
56 /// \param[out] buf work space
57 /// \param[out] ibuf integer work space
58 /// \ingroup schur
59 template <class IndPtrArray, class IntArray, class ValueArray, class BufArray,
60  class IntBufArray>
61 inline void drop_offsets_kernel(const IndPtrArray &ref_indptr,
62  const double alpha, IndPtrArray &indptr,
63  IntArray &indices, ValueArray &vals,
64  BufArray &buf, IntBufArray &ibuf) {
65  using size_type = typename IntArray::size_type;
66  using index_type = typename IntArray::value_type;
67 
68  const size_type n = indptr.size() - 1;
69 
70  // loop starts here
71  auto &gaps = ibuf;
72  for (size_type i(0); i < n; ++i) {
73  const size_type A_sz = ref_indptr[i + 1] - ref_indptr[i],
74  sz_thres = std::ceil(alpha * A_sz),
75  nnz = indptr[i + 1] - indptr[i];
76  if (sz_thres >= nnz) {
77  // if the threshold is no smaller than the local nnz
78  gaps[i] = 0;
79  continue;
80  }
81  gaps[i] = nnz - sz_thres;
82  // fetch the value to the buffer
83  const size_type first = indptr[i], last = indptr[i + 1];
84  for (size_type j(first); j < last; ++j) buf[indices[j]] = vals[j];
85  std::nth_element(
86  indices.begin() + first, indices.begin() + first + sz_thres - 1,
87  indices.begin() + last, [&](const index_type ii, const index_type jj) {
88  return std::abs(buf[ii]) > std::abs(buf[jj]);
89  });
90  // sort
91  // std::sort(indices.begin() + first, indices.begin() + first + sz_thres);
92  // fetch back the value
93  for (size_type j(first); j < first + sz_thres; ++j)
94  vals[j] = buf[indices[j]];
95  }
96 
97  // compress
98  auto i_itr = indices.begin();
99  auto v_itr = vals.begin();
100  auto prev = indptr[0];
101  for (size_type i(0); i < n; ++i) {
102  const size_type first = prev, last = indptr[i + 1];
103  auto itr_bak = i_itr;
104  i_itr = std::copy(indices.cbegin() + first,
105  indices.cbegin() + last - gaps[i], i_itr);
106  v_itr =
107  std::copy(vals.cbegin() + first, vals.cbegin() + last - gaps[i], v_itr);
108  prev = indptr[i + 1];
109  indptr[i + 1] = indptr[i] + (i_itr - itr_bak);
110  }
111  // need to resize the nnz arrays
112  indices.resize(indptr[n]);
113  vals.resize(indptr[n]);
114 }

123 /// \brief drop \a L_E
124 /// \tparam CrsType crs storage type, see \ref CRS
125 /// \tparam BufArray value buffer type
126 /// \tparam IntBufArray integer buffer type
127 /// \param[in] ref_indptr reference starting array
128 /// \param[in] alpha drop space limiter threshold
129 /// \param[in,out] L_E in and out puts of L offset
130 /// \param[out] buf work space
131 /// \param[out] ibuf integer work space
132 /// \note buffers can be got from Crout work spaces
133 /// \sa drop_U_F
134 template <class CrsType, class BufArray, class IntBufArray>
135 inline void drop_L_E(const typename CrsType::iparray_type &ref_indptr,
136  const double alpha, CrsType &L_E, BufArray &buf,
137  IntBufArray &ibuf) {
138  static_assert(CrsType::ROW_MAJOR, "must be CRS");
139 
140  if (alpha > 0)
141  internal::drop_offsets_kernel(ref_indptr, alpha, L_E.row_start(),
142  L_E.col_ind(), L_E.vals(), buf, ibuf);
143  else {
144  std::fill_n(L_E.row_start().begin(), L_E.nrows() + 1, 0);
145  L_E.col_ind().resize(0);
146  L_E.vals().resize(0);
147  }
148 }

177 /// \brief compute the simple version of Schur complement
178 /// \tparam ScaleArray row/column scaling vector type, see \ref Array
179 /// \tparam CrsType crs matrix used for input, see \ref CRS
180 /// \tparam PermType permutation vector type, see \ref BiPermMatrix
181 /// \tparam DiagArray diagonal vector type, see \ref Array
182 /// \tparam SpVecType sparse vector buffer type, see \ref SparseVector
183 /// \param[in] s row scaling vector
184 /// \param[in] A input crs matrix
185 /// \param[in] t column scaling vector
186 /// \param[in] p row permutation vector
187 /// \param[in] q column permutation vector
188 /// \param[in] m leading block size of current level
189 /// \param[in] L_E lower part after \ref Crout update
190 /// \param[in] d diagonal entry that contains the leading block
191 /// \param[in] U_F upper part after \ref Crout update
192 /// \param[out] SC Schur complement of C version
193 /// \param[out] buf work space, can use directly passed in as from Crout bufs
194 ///
195 /// Mathematically, this is to compute:
196 ///
197 /// \f[
198 /// \mathbf{S}_C=\mathbf{A}[\mathbf{p}_{m+1:n},\mathbf{q}_{m+1:n}]
199 /// -\mathbf{L}_E\textrm{diag}(\mathbf{d_B})\mathbf{U}_F
200 /// \f]
201 ///
202 /// Where \f$\mathbf{L}_E=\mathbf{L}_{m+1:n,:}\f$,
203 /// \f$\mathbf{U}_F=\mathbf{U}_{:,m+1:n}\f$. Notice that both
204 /// \f$\mathbf{L}\f$ and \f$\mathbf{U}\f$ are lower and upper parts
205 /// \b after computing Crout updates. Therefore, they are rectangle matrices
206 /// if \f$m<n\f$. Accessing \f$\mathbf{U}_{:,m+1:n}\f$ can be efficiently
207 /// achieved with \a U_start vector, which is given after Crout updates.
208 ///
209 /// This computation involves sparse matrix multiplication, which is done in
210 /// three steps: 1) determine the total number of nonzeros, 2) determine the
211 /// index pattern (symbolic computation), and 3) fill in the numerical values.
212 template <class ScaleArray, class CrsType, class PermType, class DiagArray,
213  class SpVecType>
214 inline void compute_Schur_simple(const ScaleArray &s, const CrsType &A,
215  const ScaleArray &t, const PermType &p,
216  const PermType & q,
217  const typename CrsType::size_type m,
218  const CrsType &L_E, const DiagArray &d,
219  const CrsType &U_F, CrsType &SC,
220  SpVecType &buf) {
221  using size_type = typename CrsType::size_type;
222  using value_type = typename CrsType::value_type;
223  using boost_value_type = typename ValueTypeMixedTrait<value_type>::boost_type;
224  constexpr static bool BOOST_PRECISION =
225  !std::is_same<boost_value_type, value_type>::value;
226  using boost_scalar_type =
227  typename ValueTypeTrait<boost_value_type>::value_type;
228  static_assert(CrsType::ROW_MAJOR, "must be CRS");
229 
230  const size_type n = A.nrows();
231  if (m == n) return;
232 
233  // get the offset size
234  const size_type N(n - m);
235  SC.resize(N, N);
236  auto &row_start = SC.row_start();
237  row_start.resize(N + 1);
238  row_start.front() = 0;
239  size_type tag(n);
240 
241  // step 1, determine the overall nnz
242  for (size_type i(0); i < N; ++i, ++tag) {
243  buf.reset_counter(); // reset buffer counter
244  auto L_last = L_E.col_ind_cend(i);
245  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last; ++L_itr) {
246  auto U_last = U_F.col_ind_cend(*L_itr);
247  for (auto U_itr = U_F.col_ind_cbegin(*L_itr); U_itr != U_last; ++U_itr)
248  buf.push_back(*U_itr, tag);
249  }
250  // now add the C contribution
251  const size_type pi = p[m + i];
252  // since A is permutated, we must loop through all entries in this row
253  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
254  itr != last; ++itr) {
255  const size_type inv_q = q.inv(*itr);
256  if (inv_q >= m) buf.push_back(inv_q - m, tag);
257  }
258  // assemble row_start
259  row_start[i + 1] = row_start[i] + buf.size();
260  }
261 
262  if (row_start[N] == 0) {
263  hif_warning("Schur (S version) complement becomes empty!");
264  return;
265  }
266 
267  // step 2, fill in the indices
268  SC.reserve(row_start[N]);
269  auto &col_ind = SC.col_ind();
270  col_ind.resize(0); // just reset the internal counter, O(1) for Array
271  for (size_type i(0); i < N; ++i, ++tag) {
272  buf.reset_counter(); // reset buffer counter
273  auto L_last = L_E.col_ind_cend(i);
274  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last; ++L_itr) {
275  // get the U iter
276  auto U_last = U_F.col_ind_cend(*L_itr);
277  for (auto U_itr = U_F.col_ind_cbegin(*L_itr); U_itr != U_last; ++U_itr) {
278  hif_assert(size_type(*U_itr) < N, "%zd exceed Schur dimension %zd",
279  size_type(*U_itr), N);
280  buf.push_back(*U_itr, tag);
281  }
282  }
283  // now add the C contribution
284  const size_type pi = p[m + i];
285  // since A is permutated, we must loop through all entries in this row
286  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
287  itr != last; ++itr) {
288  const size_type inv_q = q.inv(*itr);
289  if (inv_q >= m) {
290  hif_assert(inv_q - m < N,
291  "%zd exceed Schur dimension %zd, bad index %zd in A",
292  inv_q - m, N, inv_q);
293  buf.push_back(inv_q - m, tag);
294  }
295  }
296  // we need the ensure that the indices are sorted, because it will be the
297  // input for next level
298  buf.sort_indices();
299  size_type j = col_ind.size(); // current start index
300  col_ind.resize(j + buf.size()); // resize
301  for (auto ind_itr = buf.inds().cbegin(); j < col_ind.size(); ++j, ++ind_itr)
302  col_ind[j] = *ind_itr;
303  }
304 
305  hif_assert(col_ind.size() == (size_type)row_start[N], "fatal error!");
306 
307  // step 3, fill in the numerical values
308  Array<boost_value_type> _buf;
309  boost_value_type * buf_vals;
310  if (BOOST_PRECISION) {
311  _buf.resize(N);
312  hif_error_if(_buf.status() == DATA_UNDEF, "memory allocation failed");
313  buf_vals = _buf.data();
314  } else
315  buf_vals = (boost_value_type *)buf.vals().data();
316  auto &vals = SC.vals();
317  vals.resize(col_ind.size());
318  for (size_type i(0); i < N; ++i) {
319  // first, assign values to zero
320  auto sc_ind_first = SC.col_ind_cbegin(i);
321  for (auto itr = sc_ind_first, ind_last = SC.col_ind_cend(i);
322  itr != ind_last; ++itr) {
323  hif_assert(size_type(*itr) < N, "%zd exceed Schur dimension %zd",
324  size_type(*itr), N);
325  buf_vals[*itr] = 0.0;
326  }
327 
328  // second, fetch C into the buffer
329  const size_type pi = p[m + i];
330  auto a_val_itr = A.val_cbegin(pi);
331  const boost_scalar_type s_pi = s[pi]; // row scaling
332  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
333  itr != last; ++itr, ++a_val_itr) {
334  const size_type A_idx = *itr;
335  const size_type inv_q = q.inv(A_idx);
336  // load and scale
337  if (inv_q >= m) buf_vals[inv_q - m] = s_pi * *a_val_itr * t[A_idx];
338  }
339 
340  // third, compute -L_E*D*U_F
341  auto L_last = L_E.col_ind_cend(i);
342  auto L_v_itr = L_E.val_cbegin(i);
343  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last;
344  ++L_itr, ++L_v_itr) {
345  const auto idx = *L_itr;
346  const boost_value_type ld = *L_v_itr * d[idx];
347  // get the U iter
348  auto U_last = U_F.col_ind_cend(idx);
349  auto U_v_itr = U_F.val_cbegin(idx);
350  for (auto U_itr = U_F.col_ind_cbegin(idx); U_itr != U_last;
351  ++U_itr, ++U_v_itr)
352  buf_vals[*U_itr] -= ld * *U_v_itr;
353  }
354 
355  // finally, assign values directly to SC
356  auto sc_val_itr = SC.val_begin(i);
357  for (auto itr = sc_ind_first, ind_last = SC.col_ind_cend(i);
358  itr != ind_last; ++itr, ++sc_val_itr)
359  *sc_val_itr = buf_vals[*itr];
360  }
361 }

404 /// \brief drop \a L_E and \a U_F
405 /// \tparam CrsType crs storage type, see \ref CRS
406 /// \tparam CcsType ccs storage type, see \ref CCS
407 /// \tparam BufArray value buffer type
408 /// \tparam IntBufArray integer buffer type
409 /// \param[in] ref_indptr_L reference starting array for \a L_E
410 /// \param[in] alpha_L drop space limiter threshold for \a L_E
411 /// \param[in] ref_indptr_U reference starting array for \a U_F
412 /// \param[in] alpha_U drop space limiter threshold for \a U_F
413 /// \param[in,out] L_E in and out puts of L offset
414 /// \param[in,out] U_F in and out puts of U offset
415 /// \param[out] buf_L work space for \a L_E
416 /// \param[out] ibuf_L integer work space for \a L_E
417 /// \param[out] buf_U work space for \a U_F
418 /// \param[out] ibuf_U workspace for \a U_F
419 /// \param[in] nthreads total threads used, must be at least 1, not alerted
420 /// \note buffers can be got from Crout work spaces
421 template <class CrsType, class CcsType, class BufArray, class IntBufArray>
422 inline void drop_L_E_and_U_F(const typename CrsType::iparray_type &ref_indptr_L,
423  const double alpha_L,
424  const typename CcsType::iparray_type &ref_indptr_U,
425  const double alpha_U, CrsType &L_E, CcsType &U_F,
426  BufArray &buf_L, IntBufArray &ibuf_L,
427  BufArray &buf_U, IntBufArray &ibuf_U,
428  const int nthreads) {
429  using size_type = typename CrsType::size_type;
430  using index_type = typename CrsType::index_type;
431 
432  const size_type n = L_E.nrows();
433  hif_assert(n == U_F.ncols(), "mismatched sizes between L_E and U_F");
434  if (nthreads == 1 || n <= 200u) {
435  drop_L_E(ref_indptr_L, alpha_L, L_E, buf_L, ibuf_L);
436  drop_U_F(ref_indptr_U, alpha_U, U_F, buf_U, ibuf_U);
437  return;
438  }
439 
440  // parallel
441  hif_assert(nthreads > 1, "thread number %d must be no smaller than 1",
442  nthreads);
443 
444  if (nthreads <= 3) {
445 #pragma omp parallel num_threads(2) default(shared)
446  do {
447  const int thread = get_thread();
448  if (thread)
449  drop_L_E(ref_indptr_L, alpha_L, L_E, buf_L, ibuf_L);
450  else
451  drop_U_F(ref_indptr_U, alpha_U, U_F, buf_U, ibuf_U);
452  } while (false); // end of parallel region for 2
453  return;
454  }
455 
456  std::vector<BufArray> bufs_(nthreads - 2);
457  for (int i = 0; i < nthreads - 2; ++i) bufs_[i].resize(L_E.ncols());
458  auto &gaps_L = ibuf_L, &gaps_U = ibuf_U;
459 #pragma omp parallel num_threads(nthreads) default(shared)
460  do {
461  // determine thread id and work of partition
462  const int thread = get_thread();
463  const auto part = uniform_partition(n, nthreads, thread);
464  const size_type start_idx = part.first, poe_idx = part.second;
465 
466  // create additional work buffer
467  // BufArray buf_;
468  // if (thread > 1) buf_.resize(L_E.ncols());
469  BufArray &buf = thread < 2 ? (thread ? buf_U : buf_L) : bufs_[thread - 2];
470 
471  // actual loop begins
472  for (size_type i(start_idx); i < poe_idx; ++i) {
473  // L_E part
474  const size_type sz_thres_L =
475  std::ceil(alpha_L * (ref_indptr_L[i + 1] - ref_indptr_L[i]));
476  if (sz_thres_L >= L_E.nnz_in_row(i))
477  gaps_L[i] = 0;
478  else {
479  gaps_L[i] = L_E.nnz_in_row(i) - sz_thres_L;
480  // fetch values to buffer
481  const size_type first = L_E.row_start()[i],
482  last = L_E.row_start()[i + 1];
483  for (size_type j(first); j < last; ++j)
484  buf[L_E.col_ind()[j]] = L_E.vals()[j];
485  // quickselect
486  std::nth_element(L_E.col_ind_begin(i),
487  L_E.col_ind().begin() + first + sz_thres_L - 1,
488  L_E.col_ind_end(i),
489  [&](const index_type ii, const index_type jj) {
490  return std::abs(buf[ii]) > std::abs(buf[jj]);
491  });
492  // fetch back
493  for (size_type j(first); j < first + sz_thres_L; ++j)
494  L_E.vals()[j] = buf[L_E.col_ind()[j]];
495  }
496 
497  // U_F part
498  const size_type sz_thres_U =
499  std::ceil(alpha_U * (ref_indptr_U[i + 1] - ref_indptr_U[i]));
500  if (sz_thres_U >= U_F.nnz_in_col(i))
501  gaps_U[i] = 0;
502  else {
503  gaps_U[i] = U_F.nnz_in_col(i) - sz_thres_U;
504  // fetch values to buffer
505  const size_type first = U_F.col_start()[i],
506  last = U_F.col_start()[i + 1];
507  for (size_type j(first); j < last; ++j)
508  buf[U_F.row_ind()[j]] = U_F.vals()[j];
509  // quickselect
510  std::nth_element(U_F.row_ind_begin(i),
511  U_F.row_ind().begin() + first + sz_thres_U - 1,
512  U_F.row_ind_end(i),
513  [&](const index_type ii, const index_type jj) {
514  return std::abs(buf[ii]) > std::abs(buf[jj]);
515  });
516  // fetch back
517  for (size_type j(first); j < first + sz_thres_U; ++j)
518  U_F.vals()[j] = buf[U_F.row_ind()[j]];
519  }
520  }
521 
522 #pragma omp barrier
523 
524  // compress
525  if (thread == 0) {
526  // L part
527  auto i_itr = L_E.col_ind().begin();
528  auto v_itr = L_E.vals().begin();
529  auto prev = L_E.row_start()[0];
530  for (size_type i(0); i < n; ++i) {
531  const size_type first = prev, last = L_E.row_start()[i + 1];
532  auto itr_bak = i_itr;
533  i_itr = std::copy(L_E.col_ind().cbegin() + first,
534  L_E.col_ind().cbegin() + last - gaps_L[i], i_itr);
535  v_itr = std::copy(L_E.vals().cbegin() + first,
536  L_E.vals().cbegin() + last - gaps_L[i], v_itr);
537  prev = L_E.row_start()[i + 1];
538  L_E.row_start()[i + 1] = L_E.row_start()[i] + (i_itr - itr_bak);
539  }
540  // resize nnz arrays
541  L_E.col_ind().resize(L_E.nnz());
542  L_E.vals().resize(L_E.nnz());
543  } else if (thread == 1) {
544  // U part
545  auto i_itr = U_F.row_ind().begin();
546  auto v_itr = U_F.vals().begin();
547  auto prev = U_F.col_start()[0];
548  for (size_type i(0); i < n; ++i) {
549  const size_type first = prev, last = U_F.col_start()[i + 1];
550  auto itr_bak = i_itr;
551  i_itr = std::copy(U_F.row_ind().cbegin() + first,
552  U_F.row_ind().cbegin() + last - gaps_U[i], i_itr);
553  v_itr = std::copy(U_F.vals().cbegin() + first,
554  U_F.vals().cbegin() + last - gaps_U[i], v_itr);
555  prev = U_F.col_start()[i + 1];
556  U_F.col_start()[i + 1] = U_F.col_start()[i] + (i_itr - itr_bak);
557  }
558  // resize nnz arrays
559  U_F.row_ind().resize(U_F.nnz());
560  U_F.vals().resize(U_F.nnz());
561  }
562  } while (false); // end of parallel region
563 }

565 /// \brief compute the simple version of Schur complement in MT setting
566 /// \tparam ScaleArray row/column scaling vector type, see \ref Array
567 /// \tparam CrsType crs matrix used for input, see \ref CRS
568 /// \tparam PermType permutation vector type, see \ref BiPermMatrix
569 /// \tparam DiagArray diagonal vector type, see \ref Array
570 /// \tparam SpVecType sparse vector buffer type, see \ref SparseVector
571 /// \param[in] s row scaling vector
572 /// \param[in] A input crs matrix
573 /// \param[in] t column scaling vector
574 /// \param[in] p row permutation vector
575 /// \param[in] q column permutation vector
576 /// \param[in] m leading block size of current level
577 /// \param[in] L_E lower part after \ref Crout update
578 /// \param[in] d diagonal entry that contains the leading block
579 /// \param[in] U_F upper part after \ref Crout update
580 /// \param[out] buf0 work space, can use directly passed in as from Crout bufs
581 /// \param[in] nthreads number of threads used, must be at least 1, not alerted
582 /// \return simple version of Schur complement
583 template <class ScaleArray, class CrsType, class PermType, class DiagArray,
584  class SpVecType>
585 inline CrsType compute_Schur_simple(const ScaleArray &s, const CrsType &A,
586  const ScaleArray &t, const PermType &p,
587  const PermType & q,
588  const typename CrsType::size_type m,
589  const CrsType &L_E, const DiagArray &d,
590  const CrsType &U_F, SpVecType &buf0,
591  const int nthreads) {
592  if (nthreads == 1)
593  return hif::compute_Schur_simple(s, A, t, p, q, m, L_E, d, U_F, buf0);
594 
595  hif_assert(nthreads > 1, "thread number %d must be no smaller than 1",
596  nthreads);
597 
598  using size_type = typename CrsType::size_type;
599  using value_type = typename CrsType::value_type;
600  using boost_value_type = typename ValueTypeMixedTrait<value_type>::boost_type;
601  using boost_sparse_vec_type =
602  SparseVector<boost_value_type, typename CrsType::index_type>;
603  using boost_scalar_type =
604  typename ValueTypeTrait<boost_value_type>::value_type;
605 
606  const size_type n = A.nrows();
607  if (m == n) return CrsType();
608 
609  const size_type N(n - m);
610  // if running in parallel is efficient?
611  if (N <= 200u || L_E.nnz() <= 10000u || U_F.nnz() <= 10000u)
612  return hif::compute_Schur_simple(s, A, t, p, q, m, L_E, d, U_F, buf0);
613 
614  // parallel part
615  CrsType SC;
616  SC.resize(N, N);
617  auto &row_start = SC.row_start();
618  row_start.resize(N + 1);
619  row_start.front() = 0;
620  std::vector<boost_sparse_vec_type> bufs(nthreads);
621  for (int i = 0; i < nthreads; ++i) bufs[i].resize(N);
622 #pragma omp parallel num_threads(nthreads) default(shared)
623  do {
624  const int thread = get_thread();
625  const auto part = uniform_partition(N, nthreads, thread);
626  const size_type start_idx = part.first, poe_idx = part.second;
627 
628  // create local sparse buffer
629  // boost_sparse_vec_type buf(N);
630  auto &buf = bufs[thread];
631  // SpVecType buf_;
632  // if (thread) buf_.resize(N);
633  // SpVecType &buf = thread ? buf_ : buf0;
634  size_type tag(n);
635 
636  // step 1, determine the overall nnz
637  for (size_type i(start_idx); i < poe_idx; ++i, ++tag) {
638  buf.reset_counter(); // reset buffer counter
639  auto L_last = L_E.col_ind_cend(i);
640  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last; ++L_itr) {
641  auto U_last = U_F.col_ind_cend(*L_itr);
642  for (auto U_itr = U_F.col_ind_cbegin(*L_itr); U_itr != U_last; ++U_itr)
643  buf.push_back(*U_itr, tag);
644  }
645  // now add the C contribution
646  const size_type pi = p[m + i];
647  // since A is permutated, we must loop through all entries in this row
648  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
649  itr != last; ++itr) {
650  const size_type inv_q = q.inv(*itr);
651  if (inv_q >= m) buf.push_back(inv_q - m, tag);
652  }
653  // assemble row_start
654  // NOTE we only get the local value here
655  row_start[i + 1] = buf.size();
656  }
657 
658  // bottleneck
659 
660 #pragma omp barrier
661 #pragma omp master
662  do {
663  for (size_type i(0); i < N; ++i) row_start[i + 1] += row_start[i];
664  if (row_start[N]) {
665  SC.reserve(row_start[N]);
666  SC.col_ind().resize(row_start[N]);
667  SC.vals().resize(row_start[N]);
668  }
669  } while (false);
670 #pragma omp barrier
671  if (row_start[N] == 0) break;
672 
673  // step 2, fill in the indices
674  for (size_type i(start_idx); i < poe_idx; ++i, ++tag) {
675  buf.reset_counter(); // reset buffer counter
676  auto L_last = L_E.col_ind_cend(i);
677  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last; ++L_itr) {
678  // get the U iter
679  auto U_last = U_F.col_ind_cend(*L_itr);
680  for (auto U_itr = U_F.col_ind_cbegin(*L_itr); U_itr != U_last; ++U_itr)
681  buf.push_back(*U_itr, tag);
682  }
683  // now add the C contribution
684  const size_type pi = p[m + i];
685  // since A is permutated, we must loop through all entries in this row
686  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
687  itr != last; ++itr) {
688  const size_type inv_q = q.inv(*itr);
689  if (inv_q >= m) buf.push_back(inv_q - m, tag);
690  }
691  // we need the ensure that the indices are sorted, because it will be the
692  // input for next level
693  buf.sort_indices();
694  std::copy_n(buf.inds().cbegin(), buf.size(), SC.col_ind_begin(i));
695  }
696 
697  // step 3, fill in the numerical values
698  auto &buf_vals = buf.vals();
699  for (size_type i(start_idx); i < poe_idx; ++i) {
700  // first, assign values to zero
701  auto sc_ind_first = SC.col_ind_cbegin(i);
702  for (auto itr = sc_ind_first, ind_last = SC.col_ind_cend(i);
703  itr != ind_last; ++itr)
704  buf_vals[*itr] = 0.0;
705 
706  // second, fetch C into the buffer
707  const size_type pi = p[m + i];
708  auto a_val_itr = A.val_cbegin(pi);
709  const boost_scalar_type s_pi = s[pi]; // row scaling
710  for (auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
711  itr != last; ++itr, ++a_val_itr) {
712  const size_type A_idx = *itr;
713  const size_type inv_q = q.inv(A_idx);
714  // load and scale
715  if (inv_q >= m) buf_vals[inv_q - m] = s_pi * *a_val_itr * t[A_idx];
716  }
717 
718  // third, compute -L_E*D*U_F
719  auto L_last = L_E.col_ind_cend(i);
720  auto L_v_itr = L_E.val_cbegin(i);
721  for (auto L_itr = L_E.col_ind_cbegin(i); L_itr != L_last;
722  ++L_itr, ++L_v_itr) {
723  const auto idx = *L_itr;
724  const boost_value_type ld = *L_v_itr * d[idx];
725  // get the U iter
726  auto U_last = U_F.col_ind_cend(idx);
727  auto U_v_itr = U_F.val_cbegin(idx);
728  for (auto U_itr = U_F.col_ind_cbegin(idx); U_itr != U_last;
729  ++U_itr, ++U_v_itr)
730  buf_vals[*U_itr] -= ld * *U_v_itr;
731  }
732 
733  // finally, assign values directly to SC
734  auto sc_val_itr = SC.val_begin(i);
735  for (auto itr = sc_ind_first, ind_last = SC.col_ind_cend(i);
736  itr != ind_last; ++itr, ++sc_val_itr)
737  *sc_val_itr = buf_vals[*itr];
738  }
739  } while (false); // end of parallel region
740 
741  return SC;
742 }
