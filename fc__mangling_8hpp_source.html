34 // determine Fortran name mangling
35 // 1: lower
36 // 2: lower_
37 // 3: lower__
38 // 4: UPPER
39 // 5: UPPER_
40 // 6: UPPER__
41 
42 #ifndef HIF_FC
43 /*
44  * if neither upper nor lower is defined, we first check some defines from
45  * common blas and lapack implementations
46  */
47 # ifdef F77_GLOBAL /* cblas */
48 # define HIF_FC F77_GLOBAL
49 # elif defined(LAPACK_GLOBAL) /* lapacke */
50 # define HIF_FC LAPACK_GLOBAL
51 # elif defined(OPENBLAS_NEEDBUNDERSCORE) /* openblas configuration */
52 # define HIF_FC(l, U) l##_
53 # else
54 /* fallback to lower case with single _ */
55 # define HIF_FC(l, U) l##_
56 # endif
57 
58 #elif HIF_FC == 1
59 # undef HIF_FC
60 # define HIF_FC(__l, __U) __l
61 #elif HIF_FC == 2
62 # undef HIF_FC
63 # define HIF_FC(__l, __U) __l##_
64 #elif HIF_FC == 3
65 # undef HIF_FC
66 # define HIF_FC(__l, __U) __l##__
67 #elif HIF_FC == 4
68 # undef HIF_FC
69 # define HIF_FC(__l, __U) __U
70 #elif HIF_FC == 5
71 # undef HIF_FC
72 # define HIF_FC(__l, __U) __U##_
73 #elif HIF_FC == 6
74 # undef HIF_FC
75 # define HIF_FC(__l, __U) __U##__
76 #else
77 # error "Unknown HIF_FC option, must be in (1,2,3,4,5,6)"
78 #endif
79 
80 #endif /* _HIF_UTILS_FCMANGLING_HPP */
