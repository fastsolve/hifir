49 /// \brief fix poorly scaled row and column weights due to MC64
50 /// \tparam PermType permutation matrix, see \ref BiPermMatrix
51 /// \tparam ScalingArray scaling array type, see \ref Array
52 /// \param[in] m0 leading block size
53 /// \param[in] level current factorization level
54 /// \param[in] p row permutation matrix
55 /// \param[in] q column permutation matrix
56 /// \param[in,out] s row scaling
57 /// \param[in,out] t column scaling
58 /// \param[in] beta (optional) threshold for determining bad scaling factors
59 template <class PermType, class ScalingArray>
60 inline void fix_poor_scaling(const typename ScalingArray::size_type m0,
61  const typename ScalingArray::size_type level,
62  const PermType &p, const PermType &q,
63  ScalingArray &s, ScalingArray &t,
64  const double beta = 1e3) {
65  using size_type = typename ScalingArray::size_type;
66 
67  const double beta0 = beta < 0.0 ? 1e3 : beta;
68  // Fix poorly scaled row and column scaling due to MC64
69  // We consider a combo of row (s_i) and column (t_i) scaling is bad if
70  // beta*min(s_i,t_i)<max(s_i,t_i). We fix it by setting s_i=t_i=sqrt(s_i*t_i)
71  // TODO: we need to fine tune beta. 100 for now.
72  if (level > 1u && beta0 > 1.0) // starting from 2nd level only
73  for (size_type i(0); i < m0; ++i)
74  if (std::min(s[p[i]], t[q[i]]) * beta0 < std::max(s[p[i]], t[q[i]]))
75  s[p[i]] = t[q[i]] = std::sqrt(s[p[i]] * t[q[i]]);
76 }

78 /// \brief defer any zero diags to the end
79 /// \tparam IsSymm if \a true, then assume a symmetric leading block
80 /// \tparam CcsType ccs storage for input matrix
81 /// \tparam CrsType crs storage for the input matrix
82 /// \tparam PermType permutation matrix, see \ref BiPermMatrix
83 /// \tparam BufType buffer used to stored deferred entries
84 /// \param[in] A input matrix
85 /// \param[in] A_crs input matrix of CRS storage
86 /// \param[in] m0 initial leading block size
87 /// \param[in,out] p row permutation matrix
88 /// \param[in,out] q column permutation matrix
89 /// \param[out] work_p workspace
90 /// \param[out] work_q workspace
91 /// \return actual leading block with no zero entries, <= \a m0
92 ///
93 /// For asymmetric cases, where the whole matrix is stored, thus we need binary
94 /// search to locate the diagonal entries. For symmetric cases, on the other
95 /// side, since only the \b lower part is stored, we just need to test the first
96 /// entry of each column.
97 template <bool IsSymm, class CcsType, class CrsType, class PermType,
98  class BufType>
99 inline typename CcsType::size_type defer_tiny_diags(
100  const CcsType &A, const CrsType &A_crs,
101  const typename CcsType::size_type m0, PermType &p, PermType &q,
102  BufType &work_p, BufType &work_q) {
103  using value_type = typename CcsType::value_type;
104  using size_type = typename CcsType::size_type;
105  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
106  constexpr static scalar_type EPS = Const<scalar_type>::EPS;
107 
108  // lambda for computing local maximum magnitude for a given row *and* column
109  const auto compute_max_mag = [&](const size_type i) -> scalar_type {
110  scalar_type v(0);
111  if (A_crs.nnz_in_primary(p[i]))
112  v = std::max(v, std::abs(*std::max_element(
113  A_crs.val_cbegin(p[i]), A_crs.val_cend(p[i]),
114  [](const value_type a, const value_type b) -> bool {
115  return std::abs(a) < std::abs(b);
116  })));
117  if (A.nnz_in_primary(q[i]))
118  v = std::max(v, std::abs(*std::max_element(
119  A.val_cbegin(q[i]), A.val_cend(q[i]),
120  [](const value_type a, const value_type b) -> bool {
121  return std::abs(a) < std::abs(b);
122  })));
123  if (v == Const<scalar_type>::ZERO) v = scalar_type(1);
124  return v;
125  };
126 
127  // lambda for determining small diagonals and perform statical deferring
128  const auto is_good_diag = [&](const size_type col) -> bool {
129  const auto q_col = q[col];
130  if (IsSymm) {
131  // NOTE that since we only store the lower part, and due to the the
132  // fact of symmetric pivoting, if the original is a saddle point, then
133  // it is still a saddle entry in the permutated system. This makes
134  // checking for invalid diagonal efficient for symmetric case
135  hif_assert(p[col] == q_col, "fatal");
136  auto itr = A.row_ind_cbegin(q_col);
137  if (itr == A.row_ind_cend(q_col)) return false;
138  if (*itr != q_col) return false;
139  if (std::abs(*A.val_cbegin(q_col)) <= compute_max_mag(col) * EPS)
140  return false;
141  } else {
142  const auto p_diag = p[col];
143  auto info =
144  find_sorted(A.row_ind_cbegin(q_col), A.row_ind_cend(q_col), p_diag);
145  if (!info.first) return false;
146  // test numerical value
147  if (std::abs(
148  *(A.vals().cbegin() + (info.second - A.row_ind().cbegin()))) <=
149  compute_max_mag(col) * EPS)
150  return false;
151  }
152  return true;
153  };
154 
155  size_type m = m0;
156  size_type deferrals(0);
157  for (size_type i(0); i < m; ++i) {
158  // check deferring
159  while (!is_good_diag(i + deferrals)) {
160  --m;
161  work_p[deferrals] = p[i + deferrals];
162  work_q[deferrals] = q[i + deferrals];
163  ++deferrals;
164  if (i + deferrals >= m0) {
165  m = i;
166  break;
167  }
168  }
169  if (m == i) break;
170  // compress
171  p[i] = p[i + deferrals];
172  q[i] = q[i + deferrals];
173  }
174  if (deferrals) {
175  size_type j(0);
176  for (size_type i(m); i < m0; ++i, ++j) {
177  p[i] = work_p[j];
178  q[i] = work_q[j];
179  }
180  }
181 
182  return m;
183 }

185 /// \brief permute the matrix for asymmetric cases
186 /// \tparam CcsType ccs storage for intermidiate matrix after matching
187 /// \tparam CrsType crs storage, see \ref CRS
188 /// \tparam PermType permutation matrix, see \ref BiPermMatrix
189 /// \param[in] A input matrix after performing matching
190 /// \param[in] A_crs input matrix with crs type
191 /// \param[in] m leading block size that must no larger than size(A)
192 /// \param[in] p row permutation matrix
193 /// \param[in] q column permutation matrix
194 /// \param[in] apat if \a true, then compute \f$\mathbf{A}_m^T+\mathbf{A}_m\f$
195 /// default is \a false
196 /// \note It's worth noting that \a p must be bi-directional mapping
197 /// \note If \a apat is \a true, then \a q must be bi-directional mapping
198 template <class CcsType, class CrsType, class PermType>
199 inline CcsType compute_leading_block(const CcsType &A, const CrsType &A_crs,
200  const typename CcsType::size_type m,
201  const PermType &p, const PermType &q,
202  const bool apat = false) {
203  using size_type = typename CcsType::size_type;
204  using index_type = typename CcsType::index_type;
205 
206  CcsType B(m, m);
207  auto & col_start = B.col_start();
208  col_start.resize(m + 1);
209  hif_error_if(col_start.status() == DATA_UNDEF, "memory allocation failed");
210  col_start.front() = 0; // zero based
211 
212  if (!apat) {
213  // determine nnz
214  for (size_type col = 0u; col < m; ++col) {
215  const auto q_col = q[col];
216  col_start[col + 1] =
217  col_start[col] +
218  std::count_if(A.row_ind_cbegin(q_col), A.row_ind_cend(q_col),
219  [&](decltype(q_col) i) {
220  return static_cast<size_type>(p.inv(i)) < m;
221  });
222  }
223  // for (size_type i = 0u; i < m; ++i) col_start[i + 1] += col_start[i];
224 
225  // allocate storage
226  auto &row_ind = B.row_ind();
227  // NOTE we only indices for reordering step
228  row_ind.resize(col_start[m]);
229  hif_error_if(row_ind.status() == DATA_UNDEF, "memory allocation failed");
230  auto itr = row_ind.begin();
231 
232  // assemble nnz arrays
233  for (size_type col = 0u; col < m; ++col) {
234  const auto q_col = q[col];
235  auto A_v_itr = A.val_cbegin(q_col);
236  for (auto A_itr = A.row_ind_cbegin(q_col), last = A.row_ind_cend(q_col);
237  A_itr != last; ++A_itr, ++A_v_itr) {
238  const size_type p_inv = p.inv(*A_itr);
239  if (p_inv < m) *itr++ = p_inv;
240  }
241  // sort indices
242  std::sort(B.row_ind_begin(col), itr);
243  }
244  } else {
245  // form nonzero pattern of A+A^T
246 
247  // create mask
248  std::vector<bool> masks(m, false);
249  // first determine nnz
250  do {
251  typename CcsType::iarray_type buf(m);
252  hif_error_if(buf.status() == DATA_UNDEF, "memory allocation failed");
253  for (size_type i(0); i < m; ++i) {
254  size_type nnz(0);
255  // first loop through col
256  const auto col = q[i];
257  for (auto itr = A.row_ind_cbegin(col), last = A.row_ind_cend(col);
258  itr != last; ++itr) {
259  const size_type ind = p.inv(*itr);
260  if (ind < m) {
261  masks[ind] = true;
262  buf[nnz++] = ind;
263  }
264  }
265  // then loop through row, note that we only count indices are not been
266  // counted yet
267  const auto row = p[i];
268  for (auto itr = A_crs.col_ind_cbegin(row),
269  last = A_crs.col_ind_cend(row);
270  itr != last; ++itr) {
271  const size_type ind = q.inv(*itr);
272  if (!masks[ind] && ind < m) {
273  masks[ind] = true;
274  buf[nnz++] = ind;
275  }
276  }
277  col_start[i + 1] = col_start[i] + nnz;
278  // reset flags
279  std::for_each(buf.cbegin(), buf.cbegin() + nnz,
280  [&](const index_type j) { masks[j] = false; });
281  }
282  } while (false); // free buf here
283 
284  // then, form indices
285  // allocate storage
286  auto &row_ind = B.row_ind();
287  // NOTE we only indices for reordering step
288  row_ind.resize(col_start[m]);
289  hif_error_if(row_ind.status() == DATA_UNDEF, "memory allocation failed");
290 
291  auto i_itr = row_ind.begin();
292  for (size_type i(0); i < m; ++i) {
293  const auto col = q[i];
294  for (auto itr = A.row_ind_cbegin(col), last = A.row_ind_cend(col);
295  itr != last; ++itr) {
296  const size_type ind = p.inv(*itr);
297  if (ind < m) {
298  *i_itr++ = ind;
299  masks[ind] = true;
300  }
301  }
302  const auto row = p[i];
303  for (auto itr = A_crs.col_ind_cbegin(row), last = A_crs.col_ind_cend(row);
304  itr != last; ++itr) {
305  const size_type ind = q.inv(*itr);
306  if (!masks[ind] && ind < m) {
307  masks[ind] = true;
308  *i_itr++ = ind;
309  }
310  }
311  // sort indices
312  // TODO do we need this in rcm?
313  std::sort(B.row_ind_begin(i), i_itr);
314  // reset flags
315  std::for_each(B.row_ind_cbegin(i), B.row_ind_cend(i),
316  [&](const index_type j) { masks[j] = false; });
317  }
318  }
319 
320  return B;
321 }

329 /// \brief compute the matching for preprocessing
330 /// \tparam IsSymm if \a true, then assume a symmetric leading block
331 /// \tparam CcsType ccs storage for intermidiate matrix after matching
332 /// \tparam ScalingArray scaling array for row and column, see \ref Array
333 /// \tparam PermType permutation matrix, see \ref BiPermMatrix
334 /// \param[in] A input matrix in \ref CCS order
335 /// \param[in] A_crs the \ref CRS version of \a A
336 /// \param[in] m0 leading block size
337 /// \param[in] level current factorization level
338 /// \param[in] opts control parameters, see \ref Options
339 /// \param[out] s row scaling vector
340 /// \param[out] t column scaling vector
341 /// \param[out] p row permutation vector
342 /// \param[out] q column permutation vector
343 /// \return A \a pair of \ref CCS matrix in \b C-index and the actual leading
344 /// block size, which is no larger than \a m0.
345 /// \ingroup pre
346 template <bool IsSymm, class CcsType, class CrsType, class ScalingArray,
347  class PermType>
348 inline std::pair<CcsType, typename CcsType::size_type> do_maching(
349  const CcsType &A, const CrsType &A_crs,
350  const typename CcsType::size_type m0,
351  const typename CcsType::size_type level, const Options &opts,
352  ScalingArray &s, ScalingArray &t, PermType &p, PermType &q) {
353  static_assert(!CcsType::ROW_MAJOR, "input must be CCS type");
354  static_assert(CrsType::ROW_MAJOR, "input A_crs must be CRS type");
355  using value_type = typename CcsType::value_type;
356  using index_type = typename CcsType::index_type;
357  using indptr_type = typename CcsType::indptr_type;
358  using return_type = CcsType;
359  using size_type = typename CcsType::size_type;
360 
361  const size_type M = A.nrows(), N = A.ncols();
362  p.resize(M);
363  hif_error_if(p().status() == DATA_UNDEF || p.inv().status() == DATA_UNDEF,
364  "memory allocation failed for p");
365  q.resize(N);
366  hif_error_if(q().status() == DATA_UNDEF || q.inv().status() == DATA_UNDEF,
367  "memory allocation failed for q");
368  s.resize(M);
369  hif_error_if(s.status() == DATA_UNDEF, "memory allocation failed for s");
370  t.resize(N);
371  hif_error_if(s.status() == DATA_UNDEF, "memory allocation failed for t");
372 
373  const bool timing = hif_verbose(PRE_TIME, opts);
374  const bool compute_perm = opts.reorder != REORDER_OFF;
375 
376  CrsType B;
377  if (m0 == M) {
378  // NOTE, A_crs is the input, if C index order, then the indices will be
379  // temporarily shifted to Fortran order!
380  if (!opts.pre_scale) {
381  if (level > 1u)
382  B = CrsType(A_crs); // shallow!
383  else {
384  // NOTE this is user-data, we make copy of indices
385  B.resize(M, M);
386  B.vals() = A_crs.vals();
387  // the following are deep copies
388  B.row_start() = typename CrsType::iparray_type(A_crs.row_start(), true);
389  B.col_ind() = typename CrsType::iarray_type(A_crs.col_ind(), true);
390  }
391  } else {
392  B.resize(M, M);
393  if (level > 1u) {
394  B.row_start() = A_crs.row_start();
395  B.col_ind() = A_crs.col_ind();
396  } else {
397  // the following are deep copies
398  B.row_start() = typename CrsType::iparray_type(A_crs.row_start(), true);
399  B.col_ind() = typename CrsType::iarray_type(A_crs.col_ind(), true);
400  }
401  B.vals().resize(A_crs.nnz());
402  hif_error_if(B.vals().status() == DATA_UNDEF, "memory allocation failed");
403  std::copy(A_crs.vals().cbegin(), A_crs.vals().cend(), B.vals().begin());
404  }
405  } else
406  B = A_crs.extract_leading(m0); // for explicit leading block do copy
407 
408  using eq_kernel = EqlDriver<value_type, index_type, indptr_type>;
409  do {
410  DefaultTimer timer;
411  timer.start();
412  eq_kernel::template do_matching<IsSymm>(opts.verbose, B, p(), q(), s, t,
413  opts.pre_scale);
414  timer.finish();
415  if (timing) hif_info("Equilibrator took %gs.", (double)timer.time());
416  } while (false);
417  // fill identity mapping and add one to scaling vectors for offsets, if any
418  for (size_type i = m0; i < M; ++i) {
419  p[i] = i;
420  s[i] = 1;
421  }
422  for (size_type i = m0; i < N; ++i) {
423  q[i] = i;
424  t[i] = 1;
425  }
426 
427  // revert indices
428  if (m0 == M && level > 1u) {
429  for (auto &v : B.row_start()) --v;
430  for (auto &v : B.col_ind()) --v;
431  }
432 
433  // fix potentially poorly scaled row and column weights
434  internal::fix_poor_scaling(m0, level, p, q, s, t, opts.beta);
435 
436  // then determine tiny diags
437  // using the inverse mappings are buffers since we don't need them for now
438  auto & p_buf = p.inv(), &q_buf = q.inv();
439  const size_type m =
440  internal::defer_tiny_diags<false>(A, A_crs, m0, p, q, p_buf, q_buf);
441  return_type BB;
442  if (compute_perm && m) {
443  p.build_inv();
444  // to see if we need to build A^T+A pattern
445  const bool build_apat = !IsSymm && opts.reorder == REORDER_RCM;
446  if (build_apat) q.build_inv();
447  BB = internal::compute_leading_block(A, A_crs, m, p, q, build_apat);
448  }
449  return std::make_pair(BB, m);
450 }
