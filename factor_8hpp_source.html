75 /// \brief adjust parameters based on levels
76 /// \param[in] opts control parameters, i.e. Options
77 /// \param[in] lvl levels
78 /// \return refined parameters
79 inline std::tuple<double, double, double, double, double, double>
80 determine_fac_pars(const Options &opts, const int lvl) {
81  double kappa_d, kappa, tau_U, tau_L, alpha_L, alpha_U;
82  if (opts.rf_par) {
83  const int fac = std::min(lvl, 2);
84  const double fac2 = 1. / std::min(10.0, std::pow(10.0, lvl - 1));
85  kappa_d = std::max(2.0, std::pow(opts.kappa_d, 1. / fac));
86  kappa = std::max(2.0, std::pow(opts.kappa, 1. / fac));
87  tau_U = opts.tau_U * fac2;
88  tau_L = opts.tau_L * fac2;
89  if (lvl > 2) {
90  alpha_L = opts.alpha_L;
91  alpha_U = opts.alpha_U;
92  } else {
93  alpha_L = opts.alpha_L * fac;
94  alpha_U = opts.alpha_U * fac;
95  }
96  } else {
97  kappa_d = opts.kappa_d;
98  kappa = opts.kappa;
99  tau_U = opts.tau_U;
100  tau_L = opts.tau_L;
101  alpha_L = opts.alpha_L;
102  alpha_U = opts.alpha_U;
103  }
104  return std::make_tuple(kappa_d, kappa, tau_U, tau_L, alpha_L, alpha_U);
105 }

107 /// \brief extract permutated diagonal
108 /// \tparam CcsType input ccs matrix, see \ref CCS
109 /// \tparam ScalingType scaling vector type, see \ref Array
110 /// \tparam PermType permutation matrix type, see \ref BiPermMatrix
111 /// \param[in] s row scaling vector
112 /// \param[in] A input matrix in CCS format
113 /// \param[in] t column scaling vector
114 /// \param[in] m leading block size
115 /// \param[in] p row permutation vector
116 /// \param[in] q column permutation vector
117 /// \param[in] m0 actual extrated size, default is \a m
118 /// \return permutated diagonal of \a A
119 ///
120 /// This routine, essentially, is to compute:
121 ///
122 /// \f[
123 /// \mathbf{D}=\left(\mathbf{SAT}\right)_{\mathbf{p}_{1:m},
124 /// \mathbf{q}_{1:m}}
125 /// \f]
126 ///
127 /// This routine is used before \ref Crout update to extract the initial
128 /// diagonal entries.
129 template <class CcsType, class ScalingType, class PermType>
130 inline Array<typename CcsType::value_type> extract_perm_diag(
131  const ScalingType &s, const CcsType A, const ScalingType &t,
132  const typename CcsType::size_type m, const PermType &p, const PermType &q,
133  const typename CcsType::size_type m0 = 0) {
134  using value_type = typename CcsType::value_type;
135  using size_type = typename CcsType::size_type;
136  using array_type = Array<value_type>;
137 
138  hif_error_if(m > A.nrows() || m > A.ncols(), "invalid leading block size %zd",
139  m);
140 
141  array_type diag(m);
142  hif_error_if(diag.status() == DATA_UNDEF, "memory allocation failed");
143 
144  auto v_begin = A.vals().cbegin();
145  auto i_begin = A.row_ind().cbegin();
146  const size_type M = m0 ? m0 : m;
147  for (size_type i = 0u; i < M; ++i) {
148  hif_assert((size_type)q[i] < A.ncols(),
149  "permutated index %zd exceeds the col bound", (size_type)q[i]);
150  hif_assert((size_type)p[i] < A.nrows(),
151  "permutated index %zd exceeds the row bound", (size_type)p[i]);
152  // using binary search
153  auto info = find_sorted(A.row_ind_cbegin(q[i]), A.row_ind_cend(q[i]), p[i]);
154  if (info.first)
155  diag[i] = s[p[i]] * *(v_begin + (info.second - i_begin)) * t[q[i]];
156  else {
157  // hif_warning("zero diagonal entry %zd detected!", i);
158  diag[i] = 0;
159  }
160  }
161  return diag;
162 }

164 /// \brief extract the \a E part
165 /// \tparam CrsType input crs matrix, see \ref CRS
166 /// \tparam ScalingType scaling vector type, see \ref Array
167 /// \tparam PermType permutation matrix type, see \ref BiPermMatrix
168 /// \param[in] s row scaling vector
169 /// \param[in] A input matrix in crs format
170 /// \param[in] t column scaling vector
171 /// \param[in] m leading block size (final)
172 /// \param[in] p row permutation vector
173 /// \param[in] q column permutation vector
174 /// \return The \a E part in \ref CCS format
175 /// \sa extract_F
176 ///
177 /// This routine is to extract the \a E part \b after \ref Crout update.
178 /// Essentially, this routine is to perform:
179 ///
180 /// \f[
181 /// \mathbf{E}=\left(\mathbf{SAT}\right)_{\mathbf{p}_{m+1:n},
182 /// \mathbf{q}_{1:m}}
183 /// \f]
184 template <class CrsType, class ScalingType, class PermType>
185 inline typename CrsType::other_type extract_E(
186  const ScalingType &s, const CrsType &A, const ScalingType &t,
187  const typename CrsType::size_type m, const PermType &p, const PermType &q) {
188  // it's efficient to extract E from CRS
189  static_assert(CrsType::ROW_MAJOR, "input A must be CRS!");
190  using ccs_type = typename CrsType::other_type;
191  using size_type = typename CrsType::size_type;
192  using indptr_type = typename CrsType::indptr_type;
193 
194  const size_type n = A.nrows();
195 
196  hif_error_if(m > n || m > A.ncols(),
197  "leading block size %zd should not exceed matrix size", m);
198  const size_type N = n - m;
199  ccs_type E(N, m);
200  if (!N) {
201  // hif_warning("empty E matrix detected!");
202  return E;
203  }
204 
205  auto &col_start = E.col_start();
206  col_start.resize(m + 1);
207  hif_error_if(col_start.status() == DATA_UNDEF, "memory allocation failed");
208  std::fill(col_start.begin(), col_start.end(), 0);
209 
210  for (size_type i = m; i < n; ++i) {
211  for (auto itr = A.col_ind_cbegin(p[i]), last = A.col_ind_cend(p[i]);
212  itr != last; ++itr) {
213  const size_type qi = q.inv(*itr);
214  if (qi < m) ++col_start[qi + 1];
215  }
216  }
217 
218  // accumulate for nnz
219  for (size_type i = 0u; i < m; ++i) col_start[i + 1] += col_start[i];
220 
221  if (!col_start[m]) {
222  // hif_warning(
223  // "exactly zero E, this most likely is a bug! Continue anyway...");
224  return E;
225  }
226 
227  E.reserve(col_start[m]);
228  auto &row_ind = E.row_ind();
229  auto &vals = E.vals();
230 
231  hif_error_if(row_ind.status() == DATA_UNDEF || vals.status() == DATA_UNDEF,
232  "memory allocation failed");
233 
234  row_ind.resize(col_start[m]);
235  vals.resize(col_start[m]);
236 
237  for (size_type i = m; i < n; ++i) {
238  const size_type pi = p[i];
239  auto itr = A.col_ind_cbegin(pi), last = A.col_ind_cend(pi);
240  auto v_itr = A.val_cbegin(pi);
241  const auto si = s[pi];
242  for (; itr != last; ++itr, ++v_itr) {
243  const size_type j = *itr;
244  const size_type qi = q.inv(j);
245  if (qi < m) {
246  row_ind[col_start[qi]] = i - m;
247  vals[col_start[qi]] = si * *v_itr * t[j];
248  ++col_start[qi];
249  }
250  }
251  }
252 
253  // revert
254  indptr_type tmp(0);
255  for (size_type i = 0u; i < m; ++i) std::swap(col_start[i], tmp);
256 
257  return E;
258 }

260 /// \brief extract the \a F part
261 /// \tparam CcsType input ccs matrix, see \ref CCS
262 /// \tparam ScalingType scaling vector type, see \ref Array
263 /// \tparam PermType permutation matrix type, see \ref BiPermMatrix
264 /// \tparam BufferType work space array, can be \ref Array or \a std::vector
265 /// \param[in] s row scaling vector
266 /// \param[in] A input matrix in CCS format
267 /// \param[in] t column scaling vector
268 /// \param[in] m leading block size
269 /// \param[in] p row permutation vector
270 /// \param[in] q column permutation vector
271 /// \param buf work space
272 /// \return The \a F part in ccs storage
273 /// \sa extract_E
274 ///
275 /// Note that unlike extracting the \a E part, this routine takes \ref CCS as
276 /// input, and with the permutation vectors, we need a value buffer space as
277 /// an intermidiate storage to store the values so that is will make sorting
278 /// much easier. The buffer space is a dense array, and can be passed in from
279 /// that of \a l or \a ut (if squared systems).
280 ///
281 /// This routine, essentially, is to compute:
282 ///
283 /// \f[
284 /// \mathbf{F}=\left(\mathbf{SAT}\right)_{\mathbf{p}_{1:m},
285 /// \mathbf{q}_{m+1:n}}
286 /// \f]
287 template <class CcsType, class ScalingType, class PermType, class BufferType>
288 inline CcsType extract_F(const ScalingType &s, const CcsType &A,
289  const ScalingType & t,
290  const typename CcsType::size_type m, const PermType &p,
291  const PermType &q, BufferType &buf) {
292  static_assert(!CcsType::ROW_MAJOR, "input A must be CCS!");
293  using size_type = typename CcsType::size_type;
294  using index_type = typename CcsType::index_type;
295 
296  const size_type n = A.ncols();
297 
298  hif_error_if(m > n || m > A.nrows(),
299  "leading block size %zd should not exceed matrix size", m);
300 
301  const size_type N = n - m;
302  CcsType F(m, N);
303  if (!N) {
304  // hif_warning("empty F matrix detected!");
305  return F;
306  }
307 
308  auto &col_start = F.col_start();
309  col_start.resize(N + 1);
310  hif_error_if(col_start.status() == DATA_UNDEF, "memory allocation failed");
311  col_start.front() = 0;
312 
313  for (size_type i = 0u; i < N; ++i) {
314  const size_type qi = q[i + m];
315  size_type nnz(0);
316  std::for_each(A.row_ind_cbegin(qi), A.row_ind_cend(qi),
317  [&](const index_type i) {
318  if (static_cast<size_type>(p.inv(i)) < m) ++nnz;
319  });
320  col_start[i + 1] = col_start[i] + nnz;
321  }
322 
323  if (!(col_start[N])) {
324  // hif_warning(
325  // "exactly zero F, this most likely is a bug! Continue anyway...");
326  return F;
327  }
328 
329  F.reserve(col_start[N]);
330  auto &row_ind = F.row_ind();
331  auto &vals = F.vals();
332  hif_error_if(row_ind.status() == DATA_UNDEF || vals.status() == DATA_UNDEF,
333  "memory allocation failed");
334 
335  row_ind.resize(col_start[N]);
336  vals.resize(col_start[N]);
337 
338  auto v_itr = vals.begin();
339 
340  for (size_type i = 0u; i < N; ++i) {
341  const size_type qi = q[i + m];
342  auto itr = F.row_ind_begin(i);
343  auto A_itr = A.row_ind_cbegin(qi);
344  auto A_v_itr = A.val_cbegin(qi);
345  const auto ti = t[qi];
346  for (auto A_last = A.row_ind_cend(qi); A_itr != A_last;
347  ++A_itr, ++A_v_itr) {
348  const size_type j = *A_itr;
349  const size_type pi = p.inv(j);
350  if (pi < m) {
351  *itr = pi;
352  ++itr;
353  buf[pi] = s[j] * *A_v_itr * ti;
354  }
355  }
356  std::sort(F.row_ind_begin(i), itr);
357  for (auto itr = F.row_ind_cbegin(i), last = F.row_ind_cend(i); itr != last;
358  ++itr, ++v_itr)
359  *v_itr = buf[*itr];
360  }
361  return F;
362 }

424 /// \brief compress offsets to have a compact L and U
425 /// \tparam L_Type storage for \a L, see \ref CCS
426 /// \tparam U_Type storage for \a U, see \ref CRS
427 /// \tparam IndPtrArray array for storing starting positions, see \ref Array
428 /// \param[in,out] U uncompressed \a U part
429 /// \param[in,out] L uncompressed \a L part
430 /// \param[in] U_start starting positions of the offset of \a U
431 /// \param[in] L_start starting positions of the offset of \a L
432 /// \param[in] m leading block size
433 /// \param[in] dfrs total number of deferrals
434 template <class L_Type, class U_Type, class IndPtrArray>
435 inline void compress_tails(U_Type &U, L_Type &L, const IndPtrArray &U_start,
436  const IndPtrArray & L_start,
437  const typename IndPtrArray::size_type m,
438  const typename IndPtrArray::size_type dfrs) {
439  using size_type = typename IndPtrArray::size_type;
440  using index_type = typename L_Type::index_type;
441 
442  if (dfrs) {
443  const auto comp_index = [=](index_type &j) { j -= dfrs; };
444  auto U_first = U.col_ind().begin(), L_first = L.row_ind().begin();
445  for (size_type i(0); i < m; ++i) {
446  std::for_each(U_first + U_start[i], U.col_ind_end(i), comp_index);
447  std::for_each(L_first + L_start[i], L.row_ind_end(i), comp_index);
448  }
449  }
450 
451  // reshape the secondary axis of the matrices
452  L.resize_nrows(L.nrows() / 2);
453  U.resize_ncols(U.ncols() / 2);
454 
455 #ifdef HIF_DEBUG
456  L.check_validity();
457  U.check_validity();
458 #endif
459 }

491 /// \brief Determine pattern symmetry
492 /// \tparam MatType Matrix type, either \ref CRS or \ref CCS
493 /// \tparam OtherType Opposite of \a MatType, e.g., if \a MatType is CRS, then
494 /// this type is CCS
495 /// \param[in] A Input matrix A
496 /// \param[in] AT Transpose of A
497 /// \return The pattern symmetry ratio
498 template <class MatType, class OtherType>
499 inline double compute_pattern_symm_ratio(const MatType & A,
500  const OtherType &AT) {
501  using value_type = typename MatType::value_type;
502  using index_type = typename MatType::index_type;
503  using size_type = typename MatType::size_type;
504 
505  hif_assert(A.nnz() == AT.nnz(), "They should equal");
506 
507  const auto n = A.nrows();
508  SparseVector<value_type, index_type> work(n);
509  size_type counts(0);
510  for (size_type i(0); i < n; ++i) {
511  // first push all nonzeros of i-th row/column to sparse vector
512  for (auto it = A.ind_cbegin(i); it != A.ind_cend(i); ++it)
513  work.push_back(*it, i);
514  // then determine for B, if we meet an already-registered index, then
515  // we increment pattern symmetry counter
516  for (auto it = AT.ind_cbegin(i); it != AT.ind_cend(i); ++it)
517  if (!work.push_back(*it, i)) ++counts;
518  work.reset_counter();
519  }
520  return (double)counts / A.nnz();
521 }

530 /// \tparam IsSymm if \a true, then assume a symmetric leading block

551 template <bool IsSymm, class CsType, class CroutStreamer, class PrecsType,

553 inline CsType level_factorize(
554  const CsType &A, const typename CsType::size_type m0,
555  const typename CsType::size_type N, const Options &opts,
556  const CroutStreamer &Crout_info, PrecsType &precs, IntArray &row_sizes,
557  IntArray &col_sizes, typename CsType::size_type *stats,
558  const int schur_threads = 1, int *auto_pivot_tag = nullptr) {

569  typedef DenseMatrix<value_type> dense_type;

905 #else
906  (void)
907 #endif

917 #else
918  (void)
919 #endif

1001  step.assign_gap_array(Q, q);
1002  }

1004  // rebuild the inverse mappings

1006  q.build_inv();

1019  internal::compress_tails(U, L, U_start, L_start, m, step.defers());

1046  stats[1] += m ? step.defers() : size_type(0); // dynamic deferrals
1047  stats[2] += info_counter[0]; // diagonal deferrals
1048  stats[3] += info_counter[1]; // conditioning deferrals
1049

1052  stats[5] += info_counter[3] + info_counter[4]; // space-based droppings

1069  "\tmax |kappa_u|=%g\n"
1070  "\tmin |kappa_l|=%g\n"
1071  "\tmax |kappa_l|=%g\n"
1072  "\tmax |d|=%g",

1074  (size_type)info_counter[1], (size_type)info_counter[5],
1075  (size_type)info_counter[3], (size_type)info_counter[6],
1076  (size_type)info_counter[4],
1077  (double)std::abs(
1078  *std::min_element(kappa_ut.cbegin(), kappa_ut.cbegin() + m,
1079  [](const value_type l, const value_type r) {
1080  return std::abs(l) < std::abs(r);
1081  })),
1082  (double)std::abs(
1083  *std::max_element(kappa_ut.cbegin(), kappa_ut.cbegin() + m,
1084  [](const value_type l, const value_type r) {
1085  return std::abs(l) < std::abs(r);

1089  [](const value_type l, const value_type r) {

1091  })),

1095  return std::abs(l) < std::abs(r);
1096  })),
1097  (double)std::abs(
1098  *std::max_element(d.cbegin(), d.cbegin() + m,
1099  [](const value_type l, const value_type r) {

1101  })));
