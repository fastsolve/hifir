46 /// \brief apply approximated mimimum degree ordering
47 /// \tparam IsSymm if \a true, then assume symmetric pattern input
48 /// \tparam CcsType ccs sparse storage, see \ref CCS
49 /// \param[in] B matrix with pattern given
50 /// \param[in] opt options
51 /// \return symmetric permutation so that PAP' satisties AMD property
52 template <bool IsSymm, class CcsType>
53 inline typename CcsType::iparray_type run_amd(const CcsType &B,
54  const Options &opt) {
55  using size_type = typename CcsType::size_type;
56  using indptr_type = typename CcsType::indptr_type;
57  using amd = amd::AMD<indptr_type>;
58 
59  const size_type m = B.nrows();
60  hif_assert(B.nrows() == m, "the leading block size should be size(B)");
61 
62  // reordering
63  double Control[HIF_AMD_CONTROL], Info[AMD_INFO];
64  amd::defaults(Control);
65  Control[HIF_AMD_SYMM_FLAG] = !IsSymm;
66 
67  if (hif_verbose(PRE, opt)) {
68  hif_info("performing AMD reordering");
69  std::stringstream s;
70  amd::control(s, Control);
71  hif_info(s.str().c_str());
72  }
73  Array<indptr_type> P(m);
74  Array<indptr_type> row_ind;
75  if (sizeof(indptr_type) == sizeof(typename CcsType::index_type))
76  row_ind = wrap_const_array(B.row_ind().size(),
77  (const indptr_type *)B.row_ind().data());
78  else {
79  row_ind.resize(B.row_ind().size());
80  std::copy(B.row_ind().cbegin(), B.row_ind().cend(), row_ind.begin());
81  }
82  hif_error_if(P.status() == DATA_UNDEF, "memory allocation failed");
83  const int result = amd::order(m, B.col_start().data(), row_ind.data(),
84  P.data(), Control, Info);
85  if (result != AMD_OK && result != AMD_OK_BUT_JUMBLED) {
86  // NOTE that we modified AMD to utilize jumbled return to automatically
87  // compute the transpose
88  std::stringstream s;
89  amd::info(s, Info);
90  const std::string msg =
91  "AMD returned invalid flag " + std::to_string(result) +
92  ", the following message was loaded from AMD info routine:\n" + s.str();
93  hif_error(msg.c_str());
94  }
95 
96  if (hif_verbose(PRE, opt)) {
97  hif_info("AMD reordering done with information:\n");
98  std::stringstream s;
99  amd::info(s, Info);
100  hif_info(s.str().c_str());
101  }
102 
103  return P;
104 }

106 /// \brief apply reversed Cuthill-Mckee ordering
107 /// \tparam CcsType ccs sparse storage, see \ref CCS
108 /// \param[in] B matrix with pattern given
109 /// \param[in] opt options
110 /// \return symmetric permutation so that PAP' satisties RCM property
111 template <class CcsType>
112 inline typename CcsType::iparray_type run_rcm(const CcsType &B,
113  const Options &opt) {
114  using indptr_type = typename CcsType::indptr_type;
115  using iparray_type = typename CcsType::iparray_type;
116  using rcm_type = rcm::RCM<indptr_type>;
117 
118  if (hif_verbose(PRE, opt)) hif_info("begin running RCM reordering...");
119  const auto n = B.nrows();
120  iparray_type adjncy;
121  if (sizeof(indptr_type) == sizeof(typename CcsType::index_type))
122  adjncy = wrap_array(B.row_ind().size(), (indptr_type *)B.row_ind().data());
123  else {
124  adjncy.resize(B.row_ind().size());
125  std::copy(B.row_ind().cbegin(), B.row_ind().cend(), adjncy.begin());
126  }
127  iparray_type &xadj = const_cast<iparray_type &>(B.col_start());
128 
129  for (auto &v : xadj) ++v;
130  for (auto &v : adjncy) ++v;
131  iparray_type P(n);
132  rcm_type().apply(n, xadj.data(), adjncy.data(), P.data());
133  for (auto &v : P) --v;
134  if (hif_verbose(PRE, opt)) hif_info("finish RCM reordering...");
135  return P;
136 }
