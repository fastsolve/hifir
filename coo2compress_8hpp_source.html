38 /// \brief Convert a coordinate sparse format to compressed storage
39 /// \tparam IndPtrType Index pointer data type
40 /// \tparam IndexType Index data type
41 /// \tparam ValueType Value data type
42 /// \param[in] primary_size Size in the compressed axis, e.g., nrows for CRS
43 /// \param[in] secondary_size Size in the uncompressed axis, e.g., ncols for CRS
44 /// \param[in] pidx Primary index list, e.g., rows for CRS
45 /// \param[in] sids Secondary index list, e.g., cols for CRS
46 /// \param[in] coovals Coo format values
47 /// \param[out] ind_start Index pointer
48 /// \param[out] indices Index list
49 /// \param[out] vals Values corresponding to \a indices
50 ///
51 /// For CRS, call this function as convert_coo2crs(nrows, ncols, rows, cols,
52 /// coovals, rowptr, colind, vals). For CCS, call this function as
53 /// convert_coo2crs(ncols, nrows, cols, rows, coovals, colptr, rowind, vals).
54 template <class IndPtrType, class IndexType, class ValueType>
55 inline void convert_coo2cs(const std::size_t primary_size,
56  const std::size_t secondary_size,
57  const Array<IndexType> &pidx,
58  const Array<IndexType> &sids,
59  const Array<ValueType> &coovals,
60  Array<IndPtrType> & ind_start,
61  Array<IndexType> &indices, Array<ValueType> &vals) {
62  using size_type = typename Array<IndexType>::size_type;
63 
64  ind_start.resize(primary_size + 1);
65  hif_error_if(ind_start.status() == DATA_UNDEF, "memory allocation failed");
66  std::fill(ind_start.begin(), ind_start.end(), IndPtrType(0));
67  const size_type nnz = pidx.size();
68  hif_assert(nnz == sids.size(), "unmatched sizes");
69  hif_assert(nnz == coovals.size(), "unmatched sizes");
70 
71  // determine structure first
72  for (size_type i(0); i < nnz; ++i) ++ind_start[pidx[i] + 1];
73  // accumulate
74  for (size_type i(0); i < primary_size; ++i) ind_start[i + 1] += ind_start[i];
75  hif_assert(nnz == (size_type)ind_start[primary_size], "fatal bug");
76 
77  // assemble to compressed storage
78  indices.resize(nnz);
79  hif_error_if(indices.status() == DATA_UNDEF, "memory allocation failed");
80  vals.resize(nnz);
81  hif_error_if(vals.status() == DATA_UNDEF, "memory allocation failed");
82  for (size_type i(0); i < nnz; ++i) {
83  const auto pos = ind_start[pidx[i]];
84  indices[pos] = sids[i];
85  vals[pos] = coovals[i];
86  ++ind_start[pidx[i]];
87  }
88 
89  // reset ind_start
90  for (size_type i(primary_size); i != 0u; --i) ind_start[i] = ind_start[i - 1];
91  ind_start[0] = 0u;
92 
93  // sort locally
94  Array<ValueType> vbuf;
95  for (size_type i(0); i < primary_size; ++i)
96  if (!std::is_sorted(indices.cbegin() + ind_start[i],
97  indices.cbegin() + ind_start[i + 1])) {
98  // not sorted
99  vbuf.resize(secondary_size);
100  hif_error_if(vbuf.status() == DATA_UNDEF, "memory allocation failed");
101  // backup values
102  for (size_type k = ind_start[i]; k < (size_type)ind_start[i + 1]; ++k)
103  vbuf[indices[k]] = vals[k];
104  // sort
105  std::sort(indices.begin() + ind_start[i],
106  indices.begin() + ind_start[i + 1]);
107  // copy back
108  for (size_type k = ind_start[i]; k < (size_type)ind_start[i + 1]; ++k)
109  vals[k] = vbuf[indices[k]];
110  }
111 }
