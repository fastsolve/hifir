46 /// \brief routine to perform preprocessing for improving the quality of input
47 /// \tparam IsSymm if \a true, the assume the leading block is symmetric
48 /// \tparam CcsType input matrix type, see \ref CCS
49 /// \tparam ScalingArray array type for scaling vectors, see \ref Array
50 /// \tparam PermType permutation type, see \ref BiPermMatrix
51 /// \param[in] A input matrix
52 /// \param[in] A_crs CRS format of input \a A
53 /// \param[in] m0 leading block size
54 /// \param[in] level current factorization level
55 /// \param[in] opt control parameters
56 /// \param[out] s row scaling vector
57 /// \param[out] t column scaling vector
58 /// \param[out] p row permutation
59 /// \param[out] q column permutation
60 /// \return The actual leading block size, no larger than \a m0
61 /// \ingroup pre
62 ///
63 /// Notice that, in general, the preprocessing involves two steps: 1) perform
64 /// matching/scaling to improve the diagonal dominance and conditioning, and 2)
65 /// perform reordering to improve the sparsity of LU decomposition.
66 template <bool IsSymm, class CcsType, class CrsType, class ScalingArray,
67  class PermType>
68 inline typename CcsType::size_type do_preprocessing(
69  const CcsType &A, const CrsType &A_crs,
70  const typename CcsType::size_type m0,
71  const typename CcsType::size_type level, const Options &opt,
72  ScalingArray &s, ScalingArray &t, PermType &p, PermType &q) {
73  static_assert(!CcsType::ROW_MAJOR, "must be CCS");
74  using index_type = typename CcsType::index_type;
75  using size_type = typename CcsType::size_type;
76 
77  if (opt.reorder < 0 || opt.reorder >= REORDER_NULL)
78  hif_error("invalid reorder flag %d", opt.reorder);
79 
80  DefaultTimer timer;
81 
82  if (hif_verbose(PRE, opt)) hif_info("performing matching step");
83 
84  timer.start();
85 
86  auto match_res = do_maching<IsSymm>(A, A_crs, m0, level, opt, s, t, p, q);
87 
88  timer.finish();
89  if (hif_verbose(PRE_TIME, opt))
90  hif_info("matching took %gs.", (double)timer.time());
91 
92  const size_type m = match_res.second;
93  if (opt.reorder != REORDER_OFF && m) {
94  if (hif_verbose(PRE, opt)) hif_info("performing reordering step");
95  timer.start();
96 
97  std::string reorder_name = "AMD";
98 
99  auto & B = match_res.first;
100  Array<index_type> P;
101  if (opt.reorder == REORDER_AUTO) {
102  // for auto reordering, we use rcm only if first level symmetry and
103  // have static deferrals
104  const bool try_use_rcm = IsSymm && level == 1u && B.nrows() != m0;
105  P = try_use_rcm ? run_rcm(B, opt) : run_amd<false>(B, opt);
106  if (try_use_rcm) reorder_name = "RCM";
107  } else if (opt.reorder == REORDER_AMD) {
108  P = run_amd<false>(B, opt);
109  } else {
110  P = run_rcm(B, opt);
111  reorder_name = "RCM";
112  }
113 
114  timer.finish();
115 
116  if (hif_verbose(PRE_TIME, opt))
117  hif_info("reordering %s took %gs.", reorder_name.c_str(),
118  (double)timer.time());
119 
120  // now let's reorder the permutation arrays
121  // we use the inverse mapping as buffer
122  const auto reorder_finalize_perm = [&P, m](PermType &Q) {
123  auto & forward = Q(), &buf = Q.inv();
124  const size_type N = forward.size();
125  size_type i(0);
126  for (; i < m; ++i) buf[i] = forward[P[i]];
127  for (; i < N; ++i) buf[i] = forward[i];
128  forward.swap(buf);
129  Q.build_inv();
130  };
131 
132  reorder_finalize_perm(p);
133  reorder_finalize_perm(q);
134  } else {
135  if (hif_verbose(PRE, opt))
136  m ? hif_info("reordering skipped")
137  : hif_info("reordering skipped due to empty leading block");
138  p.build_inv();
139  q.build_inv();
140  }
141  return m;
142 }
