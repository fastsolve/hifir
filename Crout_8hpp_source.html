42 /// \class Crout
43 /// \brief Crout update in deferred fashion
44 /// \ingroup crout
45 ///
46 /// The idea of this class is to implement Crout kernels that wrap around a
47 /// common factor---step. In this case, the class Crout is as simple (in terms
48 /// of data size) as an integer. It also supports implicitly casting to integer
49 /// as well as increment operations (both suffix and prefix versions). The whole
50 /// disign can be demonstrate by the following C++ pseudocode:
51 ///
52 /// \code{.cpp}
53 /// for (Crout crout; crout < m; ++crout) {
54 /// crout.do_work1(...);
55 /// crout.do_work2(...);
56 /// ...
57 /// }
58 /// \endcode
59 class Crout {
60  public:
61  using size_type = std::size_t; ///< size type
62 
63  /// \brief default constructor
64  Crout() : _step(0), _defers(0) {}
65 
66  // default copy and assign
67  Crout(const Crout &) = default;
68  Crout &operator=(const Crout &) = default;

71  /// \param[in] step step number
72  Crout &operator=(const size_type step) {
73  _step = step;
74  return *this;
75  }

78  inline const Crout &operator++() {
79  ++_step;
80  return *this;

85  const Crout tmp(*this);

121  /// \brief dual of the \ref compute_l, i.e. computing the row of U
122  /// \tparam ScaleArray row scaling diagonal matrix type, see \ref Array
123  /// \tparam CrsType crs format, see \ref CRS
124  /// \tparam PermType permutation matrix type, see \ref Array
125  /// \tparam ListArray "linked-list" array type, see \ref Array
126  /// \tparam IndPtrArray index pointer array, see \ref Array
127  /// \tparam DiagType diagonal matrix type, see \ref Array
128  /// \tparam CcsType ccs format, see \ref CCS
129  /// \tparam SpVecType work array for current row vector, see \ref SparseVector
130  /// \param[in] s row scaling matrix from preprocessing
131  /// \param[in] crs_A input matrix in crs scheme
132  /// \param[in] t column scaling matrix from preprocessing
133  /// \param[in] pk permutated row index
134  /// \param[in] q column inverse permutation (deferred)
135  /// \param[in] L lower part
136  /// \param[in] L_start starting positions for current step
137  /// \param[in] L_list linked list of column indices for current step
138  /// \param[in] d diagonal vector
139  /// \param[in] U upper part
140  /// \param[in] U_start leading row positions of current \ref _step
141  /// \param[out] ut current row vector of U
142  /// \sa compute_l
143  ///
144  /// This routine computes the current row vector of \f$\mathbf{U}\f$
145  /// (w/o diagonal scaling). Mathematically, this routine is to compute:
146  ///
147  /// \f[
148  /// \mathbf{u}_{k}^{T}=
149  /// \hat{\mathbf{A}}[p_{k},\mathbf{q}_{k+1:n}]-
150  /// \mathbf{L}_{k,1:k-1}\mathbf{D}_{k-1}
151  /// \mathbf{U}_{1:k-1,k+1:n}
152  /// \f]
153  ///
154  /// It's worth noting that, conceptally, the formula above is nothing but
155  /// a vector matrix operation. However, standard implementation won't give
156  /// good performance (especially with consideration of cache performance),
157  /// this is because \f$\mathbf{L}\f$ is stored in column major whereas
158  /// row major for \f$\mathbf{U}\f$. Therefore, the actual implementation
159  /// is in the fashion that loops through \f$\mathbf{U}\f$ in row major
160  /// while keeping \f$\mathbf{L}\f$ as much static as possible.
161  ///
162  /// Regarding the complexity cost, it takes
163  /// \f$\mathcal{O}(\textrm{nnz}(\mathbf{A}_{p_k,:}))\f$ to first load the
164  /// row of A to \f$\mathbf{u}_k^T\f$, and takes
165  /// \f$\mathcal{O}(\cup_j \textrm{nnz}(\mathbf{U}_{j,:}))\f$, where
166  /// \f$l_{kj}\neq 0\f$ for computing the vector matrix operation.
167  template <class ScaleArray, class CrsType, class PermType, class ListArray,
168  class IndPtrArray, class DiagType, class CcsType, class SpVecType>
169  void compute_ut(const ScaleArray &s, const CrsType &crs_A,
170  const ScaleArray &t, const size_type pk, const PermType &q,
171  const CcsType &L, const IndPtrArray &L_start,
172  const ListArray &L_list, const DiagType &d, const CrsType &U,
173  const IndPtrArray &U_start, SpVecType &ut) const {
174  // compilation checking
175  static_assert(CrsType::ROW_MAJOR, "input A must be CRS for loading ut");
176  using index_type = typename ListArray::value_type;
177  constexpr static index_type nil = static_cast<index_type>(-1);
178 
179  // reset sparse buffer
180  ut.reset_counter();
181 
182  // first load the A row
183  _load_arow(s, crs_A, t, pk, q, ut);
184 
185  // if not first step
186  if (_step) {
187  index_type L_col = L_list[deferred_step()];
188  while (L_col != nil) {
189  hif_assert(
190  (size_type)L_col < _step,
191  "compute_ut column index %zd should not exceed step %zd for L",
192  (size_type)L_col, _step);
193  // compute L*d
194  const auto ld = *(L.val_cbegin(L_col) + L_start[L_col]) * d[L_col];
195  auto U_v_itr = U.val_cbegin(L_col) + U_start[L_col];
196  auto U_i_itr = U.col_ind_cbegin(L_col) + U_start[L_col],
197  U_last = U.col_ind_cend(L_col);
198  if (U_i_itr != U_last && size_type(*U_i_itr) == deferred_step()) {
199  // advance to offsets
200  ++U_i_itr;
201  ++U_v_itr;
202  }
203  for (; U_i_itr != U_last; ++U_i_itr, ++U_v_itr) {
204  // convert to c index
205  const size_type idx = *U_i_itr;
206  hif_assert(idx > deferred_step(),
207  "U index %zd in computing ut should greater than step "
208  "%zd(defers:%zd)",
209  idx, _step, _defers);
210  if (ut.push_back(idx, _step))
211  ut.vals()[idx] = -ld * *U_v_itr;
212  else
213  ut.vals()[idx] -= ld * *U_v_itr;
214  }
215  // get to next column index
216  L_col = L_list[L_col];
217  } // while
218  }
219  }

221  /// \brief compute the column vector of L of current step
222  /// \tparam IsSymm if \a true, then a symmetric leading block is assumed
223  /// \tparam ScaleArray row/column scaling diagonal matrix type, see \ref Array
224  /// \tparam CcsType ccs format, see \ref CCS
225  /// \tparam PermType permutation matrix type, see \ref Array
226  /// \tparam ListArray "linked-list" array type, see \ref Array
227  /// \tparam IndPtrArray index pointer array type, see \ref Array
228  /// \tparam DiagType array used for storing diagonal array, see \ref Array
229  /// \tparam CrsType crs format, see \ref CRS
230  /// \tparam SpVecType sparse vector for storing l, see \ref SparseVector
231  /// \param[in] s row scaling matrix from preprocessing
232  /// \param[in] ccs_A input matrix in ccs scheme
233  /// \param[in] t column scaling matrix from preprocessing
234  /// \param[in] p row inverse permutation matrix (deferred)
235  /// \param[in] qk permutated column index
236  /// \param[in] m leading block size
237  /// \param[in] L lower part of decomposition
238  /// \param[in] L_start "linked-list" array storing the starting locations of
239  /// \a L \param[in] d diagonal entries \param[in] U augmented U matrix
240  /// \param[in] U_start starting positions of U
241  /// \param[in] U_list linked list of row indices of current step in U
242  /// \param[out] l column vector of L at current \ref _step
243  /// \sa compute_ut
244  ///
245  /// This routine computes the current column vector of \f$\mathbf{L}\f$
246  /// (w/o diagonal scaling). Mathematically, this routine is to compute:
247  ///
248  /// \f[
249  /// \mathbf{\ell}_{k}=
250  /// \hat{\mathbf{A}}[\mathbf{p}_{k+1:n},q_{k}]-
251  /// \mathbf{L}_{k+1:n,1:k-1}\mathbf{D}_{k-1}
252  /// \mathbf{U}_{1:k-1,k}
253  /// \f]
254  ///
255  /// It's worth noting that, conceptally, the formula above is nothing but
256  /// a matrix vector operation. However, standard implementation won't give
257  /// good performance (especially with consideration of cache performance),
258  /// this is because \f$\mathbf{L}\f$ is stored in column major whereas
259  /// row major for \f$\mathbf{U}\f$. Therefore, the actual implementation
260  /// is in the fashion that loops through \f$\mathbf{L}\f$ in column major
261  /// while keeping \f$\mathbf{U}\f$ as much static as possible.
262  ///
263  /// Regarding the complexity cost, it takes
264  /// \f$\mathcal{O}(\textrm{nnz}(\mathbf{A}_{:,q_k}))\f$ to first load the
265  /// column of A to \f$\mathbf{\ell}_k\f$, and takes
266  /// \f$\mathcal{O}(\cup_i \textrm{nnz}(\mathbf{L}_{:,i}))\f$, where
267  /// \f$u_{ik}\neq 0\f$ for computing the matrix vector operation.
268  template <bool IsSymm, class ScaleArray, class CcsType, class PermType,
269  class ListArray, class IndPtrArray, class DiagType, class CrsType,
270  class SpVecType>
271  void compute_l(const ScaleArray &s, const CcsType &ccs_A, const ScaleArray &t,
272  const PermType &p, const size_type qk, const size_type m,
273  const CcsType &L, const IndPtrArray &L_start,
274  const DiagType &d, const CrsType &U,
275  const IndPtrArray &U_start, const ListArray &U_list,
276  SpVecType &l) const {
277  // compilation checking
278  static_assert(!CcsType::ROW_MAJOR, "input A must be CCS for loading l");
279  using index_type = typename ListArray::value_type;
280  constexpr static index_type nil = static_cast<index_type>(-1);
281 
282  // clear sparse counter
283  l.reset_counter();
284 
285  if (!IsSymm || (_defers || m != ccs_A.nrows())) {
286  // load A column
287  _load_acol<IsSymm>(s, ccs_A, t, p, qk, m, l);
288 
289  // if not first step
290  if (_step) {
291  index_type U_row = U_list[deferred_step()];
292  while (U_row != nil) {
293  hif_assert(
294  IsSymm || (size_type)U_row < _step,
295  "compute_ut row index %zd should not exceed step %zd for U",
296  (size_type)U_row, _step);
297 
298  // compute d*U
299  const auto du = d[U_row] * *(U.val_cbegin(U_row) + U_start[U_row]);
300  auto L_v_itr = L.val_cbegin(U_row) + L_start[U_row];
301  auto L_i_itr = L.row_ind_cbegin(U_row) + L_start[U_row],
302  L_last = L.row_ind_cend(U_row);
303  if (L_i_itr != L_last && size_type(*L_i_itr) == deferred_step()) {
304  ++L_i_itr;
305  ++L_v_itr;
306  }
307  for (; L_i_itr != L_last; ++L_i_itr, ++L_v_itr) {
308  const auto idx = *L_i_itr;
309  hif_assert((size_type)idx > deferred_step(),
310  "L index %zd in computing l should greater than step "
311  "%zd(defers:%zd)",
312  (size_type)idx, _step, _defers);
313  // compute this entry, if index does not exist, assign new value to
314  // -L*d*u, else, -= L*d*u
315  if (l.push_back(idx, _step))
316  l.vals()[idx] = -du * *L_v_itr;
317  else
318  l.vals()[idx] -= *L_v_itr * du;
319  }
320  // update U linked list
321  U_row = U_list[U_row];
322  } // while
323  }
324  }
325  }

327  /// \brief compute the column vector of L of current step for \f$LDL^{T}\f$
328  /// \tparam ScaleArray row/column scaling diagonal matrix type, see \ref Array
329  /// \tparam CcsType ccs format, see \ref CCS
330  /// \tparam PermType permutation matrix type, see \ref Array
331  /// \tparam ListArray "linked-list" array type, see \ref Array
332  /// \tparam IndPtrArray index pointer array type, see \ref Array
333  /// \tparam DiagType array used for storing diagonal array, see \ref Array
334  /// \tparam SpVecType sparse vector for storing l, see \ref SparseVector
335  /// \param[in] s row scaling matrix from preprocessing
336  /// \param[in] ccs_A input matrix in ccs scheme
337  /// \param[in] t column scaling matrix from preprocessing, should be \a s
338  /// \param[in] p row inverse permutation matrix (deferred)
339  /// \param[in] qk permutated column index
340  /// \param[in] m leading block size
341  /// \param[in] L lower part of decomposition
342  /// \param[in] L_start position array storing the starting locations of \a L
343  /// \param[in] L_list linked list of row indices of current step in \a L
344  /// \param[in] d diagonal entries
345  /// \param[out] l column vector of L at current \ref _step
346  /// \sa compute_ut
347  ///
348  /// This routine computes the current column vector of \f$\mathbf{L}\f$
349  /// (w/o diagonal scaling). Mathematically, this routine is to compute:
350  ///
351  /// \f[
352  /// \mathbf{\ell}_{k}=
353  /// \hat{\mathbf{A}}[\mathbf{p}_{k+1:n},q_{k}]-
354  /// \mathbf{L}_{k+1:n,1:k-1}\mathbf{D}_{k-1}
355  /// \mathbf{L}_{k,1:k-1}^{T}
356  /// \f]
357  template <class ScaleArray, class CcsType, class PermType, class ListArray,
358  class IndPtrArray, class DiagType, class SpVecType>
359  void compute_symm(const ScaleArray &s, const CcsType &ccs_A,
360  const ScaleArray &t, const PermType &p, const size_type qk,
361  const size_type m, const CcsType &L,
362  const IndPtrArray &L_start, const ListArray &L_list,
363  const DiagType &d, SpVecType &l) const {
364  // compilation checking
365  static_assert(!CcsType::ROW_MAJOR, "input A must be CCS for loading l");
366  using index_type = typename ListArray::value_type;
367  constexpr static index_type nil = static_cast<index_type>(-1);
368 
369  // clear sparse counter
370  l.reset_counter();
371 
372  // load A column
373  _load_acol<false>(s, ccs_A, t, p, qk, m, l);
374 
375  // if not first step
376  if (_step) {
377  index_type L_col = L_list[deferred_step()];
378  while (L_col != nil) {
379  hif_assert((size_type)L_col < _step,
380  "compute_ut row index %zd should not exceed step %zd for U",
381  (size_type)L_col, _step);
382  // compute d*L^T
383  auto L_v_itr = L.val_cbegin(L_col) + L_start[L_col];
384  const auto dlt = d[L_col] * *L_v_itr;
385  auto L_i_itr = L.row_ind_cbegin(L_col) + L_start[L_col],
386  L_last = L.row_ind_cend(L_col);
387  if (L_i_itr != L_last) {
388  ++L_i_itr;
389  ++L_v_itr;
390  }
391  for (; L_i_itr != L_last; ++L_i_itr, ++L_v_itr) {
392  const auto idx = *L_i_itr;
393  hif_assert((size_type)idx > deferred_step(),
394  "L index %zd in computing l should greater than step "
395  "%zd(defers:%zd)",
396  (size_type)idx, _step, _defers);
397  // compute this entry, if index does not exist, assign new value to
398  // -L*d*L^T, else, -= L*d*L^T
399  if (l.push_back(idx, _step))
400  l.vals()[idx] = -dlt * *L_v_itr;
401  else
402  l.vals()[idx] -= *L_v_itr * dlt;
403  }
404  // update L linked list
405  L_col = L_list[L_col];
406  } // while
407  }
408  }

410  /// \brief compress L and U and update their corresponding starting positions
411  /// and linked lists
412  /// \tparam CsType either \ref CRS or \ref CCS
413  /// \tparam ListArray "linked-list" array type, see \ref Array
414  /// \tparam IndPtrArray index pointer array, see \ref Array
415  /// \param[in,out] T strictly lower (L) or upper (U) matrices
416  /// \param[in,out] list linked list of current step with primary indices
417  /// \param[in,out] start local positions
418  ///
419  /// The overall idea involves the following steps: (1) loop through the
420  /// current secondary axis to get the primary indices in \a list, then (2)
421  /// from these primary indices combining with starting positions, \a start,
422  /// to get the index and value arrays position, (3) next step is to compress
423  /// the index wrt to the current overall deferals, finally (4) advance the
424  /// starting position and update the linked list accordingly.
425  ///
426  /// Clearly, all these steps must be done concurrently with in a single loop.
427  template <class CsType, class ListArray, class IndPtrArray>
428  inline void update_compress(CsType &T, ListArray &list,
429  IndPtrArray &start) const {
430  using index_type = typename ListArray::value_type;
431  constexpr static index_type nil = static_cast<index_type>(-1);
432 
433  // now update k
434  index_type idx = list[deferred_step()];
435  while (idx != nil) {
436  auto itr = T.ind_begin(idx) + start[idx];
437  *itr -= _defers; // compress
438  ++start[idx]; // increment next position
439  const auto next = list[idx];
440  if (++itr != T.ind_end(idx)) {
441  list[idx] = list[*itr];
442  list[*itr] = idx;
443  } else
444  list[idx] = nil;
445  idx = next; // update linked list
446  } // while
447  list[_step] = nil;
448  // need to update newly added entry at the end
449  start[_step] = 0;
450  if (T.nnz_in_primary(_step)) {
451  const size_type first = *T.ind_cbegin(_step);
452  if (list[first] != nil) list[_step] = list[first];
453  list[first] = _step;
454  hif_assert(first != deferred_step(),
455  "newly added node cannot have diagonal entry");
456  }
457  }

459  /// \brief another starting array needed in symmetric computation to get
460  /// the offsets
461  /// \tparam CcsType ccs type for L, see \ref CCS
462  /// \tparam IndPtrArray index pointer array, see \ref Array
463  /// \param[in] L lower factor
464  /// \param[in] m leading block size
465  /// \param[out] L_start offset starting positions
466  template <class CcsType, class IndPtrArray>
467  inline void symm_update_lstart(const CcsType & L,
468  const typename CcsType::size_type m,
469  IndPtrArray &L_start) const {
470  static_assert(!CcsType::ROW_MAJOR, "must be CCS");
471 
472  auto info = find_sorted(L.row_ind_cbegin(_step), L.row_ind_cend(_step), m);
473  L_start[_step] = info.second - L.row_ind_cbegin(_step);
474  }

476  /// \brief estimate the inverse norm
477  /// \tparam CsType either \ref CRS (U) or \ref CCS (L)
478  /// \tparam ListArray "linked-list" array type, see \ref Array
479  /// \tparam IndPtrArray index pointer array
480  /// \tparam KappaArray array type for storing inverse norms, see \ref Array
481  /// \param[in] T lower or upper factors
482  /// \param[in] list linked list of primary indices of current step
483  /// \param[in] start starting positions in index and value arrays
484  /// \param[in,out] kappa inverse norm solutions
485  template <class CsType, class ListArray, class IndPtrArray, class KappaArray>
486  inline bool update_kappa(const CsType &T, const ListArray &list,
487  const IndPtrArray &start, KappaArray &kappa) const {
488  using value_type = typename CsType::value_type;
489  using index_type = typename ListArray::value_type;
490  constexpr static index_type nil = static_cast<index_type>(-1);
491  constexpr static bool one = true, neg_one = false;
492  if (!_step) {
493  kappa[0] = value_type(1);
494  return one;
495  }
496 
497  // need to sum all values
498  value_type sum(0);
499  index_type idx = list[deferred_step()];
500  while (idx != nil) {
501  hif_assert((size_type)idx < _step,
502  "the triangular matrices should be strict lower/upper, "
503  "idx=%zd/step=%zd",
504  (size_type)idx, _step);
505  sum += kappa[idx] * *(T.val_cbegin(idx) + start[idx]);
506  idx = list[idx]; // traverse to next node
507  }
508  const value_type k1 = value_type(1) - sum, k2 = -value_type(1) - sum;
509  if (std::abs(k1) < std::abs(k2)) {
510  kappa[_step] = k2;
511  return neg_one;
512  }
513  kappa[_step] = k1;
514  return one;
515  }

517  /// \brief update the current diagonal entry
518  /// \tparam IsSymm if \a true, then assuming we have a leading block
519  /// \tparam SpVecType sparse vector for storing l/U, see \ref SparseVector
520  /// \tparam DiagType diagonal vector array type, see \ref Array
521  /// \param[in] l newly computed column vector of L
522  /// \param[in] ut newly computed row vector of U
523  /// \param[in] m leading block size (not necessary for symmetric)
524  /// \param[in,out] d diagonal vector
525  ///
526  /// This routine is \a SFINAE-able by \a IsSymm and it's for asymmetric cases.
527  /// This routine is to compute the last step in Crout updates, i.e. updating
528  /// the diagonal entries:
529  ///
530  /// \f[
531  /// \mathbf{D}_{k+1:m,k+1:m}=\mathbf{D}_{k+1:m,k+1:m}-
532  /// \mathbf{D}_{k,k}
533  /// \left(\mathbf{\ell}_k\otimes\mathbf{u}_k\right)_{k+1:m,k+1:m}
534  /// \f]
535  ///
536  /// The complexity is bounded by:
537  ///
538  /// \f[
539  /// \mathcal{O}(\min(\textrm{nnz}(\mathbf{\ell}_k),
540  /// \textrm{nnz}(\mathbf{u}_k)))
541  ///\f]
542  template <bool IsSymm, class SpVecType, class DiagType>
543  inline typename std::enable_if<!IsSymm>::type update_diag(const SpVecType &l,
544  const SpVecType &ut,
545  const size_type m,
546  DiagType &d) const {
547  // NOTE that we need the internal dense tag from sparse vector
548  // thus, we can either:
549  // 1) make this function a friend of SparseVec, or
550  // 2) create a caster class
551  using extractor = internal::SpVInternalExtractor<SpVecType>;
552 
553  // assume l is not scaled by the diagonal
554 
555  // get the current diagonal entry
556  // const auto dk = d[_step];
557  if (ut.size() <= l.size()) {
558  const auto & l_d_tags = static_cast<const extractor &>(l).dense_tags();
559  const size_type n = ut.size();
560  const auto & l_vals = l.vals();
561  for (size_type i = 0u; i < n; ++i) {
562  const size_type idx = ut.idx(i);
563  hif_assert(idx > deferred_step(),
564  "should only contain the upper part of ut, "
565  "(idx,step(defer))=(%zd,%zd(%zd))!",
566  idx, _step, _defers);
567  // if the dense tags of this entry points to this step, we know l has
568  // an element in this slot
569  if (idx < m && (size_type)l_d_tags[idx] == _step)
570  d[idx] -= ut.val(i) * l_vals[idx];
571  }
572  } else {
573  const auto &ut_d_tags = static_cast<const extractor &>(ut).dense_tags();
574  const size_type n = l.size();
575  const auto & u_vals = ut.vals();
576  for (size_type i = 0u; i < n; ++i) {
577  const size_type idx = l.idx(i);
578  hif_assert(idx > deferred_step(),
579  "should only contain the lower part of l, "
580  "(idx,step(defer))=(%zd,%zd(%zd))!",
581  idx, _step, _defers);
582  // if the dense tags of this entry points to this step, we know ut has
583  // an element in this slot
584  if (idx < m && (size_type)ut_d_tags[idx] == _step)
585  d[idx] -= u_vals[idx] * l.val(i);
586  }
587  }
588  }

590  /// \brief update the current diagonal entry
591  /// \tparam IsSymm if \a true, then assuming we have a leading block
592  /// \tparam SpVecType sparse vector for storing l, see \ref SparseVector
593  /// \tparam DiagType diagonal vector array type, see \ref Array
594  /// \param[in] ut newly computed row vector of U
595  /// \param[in] m leading block size (not necessary for symmetric)
596  /// \param[in,out] d diagonal vector
597  ///
598  /// This routine is \a SFINAE-able by \a IsSymm and it's for symmetric cases.
599  /// This routine is to compute the last step in Crout updates, i.e. updating
600  /// the diagonal entries:
601  ///
602  /// \f[
603  /// \mathbf{D}_{k+1:m,k+1:m}=\mathbf{D}_{k+1:m,k+1:m}-
604  /// \mathbf{D}_{k,k}
605  /// \left(\mathbf{\ell}_k\otimes\mathbf{u}_k\right)_{k+1:m,k+1:m}
606  /// \f]
607  ///
608  /// The complexity is bounded by:
609  ///
610  /// \f[
611  /// \mathcal{O}(\textrm{nnz}(\mathbf{u}_k))
612  ///\f]
613  template <bool IsSymm, class SpVecType, class DiagType>
614  inline typename std::enable_if<IsSymm>::type update_diag(
615  const SpVecType & /* l */, const SpVecType &ut, const size_type m,
616  DiagType &d) const {
617  hif_assert(m, "fatal, symmetric block cannot be empty!");
618  // get the current diagonal entry
619  const auto dk = d[_step];
620  // we only need to deal with ut
621  const size_type n = ut.size();
622  for (size_type i(0); i < n; ++i) {
623  const size_type idx = ut.idx(i);
624  hif_assert(idx > deferred_step(),
625  "ut should only contain the upper part, "
626  "(idx,step(defer))=(%zd,%zd(%zd))!",
627  idx, _step, _defers);
628  if (idx < m) d[idx] -= dk * ut.val(i) * ut.val(i);
629  }
630  }

632  /// \brief scale the computed row/column vectors by current diagonal inverse
633  /// \tparam DiagType diagonal matrix type, see \ref Array
634  /// \tparam SpVecType sparse vector for storing l/U, see \ref SparseVector
635  /// \param[in] d diagonal vector
636  /// \param[in,out] v vector to be scaled
637  /// \return The returned boolean flag indices a singularity occurs if \a true
638  ///
639  /// This routine computes \f$\mathbf{v}=\mathbf{v}/d_k\f$; firstly,
640  /// singularity is checked to ensure the validation of computation. Then,
641  /// we test if it's safe to invert \f$d_k\f$ so that we can use multiplication
642  /// instead of division.
643  ///
644  /// Regarding complexity, this routine takes
645  /// \f$\mathcal{O}(\textrm{nnz}(\mathbf{v}))\f$
646  template <class DiagType, class SpVecType>
647  inline bool scale_inv_diag(const DiagType &d, SpVecType &v) const {
648  using value_type = typename DiagType::value_type;
649  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
650  constexpr static scalar_type zero = Const<scalar_type>::ZERO;
651  constexpr static scalar_type safe_min = Const<scalar_type>::MIN;
652  constexpr static bool okay = false;
653 
654  const value_type dk = d[_step];
655  // first, if exactly zero, return fail
656  if (std::abs(dk) == zero) return !okay;
657 
658  const size_type n = v.size();
659  auto & vals = v.vals();
660 
661  if (std::abs(dk) > safe_min) {
662  // take the inverse, do multiply
663  const value_type dk_inv = value_type(1) / dk;
664  for (size_type i = 0u; i < n; ++i) vals[v.idx(i)] *= dk_inv;
665  } else
666  for (size_type i = 0u; i < n; ++i) vals[v.idx(i)] /= dk;
667 
668  return okay;
669  }

671  /// \brief defer an secondary entry
672  /// \tparam CsType either \ref CRS (U) or \ref CCS (L)
673  /// \tparam IndPtrArray index pointer array
674  /// \tparam ListArray "linked-list" array, see \ref Array
675  /// \param[in] to_idx deferred destination
676  /// \param[in] start starting positions
677  /// \param[in,out] T lower or upper factors
678  /// \param[in,out] list linked list of primary indices
679  /// \ingroup defer
680  template <class CsType, class IndPtrArray, class ListArray>
681  inline void defer_entry(const size_type to_idx, const IndPtrArray &start,
682  CsType &T, ListArray &list) const {
683  using index_type = typename ListArray::value_type;
684  constexpr static index_type nil = static_cast<index_type>(-1);
685 
686  hif_assert(list[to_idx] == nil, "deferred location %zd must be nil",
687  to_idx);
688 
689  index_type idx = list[deferred_step()];
690  while (idx != nil) {
691  auto itr = T.ind_begin(idx) + start[idx];
692  hif_assert(itr != T.ind_end(idx), "fatal");
693  const size_type n = T.ind_end(idx) - itr,
694  vp = T.ind_start()[idx] + start[idx];
695  *itr = to_idx; // set deferred index
696  rotate_left(n, vp, T.inds());
697  rotate_left(n, vp, T.vals());
698  // update linked list
699  itr = T.ind_begin(idx) + start[idx];
700  const size_type j = *itr;
701  const auto next = list[idx];
702  list[idx] = list[j];
703  list[j] = idx;
704  idx = next; // next node
705  }
706  list[_step] = nil;
707  }

714  /// \param[in] to_idx deferred destination

716  /// \param[in,out] L lower factor L

718  /// \param[out] L_offsets offsets of asymmetric potions
719  /// \ingroup defer
720  template <class CcsType, class IndPtrArray, class ListArray>
721  inline void symm_defer_l(const size_type to_idx, const IndPtrArray &L_start,
722  CcsType &L, ListArray &L_list,
723  IndPtrArray &L_offsets) const {
724  static_assert(!CcsType::ROW_MAJOR, "must be CCS");

726  constexpr static index_type nil = static_cast<index_type>(-1);
727 
728  index_type idx = L_list[deferred_step()];
729  while (idx != nil) {
730  auto itr = L.row_ind_begin(idx) + L_start[idx];
731  const size_type n = L.row_ind_end(idx) - itr,
732  vp = L.col_start()[idx] + L_start[idx];
733  *itr = to_idx;
734  rotate_left(n, vp, L.inds());
735  rotate_left(n, vp, L.vals());
736  itr = L.row_ind_begin(idx) + L_start[idx];
737  const size_type j = *itr;
738  const auto next = L_list[idx];
739  L_list[idx] = L_list[j];
740  L_list[j] = idx;
741  --L_offsets[idx];
742  idx = next;
743  }
