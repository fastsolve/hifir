49 /// \brief get number of threads
50 /// \param[in] threads user threads or tag
51 ///
52 /// If \a threads is the INT_MAX, then query the system threads without
53 /// initializing (if haven't). If \a threads is not positive, then set HIF
54 /// threads to be system threads, else set it to be user threads, and then
55 /// return it. Thread-safe only when \a threads is INT_MAX
56 inline int get_nthreads(const int threads = std::numeric_limits<int>::max()) {
57 #ifdef _OPENMP
58  static int _nthreads = 0;
59  if (threads == std::numeric_limits<int>::max()) return _nthreads;
60  if (threads <= 0) {
61  const char *env_threads = std::getenv("HIF_NUM_THREADS");
62  if (env_threads) {
63  _nthreads = std::atoi(env_threads);
64  } else {
65  env_threads = std::getenv("OMP_NUM_THREADS");
66  if (env_threads)
67  _nthreads = std::atoi(env_threads);
68  else
69  _nthreads = omp_get_max_threads();
70  }
71  if (_nthreads < 1) _nthreads = 1;
72  } else
73  _nthreads = threads;
74  return _nthreads;
75 #else
76  (void)threads;
77  return 1;
78 #endif
79 }

90 /// \brief uniform partition
91 inline std::pair<std::size_t, std::size_t> uniform_partition(
92  const std::size_t n, const int threads, const int thread) {
93  const std::size_t B = n / threads, offsets = n - B * threads;
94  std::size_t start_idx_ = B * thread, len_ = B;
95  if (offsets) {
96  const int shift = threads - offsets;
97  if (thread >= shift) {
98  ++len_;
99  start_idx_ += thread - shift;
100  }
101  }
102  return std::make_pair(start_idx_, len_ + start_idx_);
103 }
