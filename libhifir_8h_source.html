85 /*!
86  * We use an anonymous enum to group parameter positions in an array. The first
87  * thing one needs to do is to create a @a double array of length

89  *
90  * @code{.c}
91  * double params[LHF_NUMBER_PARAMS];

94  * @endcode

97  LHF_DROPTOL_L = 0, /*!< Drop tolerance for L factor */
98  LHF_DROPTOL_U, /*!< Drop tolerance for U factor */

100  LHF_COND, /*!< Conditioning threshold for L and U factors */
101  LHF_ALPHA_L, /*!< Scalability-oriented dropping factor for L */
102  LHF_ALPHA_U, /*!< Scalability-oriented dropping factor for U */
103  LHF_VERBOSE, /*!< Verbose level */
104  LHF_REORDER, /*!< Reorder option */
105  LHF_SYMMPRELVLS, /*!< Option for tuning symmetric preprocessing levels */
106  LHF_THREADS, /*!< Option for choosing nthreads for Schur computation */
107  LHF_RRQR_COND, /*!< Condition number threshold used in the final RRQR */
108  LHF_PIVOT, /*!< Pivoting option */
109  LHF_BETA, /*!< Option for the threshold used in preventing bad scaling factors

111  LHF_ISSYMM, /*!< Flag to tunning on symmetric/Hermitian input */
112  LHF_NOPRE, /*!< Option to turn on/off preprocessing */
113  LHF_NZP_THRES, /*!< Nonzero pattern symmetry threshold for symmetric
114  preprocessing */
115  LHF_DENSE_THRES, /*!< Dense size threshold for the Schur complement */

658 /*!
659  * @brief Apply a preconditioner with a certian operation mode
660  * @param[in] hif A HIF instance
661  * @param[in] op Operation tag
662  * @param[in] b The RHS vector
663  * @param[in] nirs Number of iterative refinements
664  * @param[in] betas Relative residual norm bounds for IR with op=solve
665  * @param[in] rank Numerical rank
666  * @param[out] x Computed solution vector
667  * @param[out] ir_status (optional) If IR and @a betas is provided, then this
668  * records the number of actual IR ( @a ir_status[0] )
669  * and the IR status ( @a ir_status[1] ).
670  *
671  * This is the core function of HIF. A HIF instance can be applied in four
672  * different modes, namely triangular solve and its Hermitian/tranpose variant,
673  * as well as matrix-vector multiplication and its Hermitian/tranpose variant.
