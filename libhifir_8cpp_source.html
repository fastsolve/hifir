57 static inline void get_factor_stats(const PrecType &M, std::size_t stats[]) {
58  if (M.empty())
59  for (int i = 0; i < 9; ++i) stats[i] = 0;
60  else {
61  stats[0] = M.nnz();
62  stats[1] = M.stats(0);
63  stats[2] = M.stats(1);

65  stats[4] = M.stats(5);
66  stats[5] = M.levels();
67  stats[6] = M.rank();
68  stats[7] = M.schur_rank();
69  stats[8] = M.schur_size();
70  }

96 // read sparse matrix from matrix market file
97 template <class ValueType, class MatHdlType>
98 static inline LhfStatus mm_read_sparse(const char *fname, MatHdlType mat) {
99  _LHF_CANNOT_ACCEPT_NULL(mat);
100  if (mat->is_rowmajor) {
101  using matrix_t = hif::CRS<ValueType, LhfInt, LhfIndPtr>;
102  matrix_t A;
103  try {
104  A = matrix_t::from_mm(fname);
105  } catch (const std::exception &e) {
106  _LHF_RETURN_FROM_HIFIR_ERROR(e);
107  }
108  if (A.nrows() != mat->n || A.ncols() != mat->n) return LHF_MISMATCHED_SIZES;
109  std::copy(A.ind_start().cbegin(), A.ind_start().cend(), mat->indptr);
110  std::copy(A.inds().cbegin(), A.inds().cend(), mat->indices);
111  std::copy(A.vals().cbegin(), A.vals().cend(), mat->vals);
112  } else {
113  using matrix_t = hif::CCS<ValueType, LhfInt, LhfIndPtr>;
114  matrix_t A;
115  try {
116  A = matrix_t::from_mm(fname);
117  } catch (const std::exception &e) {
118  _LHF_RETURN_FROM_HIFIR_ERROR(e);
119  }
120  if (A.nrows() != mat->n || A.ncols() != mat->n) return LHF_MISMATCHED_SIZES;
121  std::copy(A.ind_start().cbegin(), A.ind_start().cend(), mat->indptr);
122  std::copy(A.inds().cbegin(), A.inds().cend(), mat->indices);
123  std::copy(A.vals().cbegin(), A.vals().cend(), mat->vals);
124  }
125  return LHF_SUCCESS;
126 }

128 // factorization implementation
129 template <class PrecType, class MatHdlType>
130 static inline void factorize(PrecType &M, const MatHdlType mat,
131  const double params[]) {
132  if (!params) {
133  if (mat->is_rowmajor)
134  M.template factorize<true>(mat->n, mat->indptr, mat->indices, mat->vals);
135  else
136  M.template factorize<false>(mat->n, mat->indptr, mat->indices, mat->vals);
137  } else {
138  if (mat->is_rowmajor)
139  M.template factorize<true>(mat->n, mat->indptr, mat->indices, mat->vals,
140  create_params(params));
141  else
142  M.template factorize<false>(mat->n, mat->indptr, mat->indices, mat->vals,
143  create_params(params));
144  }
145 }

171 template <class PrecType, class MatHdlType, class ValueType>
172 static inline void hifir(const PrecType &M, const MatHdlType A,
173  const ValueType *b, const int nirs,
174  const double *betas, const bool trans, const int rank,
175  ValueType *x, int *ir_status) {
176  using array_t = hif::Array<ValueType>;
177 
178  array_t bb(A->n, (ValueType *)b, true);
179  array_t xx(A->n, (ValueType *)x, true);
180  if (!betas) {
181  // no residual bounds
182  if (A->is_rowmajor) {
183  const auto AA(hif::wrap_const_crs(A->n, A->n, A->indptr, A->indices,
184  (const ValueType *)A->vals, false));
185  M.hifir(AA, bb, nirs, xx, trans, rank);
186  } else {
187  const auto AA(hif::wrap_const_ccs(A->n, A->n, A->indptr, A->indices,
188  (const ValueType *)A->vals, false));
189  M.hifir(AA, bb, nirs, xx, trans, rank);
190  }
191  } else {
192  if (A->is_rowmajor) {
193  const auto AA(hif::wrap_const_crs(A->n, A->n, A->indptr, A->indices,
194  (const ValueType *)A->vals, false));
195  const auto info = M.hifir(AA, bb, nirs, betas, xx, trans, rank);
196  if (ir_status) {
197  ir_status[0] = info.first;
198  ir_status[1] = info.second;
199  }
200  } else {
201  const auto AA(hif::wrap_const_ccs(A->n, A->n, A->indptr, A->indices,
202  (const ValueType *)A->vals, false));
203  const auto info = M.hifir(AA, bb, nirs, betas, xx, trans, rank);
204  if (ir_status) {
205  ir_status[0] = info.first;
206  ir_status[1] = info.second;
207  }
208  }
209  }

452 // Apply
453 LhfStatus lhfdApply(const LhfdHifHdl hif, const LhfOperationType op,
454  const double *b, const int nirs, const double *betas,
455  const int rank, double *x, int *ir_status) {
456  _LHF_CANNOT_ACCEPT_NULL(hif);
457  const bool trans = op == LHF_SH || op == LHF_MH;
458  // determine rank
459  int rnk(rank);
460  if (rank == LHF_DEFAULT_RANK)
461  rnk = ((op != LHF_S && op != LHF_SH) || nirs > 1) ? -1 : 0;
462  try {
463  if (nirs <= 1 || op == LHF_M || op == LHF_MH) {
464  if (op == LHF_M || op == LHF_MH)
465  libhifir_impl::mmultiply(*hif->M, b, x, trans);
466  else
467  libhifir_impl::solve(*hif->M, b, x, trans);
468  } else {
469  _LHF_CANNOT_ACCEPT_NULL(hif->A);
470  if (hif->M->nrows() != hif->A->n) return LHF_MISMATCHED_SIZES;
471  libhifir_impl::hifir(*hif->M, hif->A, b, nirs, betas, trans, rnk, x,
472  ir_status);
473  }
474  } catch (const std::exception &e) {
475  _LHF_RETURN_FROM_HIFIR_ERROR(e);
476  }
477  return LHF_SUCCESS;
478 }
