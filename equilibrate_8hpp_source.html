65 template <typename _Integer, typename _ValReal, typename _Real>
66 inline void compute(const _Integer n, const _Integer ne, const _Integer *indptr,
67  const _Integer *indices, const _ValReal *vals, _Integer *no,
68  _Integer *cperm, _Integer liwork, _Integer *iwork,
69  _Integer lwork, _Real *work, const _Integer *pars,
70  _Integer *info) {
71  _Integer j, k;
72  _Real fact, rinf;
73  --cperm;
74  --indptr;
75  --vals;
76  --indices;
77  --iwork;
78  --work;
79  --pars;
80  --info;
81  rinf = sizeof(_Real) == 8u ? HUGE_VAL : HUGE_VALF;
82  if (n < 1) {
83  info[1] = -2;
84  info[2] = n;
85  return;
86  }
87  if (ne < 1) {
88  info[1] = -3;
89  info[2] = ne;
90  return;
91  }
92  if (liwork < n * 5) {
93  info[1] = -4;
94  info[2] = n * 5;
95  return;
96  }
97  if (lwork < n * 3 + ne) {
98  info[1] = -5;
99  info[2] = n * 3 + ne;
100  return;
101  }
102  if (pars[4] == 0) {
103  for (_Integer i = 1; i <= n; ++i) iwork[i] = 0;
104  for (j = 1; j <= n; ++j) {
105  const _Integer kstop = indptr[j + 1] - 1;
106  for (k = indptr[j]; k <= kstop; ++k) {
107  const _Integer node = indices[k];
108  if (node < 1 || node > n) {
109  info[1] = -6;
110  info[2] = j;
111  return;
112  }
113  if (iwork[node] == j) {
114  info[1] = -7;
115  info[2] = j;
116  return;
117  } else
118  iwork[node] = j;
119  }
120  }
121  }
122  for (int i = 1; i <= 10; ++i) info[i] = 0;
123  for (j = 1; j <= n; ++j) {
124  fact = 0;
125  const _Integer kstop = indptr[j + 1] - 1;
126  for (k = indptr[j]; k <= kstop; ++k) {
127  work[n * 3 + k] = std::abs(vals[k]);
128  if (work[n * 3 + k] > fact) fact = work[n * 3 + k];
129  }
130  work[(n << 1) + j] = fact;
131  fact = fact != 0 ? std::log(fact) : rinf / n;
132  for (k = indptr[j]; k <= kstop; ++k) {
133  if (work[n * 3 + k] != 0)
134  work[n * 3 + k] = fact - std::log(work[n * 3 + k]);
135  else
136  work[n * 3 + k] = rinf / n;
137  }
138  }
139  kernel(n, &indptr[1], &indices[1], &work[n * 3 + 1], &cperm[1], no, &iwork[1],
140  &iwork[n + 1], &iwork[(n << 1) + 1], &iwork[n * 3 + 1],
141  &iwork[(n << 2) + 1], &work[1], &work[n + 1]);
142  if (*no == n) {
143  for (j = 1; j <= n; ++j) {
144  if (work[(n << 1) + j] != 0)
145  work[n + j] -= std::log(work[(n << 1) + j]);
146  else
147  work[n + j] = 0;
148  }
149  }
150  fact = std::log(rinf) * .5;
151  for (j = 1; j <= n; ++j) {
152  if (work[j] < fact && work[n + j] < fact) continue;
153  info[1] = 2;
154  break;
155  }
156  if (info[1] == 0 && *no < n) info[1] = 1;
157 }

159 // if flag equals 1, node I is pushed upwards
160 // if flag equals 0, node I is pushed downwards
161 template <typename _Integer, typename _Real>
162 inline void kernel_(const _Integer j, const _Integer n, _Integer *q, _Real *d__,
163  _Integer *l, const bool flag) {
164  _Integer qk, pos, pos_h;
165  --l;
166  --d__;
167  --q;
168  pos = l[j];
169  // pos is index of current position of node I in the tree
170  if (pos > 1) {
171  const _Real dj = d__[j];
172  if (flag) {
173  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
174  pos_h = pos / 2;
175  qk = q[pos_h];
176  if (dj <= d__[qk]) break;
177  q[pos] = qk;
178  l[qk] = pos;
179  pos = pos_h;
180  if (pos <= 1) break;
181  }
182  } else {
183  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
184  pos_h = pos / 2;
185  qk = q[pos_h];
186  if (dj >= d__[qk]) break;
187  q[pos] = qk;
188  l[qk] = pos;
189  pos = pos_h;
190  if (pos <= 1) break;
191  }
192  }
193  }
194  q[pos] = j;
195  l[j] = pos;
196 }

198 // delete root node from the binary heap
199 template <typename _Integer, typename _Real>
200 inline void kernel__(_Integer *ql, const _Integer n, _Integer *q, _Real *d__,
201  _Integer *l, const bool flag) {
202  _Real dk, dr;
203  _Integer pos, pos2;
204  // const _Integer n = *n;
205  --l;
206  --d__;
207  --q;
208  const _Integer j = q[*ql];
209  const _Real dj = d__[j];
210  --(*ql);
211  pos = 1;
212  if (flag) {
213  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
214  pos2 = pos << 1;
215  if (pos2 > *ql) break;
216  dk = d__[q[pos2]];
217  if (pos2 < *ql) {
218  dr = d__[q[pos2 + 1]];
219  if (dk < dr) {
220  ++pos2;
221  dk = dr;
222  }
223  }
224  if (dj >= dk) break;
225  // Exchange old last element with smaller child
226  q[pos] = q[pos2];
227  l[q[pos]] = pos;
228  pos = pos2;
229  }
230  }
231  // the point is never reached
232  else {
233  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
234  pos2 = pos << 1;
235  if (pos2 > *ql) break;
236  dk = d__[q[pos2]];
237  if (pos2 < *ql) {
238  dr = d__[q[pos2 + 1]];
239  if (dk > dr) {
240  ++pos2;
241  dk = dr;
242  }
243  }
244  if (dj <= dk) break;
245  q[pos] = q[pos2];
246  l[q[pos]] = pos;
247  pos = pos2;
248  }
249  }
250  q[pos] = j;
251  l[j] = pos;
252 }

254 // move last element in the heap
255 template <typename _Integer, typename _Real>
256 inline void kernel___(_Integer *pos_, _Integer *qlen, const _Integer n,
257  _Integer *q, _Real *d__, _Integer *l, const bool flag) {
258  _Real dk, dr;
259  _Integer qk, pos, posk;
260  --l;
261  --d__;
262  --q;
263  const _Integer pos_in = *pos_;
264  if (*qlen == pos_in) {
265  --(*qlen);
266  return;
267  }
268  const _Integer j = q[*qlen];
269  const _Real dj = d__[j];
270  --(*qlen);
271  pos = pos_in;
272  if (flag) {
273  if (pos > 1)
274  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
275  posk = pos / 2;
276  qk = q[posk];
277  if (dj <= d__[qk]) break;
278  q[pos] = qk;
279  l[qk] = pos;
280  pos = posk;
281  if (pos <= 1) break;
282  }
283  q[pos] = j;
284  l[j] = pos;
285  if (pos != pos_in) return;
286  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
287  posk = pos << 1;
288  if (posk > *qlen) break;
289  dk = d__[q[posk]];
290  if (posk < *qlen) {
291  dr = d__[q[posk + 1]];
292  if (dk < dr) {
293  ++posk;
294  dk = dr;
295  }
296  }
297  if (dj >= dk) break;
298  qk = q[posk];
299  q[pos] = qk;
300  l[qk] = pos;
301  pos = posk;
302  }
303  } else {
304  if (pos > 1) {
305  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
306  posk = pos / 2;
307  qk = q[posk];
308  if (dj >= d__[qk]) break;
309  q[pos] = qk;
310  l[qk] = pos;
311  pos = posk;
312  if (pos <= 1) break;
313  }
314  }
315  q[pos] = j;
316  l[j] = pos;
317  if (pos != pos_in) return;
318  for (_Integer __i__ = 1; __i__ <= n; ++__i__) {
319  posk = pos << 1;
320  if (posk > *qlen) break;
321  dk = d__[q[posk]];
322  if (posk < *qlen) {
323  dr = d__[q[posk + 1]];
324  if (dk > dr) {
325  ++posk;
326  dk = dr;
327  }
328  }
329  if (dj <= dk) break;
330  qk = q[posk];
331  q[pos] = qk;
332  l[qk] = pos;
333  pos = posk;
334  }
335  }
336  q[pos] = j;
337  l[j] = pos;
338 }

340 // clean up function to be used in the kernel
341 template <typename _Integer, typename _Real>
342 void cleanup(const _Integer n, _Integer *num, _Integer *iperm, _Integer *jperm,
343  const _Integer *irn, _Integer *out, const _Real *a, _Real *d__,
344  _Real *u) {
345  _Integer i__, j, k;
346 
347  // store dual column variable u in d(1:n)
348  for (j = 1; j <= n; ++j) {
349  k = jperm[j];
350  if (k != 0)
351  d__[j] = a[k] - u[irn[k]];
352  else
353  d__[j] = 0;
354  if (iperm[j] == 0) u[j] = 0;
355  }
356  if (*num == n) return;
357  // else, the matrix is structurally singular, complete iperm
358  // jperm, out are work arrays
359  for (j = 1; j <= n; ++j) jperm[j] = 0;
360  k = 0;
361  for (i__ = 1; i__ <= n; ++i__) {
362  if (iperm[i__] == 0) {
363  ++k;
364  out[k] = i__;
365  } else {
366  j = iperm[i__];
367  jperm[j] = i__;
368  }
369  }
370  k = 0;
371  for (j = 1; j <= n; ++j) {
372  if (jperm[j] != 0) continue;
373  ++k;
374  iperm[out[k]] = -j;
375  }
376 }

378 // main loop
379 template <typename _Integer, typename _Real>
380 inline void kernel(const _Integer n, const _Integer *ip, const _Integer *irn,
381  const _Real *a, _Integer *iperm, _Integer *num,
382  _Integer *jperm, _Integer *out, _Integer *pr, _Integer *q,
383  _Integer *l, _Real *u, _Real *d__) {
384  _Integer i__, j, k, i0, k0, k1;
385  _Real di;
386  _Integer ii, jj, kk;
387  _Real vj;
388  _Integer up;
389  _Real dq0;
390  _Integer kk1;
391  _Real csp;
392  _Integer isp, jsp, low;
393  _Real dmin_, dnew;
394  _Integer jord, qlen;
395  _Real rinf;
396  _Integer lpos;
397  --d__;
398  --u;
399  --l;
400  --q;
401  --pr;
402  --out;
403  --jperm;
404  --iperm;
405  --ip;
406  --a;
407  --irn;
408  // initialize
409  rinf = sizeof(_Real) == 8u ? HUGE_VAL : HUGE_VALF;
410  *num = 0;
411  for (k = 1; k <= n; ++k) {
412  u[k] = rinf;
413  d__[k] = 0.f;
414  iperm[k] = 0;
415  jperm[k] = 0;
416  pr[k] = ip[k];
417  l[k] = 0;
418  }
419  // initialize u
420  for (j = 1; j <= n; ++j) {
421  const _Integer kstop = ip[j + 1] - 1;
422  for (k = ip[j]; k <= kstop; ++k) {
423  i__ = irn[k];
424  if (a[k] > u[i__]) continue;
425  u[i__] = a[k];
426  iperm[i__] = j;
427  l[i__] = k;
428  }
429  }
430  for (i__ = 1; i__ <= n; ++i__) {
431  j = iperm[i__];
432  if (j == 0) continue;
433  // row i is not empty
434  iperm[i__] = 0;
435  if (jperm[j] != 0) continue;
436  // don't choose cheap assignment from dense columns
437  if (ip[j + 1] - ip[j] > n / 10 && n > 50) continue;
438  // assignment of column j to row i
439  ++(*num);
440  iperm[i__] = j;
441  jperm[j] = l[i__];
442  }
443  if (*num == n) {
444  cleanup(n, num, iperm, jperm, irn, out, a, d__, u);
445  return;
446  }
447  // scan unassigned columns
448  for (j = 1; j <= n; ++j) {
449  if (jperm[j] != 0) continue; // only when j is already assigned
450  k1 = ip[j];
451  const _Integer kstop = ip[j + 1] - 1;
452  // continue only if j is not empty
453  if (k1 > kstop) continue;
454  // allow NaNs
455  i0 = irn[k1];
456  vj = a[k1] - u[i0];
457  k0 = k1;
458  for (k = k1 + 1; k <= kstop; ++k) {
459  i__ = irn[k];
460  di = a[k] - u[i__];
461  if (di > vj) continue;
462  if (di < vj || di == rinf) {
463  vj = di;
464  i0 = i__;
465  k0 = k;
466  continue;
467  }
468  if (iperm[i__] != 0 || iperm[i0] == 0) continue;
469  vj = di;
470  i0 = i__;
471  k0 = k;
472  }
473  d__[j] = vj;
474  k = k0;
475  i__ = i0;
476  if (iperm[i__] == 0) {
477  ++(*num);
478  jperm[j] = k;
479  iperm[i__] = j;
480  pr[j] = k + 1;
481  continue;
482  }
483  for (k = k0; k <= kstop; ++k) {
484  i__ = irn[k];
485  if (a[k] - u[i__] > vj) continue;
486  jj = iperm[i__];
487  // scan remaining part of assigned column jj
488  kk1 = pr[jj];
489  const _Integer kkstop = ip[jj + 1] - 1;
490  if (kk1 > kkstop) continue;
491  bool bb = false;
492  for (kk = kk1; kk <= kkstop; ++kk) {
493  ii = irn[kk];
494  if (iperm[ii] > 0) continue;
495  if (a[kk] - u[ii] <= d__[jj]) {
496  jperm[jj] = kk;
497  iperm[ii] = jj;
498  pr[jj] = kk + 1;
499  ++(*num);
500  jperm[j] = k;
501  iperm[i__] = j;
502  pr[j] = k + 1;
503  bb = true;
504  break;
505  }
506  }
507  if (bb) break;
508  pr[jj] = kkstop + 1;
509  }
510  }
511  if (*num == n) {
512  cleanup(n, num, iperm, jperm, irn, out, a, d__, u);
513  return;
514  }
515  // prepare for the main loop
516  for (i__ = 1; i__ <= n; ++i__) {
517  d__[i__] = rinf;
518  l[i__] = 0;
519  }
520 
521  // main loop: similar to Dijkstra's algorithm
522  for (jord = 1; jord <= n; ++jord) {
523  if (jperm[jord] != 0) continue;
524  dmin_ = rinf;
525  qlen = 0;
526  low = n + 1;
527  up = n + 1;
528  csp = rinf;
529  j = jord;
530  pr[j] = -1;
531  // scan column j
532  const _Integer kstop = ip[j + 1] - 1;
533  for (k = ip[j]; k <= kstop; ++k) {
534  i__ = irn[k];
535  dnew = a[k] - u[i__];
536  if (dnew >= csp) continue;
537  if (iperm[i__] == 0) {
538  csp = dnew;
539  isp = k;
540  jsp = j;
541  } else {
542  if (dnew < dmin_) dmin_ = dnew;
543  d__[i__] = dnew;
544  ++qlen;
545  q[qlen] = k;
546  }
547  }
548  // initialize heap q and q2 with rows held in q(1:qlen)
549  const _Integer kkstop = qlen;
550  qlen = 0;
551  for (kk = 1; kk <= kkstop; ++kk) {
552  k = q[kk];
553  i__ = irn[k];
554  if (csp <= d__[i__]) {
555  d__[i__] = rinf;
556  continue;
557  }
558  if (d__[i__] <= dmin_) {
559  --low;
560  q[low] = i__;
561  l[i__] = low;
562  } else {
563  ++qlen;
564  l[i__] = qlen;
565  kernel_(i__, n, &q[1], &d__[1], &l[1], false);
566  }
567  // update tree
568  jj = iperm[i__];
569  out[jj] = k;
570  pr[jj] = j;
571  }
572  const _Integer jstop = *num;
573  // if q2 is empty, extract rows from q
574  for (_Integer __j_ = 1; __j_ <= jstop; ++__j_) {
575  if (low == up) {
576  if (qlen == 0) break;
577  i__ = q[1];
578  if (d__[i__] >= csp) break;
579  dmin_ = d__[i__];
580  for (;;) {
581  kernel__(&qlen, n, &q[1], &d__[1], &l[1], false);
582  --low;
583  q[low] = i__;
584  l[i__] = low;
585  if (qlen == 0) break;
586  i__ = q[1];
587  if (d__[i__] > dmin_) break;
588  }
589  }
590  // q0 is the row whose cost d(q0) is the smallest
591  const _Integer q0 = q[up - 1];
592  dq0 = d__[q0];
593  // exit loop if path to q0 is no longer the shortest
594  if (dq0 >= csp) break;
595  --up;
596  // scan column matching q0
597  j = iperm[q0];
598  vj = dq0 - a[jperm[j]] + u[q0];
599  const _Integer kstop = ip[j + 1] - 1;
600  for (k = ip[j]; k <= kstop; ++k) {
601  i__ = irn[k];
602  if (l[i__] >= up) continue;
603  dnew = vj + a[k] - u[i__]; // dnew is the new cost
604  if (dnew >= csp)
605  continue; // if dnew >= cost of shortest path, do not update
606  // row i is unmatched, update shortest path
607  if (iperm[i__] == 0) {
608  csp = dnew;
609  isp = k;
610  jsp = j;
611  } else {
612  // row i is matched, do not update d(i) if dnew is larger
613  di = d__[i__];
614  if (di <= dnew) continue;
615  if (l[i__] >= low) continue;
616  d__[i__] = dnew;
617  if (dnew <= dmin_) {
618  lpos = l[i__];
619  if (lpos != 0) {
620  kernel___(&lpos, &qlen, n, &q[1], &d__[1], &l[1], false);
621  }
622  --low;
623  q[low] = i__;
624  l[i__] = low;
625  } else {
626  if (l[i__] == 0) {
627  ++qlen;
628  l[i__] = qlen;
629  }
630  kernel_(i__, n, &q[1], &d__[1], &l[1], false);
631  }
632  // update tree
633  jj = iperm[i__];
634  out[jj] = k;
635  pr[jj] = j;
636  }
637  }
638  }
639  // if csp = rinf, no augmenting path is found
640  if (csp != rinf) {
641  // find augmenting path by tracing back in pr;
642  // update iperm, jperm
643  ++(*num);
644  i__ = irn[isp];
645  iperm[i__] = jsp;
646  jperm[jsp] = isp;
647  j = jsp;
648  const _Integer jstop_ = *num;
649  for (_Integer __j_ = 1; __j_ <= jstop_; ++__j_) {
650  jj = pr[j];
651  if (jj == -1) break;
652  k = out[j];
653  i__ = irn[k];
654  iperm[i__] = jj;
655  jperm[jj] = k;
656  j = jj;
657  }
658  // update u for rows in q(up:n)
659  for (kk = up; kk <= n; ++kk) {
660  i__ = q[kk];
661  u[i__] = u[i__] + d__[i__] - csp;
662  }
663  }
664  for (kk = low; kk <= n; ++kk) {
665  i__ = q[kk];
666  d__[i__] = rinf;
667  l[i__] = 0;
668  }

718

720  /// \param[in] n number of nodes
721  /// \param[in] nnz number of nonzeros
722  /// \param[in] indptr index pointer array
723  /// \param[in] indices index array
724  /// \param[in] vals value array
725  /// \param[out] perm permutation result
726  /// \return information
727  inline index_type compute(index_type n, index_type nnz,
728  const index_type *indptr, const index_type *indices,
729  const value_type *vals, index_type *perm) const {

737  lwork = _work.size();

739  (value_type *)vals, &num, perm, liwork,
740  (index_type *)_iwork.data(), lwork,
741  (real_type *)_work.data(), pars, info);
742  return info[0];
743  }

745  /// \brief equilibrate a CRS/CCS matrix with container inputs
746  /// \param[in] indptr index pointer array
747  /// \param[in] indices index array
748  /// \param[in] vals value array
749  /// \param[out] perm permutation result
750  /// \return information

752  const iarray_type &indices, const varray_type &vals,
753  iarray_type &perm) const {

756  if (perm.size() + 1 < indptr.size()) perm.resize(indptr.size() - 1);
757  return compute(indptr.size() - 1, indices.size(), indptr.data(),

761  }
