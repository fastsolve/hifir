105  /// \brief compute decomposition and determine the rank
106  /// \param[in] opts control parameters, see \ref Options
107  inline void factorize(const Options &opts) {
108  // get tolerance
109  const static scalar_type diag_tol = std::sqrt(Const<scalar_type>::EPS);
110  const static scalar_type cond_tol =
111  scalar_type(1) / std::pow(Const<scalar_type>::EPS, 2. / 3);
112 
113  hif_error_if(_mat.empty(), "matrix is still empty!");
114  hif_error_if(_mat.nrows() < _mat.ncols(),
115  "matrix must have no smaller row size");
116 
117  const scalar_type cond_thres =
118  opts.rrqr_cond <= 0.0 ? cond_tol : scalar_type(opts.rrqr_cond);
119 
120  if (hif_verbose(INFO, opts))
121  hif_info("factorizing dense level by RRQR with cond-thres %g...",
122  (double)cond_thres);
123 
124  _jpvt.resize(_mat.ncols());
125  // NOTE we need to initialize jpvt to zero
126  std::fill(_jpvt.begin(), _jpvt.end(), 0);
127  _tau.resize(std::min(_mat.nrows(), _mat.ncols()));
128 
129  // query the buffer size
130  value_type lwork;
131  lapack_kernel::geqp3(_mat.nrows(), _mat.ncols(), _mat.data(), _mat.nrows(),
132  _jpvt.data(), _tau.data(), &lwork, -1, nullptr);
133  _work.resize((size_type)abs(lwork));
134 
135  // allocate space for complex arith
136  if (!lapack_kernel::IS_REAL) _rwork.resize(2 * _mat.ncols());
137 
138  const auto info = lapack_kernel::geqp3(
139  _mat.nrows(), _mat.ncols(), _mat.data(), _mat.nrows(), _jpvt.data(),
140  _tau.data(), _work.data(), (size_type)abs(lwork), _rwork.data());
141  hif_error_if(info < 0, "GEQP3 returned negative info");
142 
143  // fast filtering to test diagonal entries
144  bool do_cond_test = false;
145  const size_type N = _tau.size(); // min (row, col)
146  // we first check if we have too small diagonals, use this as a safe guard
147  // to see if we need to call more expensive condition # estimator
148  // weighted by R(0,0)
149  const scalar_type diag_eps = diag_tol * std::abs(_mat[0]);
150  for (size_type i = N; i != 0u; --i)
151  if (std::abs(_mat(i - 1, i - 1)) < diag_eps) {
152  hif_warning(
153  "\n System is ill-conditioned (diagonal %zd is smaller\n"
154  " than tolerance %g), will switch to condition number\n"
155  " estimator to determine the final numerical rank.",
156  i, (double)diag_eps);
157  do_cond_test = true;
158  break;
159  }
160  // initial rank to be full rank
161  _rank = _mat.ncols();
162  if (do_cond_test) {
163 #ifdef HIF_TQRCP_USE_1NORM
164  static const char *nrm_sig = "1";
165  _est_rank(cond_thres <= 0.0 ? cond_tol : scalar_type(cond_thres));
166 #else
167  static const char *nrm_sig = "2";
168  _est_rank_2norm(cond_thres);
169 #endif
170  hif_warning_if(_rank != _mat.ncols(),
171  "\n The system is rank deficient with rank=%zd,\n"
172  " the tolerance used was %g, comparing wrt\n"
173  " %s-norm-based condition number estimation.",
174  _rank, (double)cond_thres, nrm_sig);
175  }
176  }

188  /// \brief solve system
189  /// \param[in,out] x input rhs, output solution
190  /// \param[in] rank (optional) rank for back solve, default is \a _rank
191  /// \param[in] tran (optional) transpose flag, default is false
192  ///
193  /// First, QRCP returns \f$\mathbf{AP}=\mathbf{QR}\f$, thus when we
194  /// have \f$\mathbf{Ax}=\mathbf{b}\f$, the derivation is:
195  ///
196  /// \f{eqnarray*}{
197  /// \mathbf{QRP}^T\mathbf{x}&=&\mathbf{b} \\
198  /// \hookrightarrow\mathbf{RP}^T\mathbf{x}&=&\mathbf{Q}^T\mathbf{b} \\
199  /// \hookrightarrow\mathbf{P}^T\mathbf{x}&=&\mathbf{R}^{-1}
200  /// \mathbf{Q}^T\mathbf{b} \\
201  /// \hookrightarrow\mathbf{x}&=&\mathbf{PR}^{-1}\mathbf{Q}^T\mathbf{b}
202  /// \f}
203  ///
204  /// Notice that \f$\mathbf{R}^{-1}\mathbf{Q}^T\f$ is the
205  /// pseudo-inverse of \f$\mathbf{AP}\f$.
206  inline void solve(Array<value_type> &x, const size_type rank = 0u,
207  const bool tran = false) const {
208  hif_error_if(x.size() != _mat.nrows(),
209  "unmatched sizes between system and rhs");
210  !tran ? _solve_nt<1>(x.data(), rank) : _solve_t<1>(x.data(), rank);
211  }

213  /// \brief wrapper if input is different from \a value_type

216  const bool tran = false) const {

218  std::copy(x.cbegin(), x.cend(), _x.begin());

223  /// \brief wrapper for solving multiple RHS with \a row-major storage
224  template <class V, size_type Nrhs>
225  inline void solve_mrhs(Array<std::array<V, Nrhs>> &x,
226  const size_type rank = 0u,
227  const bool tran = false) const {
228  hif_error_if(x.size() != _mat.nrows(),
229  "unmatched sizes between system and rhs");
230  _mrhs.resize(x.size(), Nrhs);
231  for (size_type j = 0; j < Nrhs; ++j)
232  for (size_type i(0); i < x.size(); ++i) _mrhs(i, j) = x[i][j];
233  !tran ? _solve_nt<Nrhs>(_mrhs.data(), rank)
234  : _solve_t<Nrhs>(_mrhs.data(), rank);
235  for (size_type j = 0; j < Nrhs; ++j)
236  for (size_type i(0); i < x.size(); ++i) x[i][j] = _mrhs(i, j);
237  }

280  /// \brief Incrementally estimate the condition number of R
281  /// \param[in] cond_tol condition number threshold
282  /// \note We incrementally check based on \f$\mathbf{R}^{T}\f$ by 1-norm
283  inline void _est_rank(const scalar_type cond_tol) {
284  const size_type n = _mat.ncols();
285  _work.resize(n); // allocate buffer, used in _est_cond1
286  scalar_type nrm, inrm;
287  for (_rank = 0u; _rank < n; ++_rank) {
288  // esimate norm of the current leading block
289  if (_mat(_rank, _rank) == 0) break;
290  _est_cond1(inrm, nrm);
291  if (inrm * nrm >= cond_tol) break;
292  }
293  }
294 
295  /// \brief helper function to estimate the norms of \f$\mathbf{R}^T\f$ and
296  /// \f$\mathbf{R}^{-T}\f$.
297  /// \param[in,out] inrm invser norm at step _rank
298  /// \param[in,out] nrm norm at step _rank
299  inline void _est_cond1(scalar_type &inrm, scalar_type &nrm) {
300  // NOTE, the upper part in _mat is R
301  // NOTE, keep in mind that we are dealing with $R^{T}$
302  if (_rank == 0u) {
303  _work[0] = value_type(1) / _mat(0, 0);
304  inrm = abs(_work[0]);
305  nrm = abs(_mat(0, 0));
306  } else {
307  value_type s(0);
308  scalar_type nn(0);
309  // backsolve
310  for (size_type j(0); j < _rank; ++j) {
311  s += _work[j] * _mat(j, _rank);
312  nn += abs(_mat(j, _rank));
313  }
314  const value_type k1 = value_type(1) - s, k2 = -value_type(1) - s;
315  _work[_rank] =
316  abs(k1) < abs(k2) ? k2 / _mat(_rank, _rank) : k1 / _mat(_rank, _rank);
317  // compute the invser 1-norm for R^T_{0:rank,0:rank}
318  inrm = std::max(inrm, abs(_work[_rank]));
319  // compute the 1-norm for R^T_{0:rank,0:rank}
320  nrm = std::max(nrm, nn + abs(_mat(_rank, _rank)));
321  }
322  }

324  /// \brief esimate rank based on 2-norm condition numbers
325  /// \param[out] cond_tol condition number threshold
326  /// \note This is calling after QRCP by using \a ?laic1 auxiliary routine
327  /// \note This implemention is adapted based on ?gelsy driver routine
328  inline void _est_rank_2norm(const scalar_type cond_tol) {
329  static constexpr typename lapack_kernel::int_type JOB_MIN = 2, JOB_MAX = 1;
330 
331  const size_type n = _mat.ncols();
332  _work.resize(2 * n); // allocate buffer
333  auto *x = _work.data(), *y = x + n; // used in ?laic1
334  x[0] = y[0] = value_type(1);
335  scalar_type smax(abs(_mat(0, 0))), smin(smax); // initial singular values
336  value_type s1, c1, s2, c2; // needed in ?laic1
337  scalar_type sminpr, smaxpr;
338  for (_rank = 0u; _rank < n; ++_rank) {
339  // esimate smaller
340  lapack_kernel::laic1(JOB_MIN, _rank, x, smin, &_mat(0, _rank),
341  _mat(_rank, _rank), sminpr, s1, c1);
342  // esimate larger
343  lapack_kernel::laic1(JOB_MAX, _rank, y, smax, &_mat(0, _rank),
344  _mat(_rank, _rank), smaxpr, s2, c2);
345  if (smaxpr <= sminpr * cond_tol) {
346  // still well-conditioned
347  for (size_type i(0); i < _rank; ++i) {
348  x[i] *= s1;
349  y[i] *= s2;
350  }
351  x[_rank] = c1;
352  y[_rank] = c2;
353  smin = sminpr;
354  smax = smaxpr;
355  continue;
356  }
357  break;
358  }
359  }

361  /// \brief Solve with constant number of right-hand sides
362  /// \tparam Nrhs Number of RHS
363  /// \param[in,out] x input rhs, output solution
364  /// \param[in] rank (optional) rank for back solve, default is \a _rank
365  template <size_type Nrhs>
366  inline void _solve_nt(value_type *x, const size_type rank = 0u) const {
367  hif_error_if(
368  _mat.empty() || _jpvt.empty(),
369  "either the matrix is not set or the factorization has not yet done!");
370  // determine rank, assume row size is no smaller than column size
371  const size_type rk =
372  rank == 0u ? _rank : (rank > _mat.nrows() ? _mat.nrows() : rank);
373  // query the optimal work space
374  value_type lwork;
375  lapack_kernel::ormqr('L', 'T', _mat.nrows(), Nrhs, rk, _mat.data(),
376  _mat.nrows(), _tau.data(), x, _mat.nrows(), &lwork,
377  -1);
378  _work.resize((size_type)abs(lwork));
379 
380  // compute x=Q(:,1:_rank)'*x
381  const auto info = lapack_kernel::ormqr(
382  'L', 'T', _mat.nrows(), Nrhs, rk, _mat.data(), _mat.nrows(),
383  _tau.data(), x, _mat.nrows(), _work.data(), (size_type)abs(lwork));
384  hif_error_if(info < 0, "ORMQR returned negative info (Q'*x)");
385 
386  // compute x(1:_rank)=inv(R(1:_rank,1:_rank))*x(1:_rank)
387  for (size_type i = 0; i < Nrhs; ++i)
388  lapack_kernel::trsv('U', 'N', 'N', rk, _mat.data(), _mat.nrows(),
389  x + i * _mat.nrows(), 1);
390 
391  // permutation, need to loop thru all entries in jpvt. Also, note that jpvt
392  // is column pivoting, thus we actually need the inverse mapping while
393  // performing P*x. It's worth noting that, unlike ipiv in LU routine, jpvt
394  // is a real permutation vector, i.e. not in-place swapable. Therefore, we
395  // need a buffer.
396  //
397  // for normal mapping, we have x=b(p), for inverse, we need x(p)=b
398  const size_type n = _jpvt.size();
399  _work.resize(_mat.nrows());
400  for (size_type i = 0; i < Nrhs; ++i) {
401  value_type *xi = x + i * _mat.nrows();
402  for (size_type i = 0u; i < rk; ++i) _work[_jpvt[i] - 1] = xi[i];
403  for (size_type i = rk; i < n; ++i) _work[_jpvt[i] - 1] = value_type(0);
404  std::copy_n(_work.cbegin(), n, xi);
405  }
406  }

408  /// \brief Solve with constant number of right-hand sides (Hermitian)
409  /// \tparam Nrhs Number of RHS
410  /// \param[in,out] x input rhs, output solution
411  /// \param[in] rank (optional) rank for back solve, default is \a _rank
412  template <size_type Nrhs>
413  inline void _solve_t(value_type *x, const size_type rank = 0u) const {
414  hif_error_if(
415  _mat.empty() || _jpvt.empty(),
416  "either the matrix is not set or the factorization has not yet done!");
417  // determine rank, assume row size is no smaller than column size
418  const size_type rk =
419  rank == 0u ? _rank : (rank > _mat.nrows() ? _mat.nrows() : rank);
420 
421  // permutation P^T
422  const size_type n = _jpvt.size();
423  _work.resize(_mat.nrows());
424  for (size_type i = 0; i < Nrhs; ++i) {
425  value_type *xi = x + i * _mat.nrows();
426  for (size_type i = 0u; i < rk; ++i) _work[i] = xi[_jpvt[i] - 1];
427  for (size_type i = rk; i < n; ++i) _work[i] = value_type(0);
428  std::copy_n(_work.cbegin(), n, xi);
429  }
430 
431  // compute x(1:_rank)=inv(R(1:_rank,1:_rank))'*x(1:_rank)
432  for (size_type i = 0; i < Nrhs; ++i)
433  lapack_kernel::trsv('U', 'C', 'N', rk, _mat.data(), _mat.nrows(),
434  x + i * _mat.nrows(), 1);
435 
436  // query the optimal work space
437  value_type lwork;
438  lapack_kernel::ormqr('L', 'N', _mat.nrows(), Nrhs, rk, _mat.data(),
439  _mat.nrows(), _tau.data(), x, _mat.nrows(), &lwork,
440  -1);
441  _work.resize((size_type)abs(lwork));
442 
443  // compute x=Q(:,1:_rank)*x
444  const auto info = lapack_kernel::ormqr(
445  'L', 'N', _mat.nrows(), Nrhs, rk, _mat.data(), _mat.nrows(),
446  _tau.data(), x, _mat.nrows(), _work.data(), (size_type)abs(lwork));
447  hif_error_if(info < 0, "ORMQR returned negative info (Q*x)");
448  }

450  /// \brief Matrix-vector with constant number of right-hand sides
451  /// \tparam Nrhs Number of RHS
452  /// \param[in,out] x input rhs, output solution
453  /// \param[in] rank (optional) rank for back solve, default is \a _rank
454  template <size_type Nrhs>
455  inline void _multiply_nt(value_type *x, const size_type rank = 0u) const {
456  hif_error_if(
457  _mat.empty() || _jpvt.empty(),
458  "either the matrix is not set or the factorization has not yet done!");
459  // determine rank, assume row size is no smaller than column size
460  const size_type rk =
461  rank == 0u ? _rank : (rank > _mat.nrows() ? _mat.nrows() : rank);
462 
463  // permutation P^T
464  const size_type n = _jpvt.size();
465  _work.resize(_mat.nrows());
466  for (size_type i = 0; i < Nrhs; ++i) {
467  value_type *xi = x + i * _mat.nrows();
468  for (size_type i = 0u; i < rk; ++i) _work[i] = xi[_jpvt[i] - 1];
469  for (size_type i = rk; i < n; ++i) _work[i] = value_type(0);
470  std::copy_n(_work.cbegin(), n, xi);
471  }
472 
473  // compute x(1:_rank)=R(1:_rank,1:_rank)*x(1:_rank)
474  for (size_type i = 0; i < Nrhs; ++i)
475  lapack_kernel::trmv('U', 'N', 'N', rk, _mat.data(), _mat.nrows(),
476  x + i * _mat.nrows(), 1);
477 
478  // query the optimal work space
479  value_type lwork;
480  lapack_kernel::ormqr('L', 'N', _mat.nrows(), Nrhs, rk, _mat.data(),
481  _mat.nrows(), _tau.data(), x, _mat.nrows(), &lwork,
482  -1);
483  _work.resize((size_type)abs(lwork));
484 
485  // compute x=Q(:,1:_rank)*x
486  const auto info = lapack_kernel::ormqr(
487  'L', 'N', _mat.nrows(), Nrhs, rk, _mat.data(), _mat.nrows(),
488  _tau.data(), x, _mat.nrows(), _work.data(), (size_type)abs(lwork));
489  hif_error_if(info < 0, "ORMQR returned negative info (Q*x)");
490  }

492  /// \brief Matrix-vector with constant number of right-hand sides (Hermitian)
493  /// \tparam Nrhs Number of RHS
494  /// \param[in,out] x input rhs, output solution
495  /// \param[in] rank (optional) rank for back solve, default is \a _rank
496  template <size_type Nrhs>
497  inline void _multiply_t(value_type *x, const size_type rank = 0u) const {
498  hif_error_if(
499  _mat.empty() || _jpvt.empty(),
500  "either the matrix is not set or the factorization has not yet done!");
501  // determine rank, assume row size is no smaller than column size
502  const size_type rk =
503  rank == 0u ? _rank : (rank > _mat.nrows() ? _mat.nrows() : rank);
504  // query the optimal work space
505  value_type lwork;
506  lapack_kernel::ormqr('L', 'T', _mat.nrows(), Nrhs, rk, _mat.data(),
507  _mat.nrows(), _tau.data(), x, _mat.nrows(), &lwork,
508  -1);
509  _work.resize((size_type)abs(lwork));
510 
511  // compute x=Q(:,1:_rank)'*x
512  const auto info = lapack_kernel::ormqr(
513  'L', 'T', _mat.nrows(), Nrhs, rk, _mat.data(), _mat.nrows(),
514  _tau.data(), x, _mat.nrows(), _work.data(), (size_type)abs(lwork));
515  hif_error_if(info < 0, "ORMQR returned negative info (Q'*x)");
516 
517  // compute x(1:_rank)=R(1:_rank,1:_rank)'*x(1:_rank)
518  for (size_type i = 0; i < Nrhs; ++i)
519  lapack_kernel::trmv('U', 'C', 'N', rk, _mat.data(), _mat.nrows(),
520  x + i * _mat.nrows(), 1);
521 
522  // permutation, need to loop thru all entries in jpvt. Also, note that jpvt
523  // is column pivoting, thus we actually need the inverse mapping while
524  // performing P*x. It's worth noting that, unlike ipiv in LU routine, jpvt
525  // is a real permutation vector, i.e. not in-place swapable. Therefore, we
526  // need a buffer.
527  //
528  // for normal mapping, we have x=b(p), for inverse, we need x(p)=b
529  const size_type n = _jpvt.size();
530  _work.resize(_mat.nrows());
531  for (size_type i = 0; i < Nrhs; ++i) {
532  value_type *xi = x + i * _mat.nrows();
533  for (size_type i = 0u; i < rk; ++i) _work[_jpvt[i] - 1] = xi[i];
534  for (size_type i = rk; i < n; ++i) _work[_jpvt[i] - 1] = value_type(0);
535  std::copy_n(_work.cbegin(), n, xi);
536  }
537  }
538 
539  protected:
540  dense_type _mat; ///< matrix
541  dense_type _mat_backup; ///< backup matrix
542  size_type _rank; ///< rank
543  mutable Array<value_type> _x; ///< buffer for handling solve in derived
544  mutable Array<value_type> _y; ///< buffer for handling mv in derived
545  mutable dense_type _mrhs; ///< multiple RHS buffer
546  Array<hif_lapack_int> _jpvt; ///< column pivoting array
547  Array<value_type> _tau; ///< tau array
548  mutable Array<value_type> _work; ///< work buffer
549  mutable Array<hif_lapack_int> _iwork; ///< integer work buffer
550  mutable Array<scalar_type> _rwork; ///< work space used in complex arith
551 };
