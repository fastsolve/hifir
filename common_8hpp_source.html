49 /// \brief apply binary search on a sorted array
50 /// \tparam Iter iterator type
51 /// \tparam ValueType value type that can be casted to the derefence of Iter
52 /// \param[in] first starting iterator
53 /// \param[in] last ending iterator
54 /// \param[in] v target value
55 /// \return pair<bool, Iter> that the \a first indicates if or not we find
56 /// v, while the \a second is the location
57 /// \todo replace the implementation with one-sided binary search
58 template <class Iter, class ValueType>
59 inline std::pair<bool, Iter> find_sorted(Iter first, Iter last,
60  const ValueType &v) {
61  // NOTE, we use the c++ std binary search for now
62  // WARNING! When replacing the impl, make sure the behavior aligns with
63  // standard lower_bound!
64  auto lower = std::lower_bound(first, last, v);
65  return std::make_pair(lower != last && (ValueType)*lower == v, lower);
66 }

68 /// \brief rotate a subset of vector, s.t. src appears to the left most pos
69 /// \tparam ArrayType input and output array type
70 /// \param[in] n **local** size of how many items will be shifted
71 /// \param[in] src index in **global** range
72 /// \param[in,out] v input and output array
73 /// \note Complexity: \f$\mathcal{O}(n)\f$
74 /// \sa rotate_right
75 template <class ArrayType>
76 inline void rotate_left(const typename ArrayType::size_type n,
77  const typename ArrayType::size_type src, ArrayType &v) {
78  auto itr_first = v.begin() + src, itr_last = itr_first + n;
79  std::rotate(itr_first, itr_first + 1, itr_last);
80 }

85 /// \param[in] src index in **global** range

88 /// \sa rotate_left

93  const typename ArrayType::size_type src,
94  ArrayType & v) {
95  typedef std::reverse_iterator<typename ArrayType::iterator> iterator;

97  iterator itr_first(v.begin() + src + 1), itr_last(itr_first + n);

151 /// \brief ensure consistent between data types of \a T and input array
152 /// \tparam T desired data type
153 /// \tparam ArrayType array container type, see \ref Array
154 /// \param[in] v input array
155 /// \param[in] copy_if_needed if \a true (default), then copy the values
156 ///
157 /// This helper function simply checks if \a sizeof(T) equals to that of
158 /// \a ArrayType::value_type, if not, then allocate new array with the same
159 /// size of \a v. Notice that the above condition should be used to deallocate
160 /// the memory.
161 template <class T, class ArrayType>
162 inline T *ensure_type_consistency(const ArrayType &v,
163  const bool copy_if_needed = true) {
164  constexpr static bool consist =
165  sizeof(T) == sizeof(typename ArrayType::value_type);
166  if (consist) return (T *)v.data();
167  T *ptr = new (std::nothrow) T[v.size()];
168  hif_error_if(!ptr, "memory allocation failed");
169  if (copy_if_needed) std::copy(v.cbegin(), v.cend(), ptr);
170  return ptr;
171 }
