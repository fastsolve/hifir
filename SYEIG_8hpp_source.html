105  /// \brief compute decomposition and determine the rank
106  /// \param[in] opts control parameters, see \ref Options
107  inline void factorize(const Options &opts) {
108  hif_error_if(_mat.empty(), "matrix is still empty!");
109  hif_error_if(!is_squared(), "the matrix must be squared!");
110  if (hif_verbose(INFO, opts))
111  hif_info("factorizing dense level by symmetric (%s) eigendecomp...",
112  opts.spd > 0 ? "PD" : (opts.spd < 0 ? "ND" : "ID"));
113 
114  const static scalar_type EPS = std::pow(Const<scalar_type>::EPS, 2. / 3);
115 
116  // esimate the workspace
117  if (!lapack_kernel::IS_REAL)
118  _rwork.resize(std::max(1, int(3 * _mat.nrows() - 2)));
119  value_type sz;
120  lapack_kernel::syev('L', _mat, _w, &sz, -1, _rwork);
121  const auto n = _mat.nrows();
122  _work.resize(std::max((size_type)abs(sz), n));
123 
124  // factorize
125  _w.resize(n);
126  auto info =
127  lapack_kernel::syev('L', _mat, _w, _work.data(), _work.size(), _rwork);
128  if (info > 0)
129  hif_error("?syev failed to converge with info=%d.", (int)info);
130  else if (info < 0)
131  hif_error("?syev's %d-th arg is illegal.", (int)-info);
132 
133  // handle truncation
134  _trunc_order.resize(n);
135  std::iota(_trunc_order.begin(), _trunc_order.end(), 0);
136  const scalar_type thres =
137  EPS * *std::max_element(_w.cbegin(), _w.cend(),
138  [](const value_type a, const value_type b) {
139  return std::abs(a) < std::abs(b);
140  });
141  _rank = n;
142  if (opts.spd > 0) {
143  // ensure pd-ness
144  for (int i(n - 1); i > -1; --i)
145  if (_w[i] <= 0.0 || std::abs(_w[i]) <= thres)
146  --_rank;
147  else
148  break;
149  } else if (opts.spd < 0) {
150  // ensure nd-ness
151  // reverse the order for truncations
152  std::reverse(_trunc_order.begin(), _trunc_order.end());
153  for (size_type i(0); i < n; ++i)
154  if (_w[i] >= 0.0 || std::abs(_w[i]) <= thres)
155  --_rank;
156  else
157  break;
158  } else {
159  // indefinite
160  // sort based on magnitude of the eigenvalues
161  std::sort(_trunc_order.begin(), _trunc_order.end(),
162  [&](const int a, const int b) {
163  return std::abs(_w[a]) > std::abs(_w[b]);
164  });
165  for (int i(n - 1); i > -1; --i)
166  if (std::abs(_w[_trunc_order[i]]) <= thres)
167  --_rank;
168  else
169  break;
170  }
171  if (_rank != n)
172  hif_warning("\n System is ill-conditioned with rank=%d, dim=%d.",
173  (int)_rank, (int)n);
174  }

186  /// \brief solve \f$\mathbf{Q\Lambda Q}^H\mathbf{x}=\mathbf{b}\f$
187  /// \param[in,out] x input rhs, output solution
188  /// \param[in] rank (optional) numerical rank
189  inline void solve(Array<value_type> &x, const size_type rank = 0) const {
190  hif_error_if(
191  _mat.empty() || _w.empty(),
192  "either the matrix is not set or the factorization has not yet done!");
193  hif_error_if(x.size() != _mat.nrows(),
194  "unmatched sizes between system and rhs");
195  const auto n = x.size();
196  const size_type rk = rank == 0u ? _rank : (rank > n ? n : rank);
197  // std::copy(x.cbegin(), x.cend(), _y.begin());
198  // step 1, compute y=Q^T*x
199  lapack_kernel::gemv('C', value_type(1), _mat, x, value_type(0), _work);
200  // step 2, solve inv(lambda)*y with truncation
201  for (size_type i(0); i < rk; ++i)
202  _work[_trunc_order[i]] /= _w[_trunc_order[i]];
203  for (size_type i(rk); i < n; ++i) _work[_trunc_order[i]] = 0;
204  // step 3, compute x=Q*y
205  lapack_kernel::gemv('N', value_type(1), _mat, _work, value_type(0), x);
206  }

217  /// \brief solve with multiple RHS
218  /// \sa solve
219  template <class V, size_type Nrhs>
220  inline void solve_mrhs(Array<std::array<V, Nrhs>> &x,
221  const size_type rank = 0) const {
222  hif_error_if(
223  _mat.empty() || _w.empty(),
224  "either the matrix is not set or the factorization has not yet done!");
225  hif_error_if(x.size() != _mat.nrows(),
226  "unmatched sizes between system and rhs");
227  const auto n = x.size();
228  const size_type rk = rank == 0u ? _rank : (rank > n ? n : rank);
229  // copy to internal column-major buffer
230  _mrhs.resize(x.size(), Nrhs);
231  for (size_type j = 0; j < Nrhs; ++j)
232  for (size_type i(0); i < x.size(); ++i) _mrhs(i, j) = x[i][j];
233  for (size_type i = 0; i < Nrhs; ++i) {
234  // step 1, compute y=Q^T*x
235  lapack_kernel::gemv('C', _mat.nrows(), _mat.ncols(), value_type(1),
236  _mat.data(), _mat.nrows(), &_mrhs(0, i * Nrhs),
237  value_type(0), _work.data());
238  // step 2, solve inv(lambda)*y with truncation
239  for (size_type i(0); i < rk; ++i)
240  _work[_trunc_order[i]] /= _w[_trunc_order[i]];
241  for (size_type i(rk); i < n; ++i) _work[_trunc_order[i]] = 0;
242  // step 3, compute x=Q*y
243  lapack_kernel::gemv('N', _mat.nrows(), _mat.ncols(), value_type(1),
244  _mat.data(), _mat.nrows(), _work.data(),
245  value_type(0), &_mrhs(0, i * Nrhs));
246  }
247  // copy back to the application
248  for (int j = 0; j < Nrhs; ++j)
249  for (size_type i(0); i < x.size(); ++i) x[i][j] = _mrhs(i, j);
250  }
