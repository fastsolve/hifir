76 /*!
77  * \struct hif_Options
78  * \brief POD parameter controls
79  * \note Values in parentheses are default settings
80  */
81 struct hif_Options {
82  double tau_L; /*!< inverse-based droptol for L (0.001) */
83  double tau_U; /*!< inverse-based droptol for U (0.001) */
84  double kappa_d; /*!< threshold for inverse-diagonal (3.) */
85  double kappa; /*!< inverse-norm threshold (3.) */
86  double alpha_L; /*!< growth factor of nnz per col (10) */
87  double alpha_U; /*!< growth factor of nnz per row (10) */
88  double rho; /*!< density threshold for dense LU (0.5) */
89  double c_d; /*!< size parameter for dense LU (10.0) */
90  double c_h; /*!< size parameter for H-version (2.0) */
91  int N; /*!< reference size of matrix (-1, system size) */
92  int verbose; /*!< message output level (1, i.e. info) */
93  int rf_par; /*!< parameter refinement (default 1) */
94  int reorder; /*!< reordering method (default is 2 (AMD)) */
95  int spd; /*!< SPD-ness: 0 (ID), >0 (PD), <0 (ND), (default 0) */
96  int check; /*!< check user input (default is true (!=0)) */
97  int pre_scale; /*!< prescale (default 0 (off)) */
98  int symm_pre_lvls; /*!< levels to be applied with symm preprocessing. If this
99  value is non-negative, then we will apply at most this
100  number of symmetric preprocessing. If this number if
101  negative, then within abs(symm_pre_lvls) levels, we will
102  automatically determine symmetric preprocessing via
103  pattern symmetry (default -2) */
104  int threads; /*!< user specified threads (default 0) */
105  int mumps_blr; /*!< MUMPS BLR options (default 2) *deprecated* */
106  int fat_schur_1st; /*!< double alpha for dropping L_E and U_F on 1st lvl */
107  double rrqr_cond; /*!< condition number threshold for RRQR (default 0) */
108  int pivot; /*!< pivoting flag (default is AUTO (2)) */
109  double gamma; /*!< threshold for thresholded pivoting (1.0) */
110  double beta; /*!< safeguard factor for equlibrition scaling (1e3) */
111  int is_symm; /*!< is symmetric (Hermitian) system? (default 0) */
112  int no_pre; /*!< no preprocessing (default 0) */
113  double nzp_thres; /*!< nonzero pattern symmetry threshold for symm
114  preprocessing (0.65) */
115  int dense_thres; /*!< threshold for dense for Schur complement (2000) */
116 };

317 /*!
318  * \brief represent an option control with C++ string
319  * \param[in] opt input option controls
320  * \return string representation of \a opt
321  */
322 inline std::string opt_repr(const Options &opt) {
323  using std::string;
324  using std::to_string; /* C++11 */
325  const static int leading_size = 30; /* should be enough */
326  const auto pack_int = [](const string &cat, const int v) -> string {
327  return cat + string(leading_size - cat.size(), ' ') + to_string(v) + "\n";
328  };
329  const auto pack_double = [](const string &cat, const double v) -> string {
330  return cat + string(leading_size - cat.size(), ' ') + to_string(v) + "\n";
331  };
332  const auto pack_name =
333  [&](const string & cat,
334  const std::function<string(const Options &)> &dec) -> string {
335  return cat + string(leading_size - cat.size(), ' ') + dec(opt) + "\n";
336  };
337  return pack_double("tau_L", opt.tau_L) + pack_double("tau_U", opt.tau_U) +
338  pack_double("kappa_d", opt.kappa_d) + pack_double("kappa", opt.kappa) +
339  pack_double("alpha_L", opt.alpha_L) +
340  pack_double("alpha_U", opt.alpha_U) + pack_double("rho", opt.rho) +
341  pack_double("c_d", opt.c_d) + pack_double("c_h", opt.c_h) +
342  pack_int("N", opt.N) + pack_name("verbose", get_verbose) +
343  pack_int("rf_par", opt.rf_par) +
344  pack_name("reorder", get_reorder_name) + pack_int("spd", opt.spd) +
345  pack_name(
346  "check",
347  [](const Options &opt_) { return opt_.check ? "yes" : "no"; }) +
348  pack_int("pre_scale", opt.pre_scale) +
349  pack_int("symm_pre_lvls", opt.symm_pre_lvls) +
350  pack_int("threads", opt.threads) +
351  pack_int("fat_schur_1st", opt.fat_schur_1st) +
352  pack_double("rrqr_cond", opt.rrqr_cond) +
353  pack_name("pivot",
354  [](const Options &opt_) {
355  return opt_.pivot == PIVOTING_OFF
356  ? "off"
357  : (opt_.pivot == PIVOTING_ON ? "on" : "auto");
358  }) +
359  pack_double("gamma", opt.gamma) + pack_double("beta", opt.beta) +
360  pack_int("is_symm", opt.is_symm) + pack_int("no_pre", opt.no_pre) +
361  pack_double("nzp_thres", opt.nzp_thres) +
362  pack_int("dense_thres", opt.dense_thres);
363 }
