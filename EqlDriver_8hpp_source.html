47 /// \class EqlDriver
48 /// \brief driver class wrapped around \ref eql::Equilibrator
49 /// \tparam ValueType value type
50 /// \tparam IndexType index type, e.g. \a int
51 /// \ingroup pre
52 template <class ValueType, class IndexType>
53 class EqlDriver {
54  public:
55  using value_type = ValueType; ///< value type
56  using index_type = IndexType; ///< index type
57  using ccs_type = CCS<value_type, index_type>; ///< ccs type
58  using crs_type = typename ccs_type::other_type; ///< crs type
59  using size_type = typename ccs_type::size_type; ///< size type
60  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
61  ///< scalar type
62  using kernel_type =
63  eql::Equilibrator<index_type, value_type, scalar_type, Array>;
64  ///< kernel type
65 
66  template <bool IsSymm>
67  inline static void do_matching(const int /* verbose */, crs_type &B,
68  Array<index_type> &p, Array<index_type> &q,
69  Array<scalar_type> &s, Array<scalar_type> &t,
70  const int pre_scale = 0) {
71  const size_type n = B.nrows(), nnz = B.nnz();
72  hif_error_if(B.nrows() != n, "must be squared systems");
73  hif_assert(p.size() >= n, "invalid P permutation size");
74  hif_assert(q.size() >= n, "invalid Q permutation size");
75  hif_assert(s.size() >= n, "invalid S row scaling size");
76  hif_assert(t.size() >= n, "invalid T column scaling size");
77 
78  do {
79  // check if pre-scaling is requested
80  switch (pre_scale) {
81  case 0:
82  scale_eye(B, s, t);
83  break;
84  case 1:
85  scale_extreme_values<IsSymm>(B, s, t);
86  break;
87  default:
88  iterative_scale<IsSymm>(B, s, t, 1e-10, 5);
89  break;
90  }
91  // deal with scaling and matching
92  kernel_type eq;
93  const int info =
94  eq.compute(n, nnz, B.row_start().data(), B.col_ind().data(),
95  B.vals().data(), p.data());
96 
97  if (info < 0) {
98  hif_error(
99  "MC64 matching returned negative %d flag!\n"
100  "Please refer MC64 documentation section 2.1.2 for error info "
101  "interpretation.",
102  info);
103  } else if (info == 1) {
104  hif_warning("MC64 the input matrix is structurally singular!");
105  } else if (info == 2) {
106  hif_warning("MC64 the result scaling may cause overflow issue!");
107  }
108 
109  // update scaling if necessary, NOTE we pass in CRS, thus it matching
110  // rescaling on the transpose of CCS
111  for (size_type i(0); i < n; ++i) {
112  s[i] *= eq.t(i);
113  t[i] *= eq.s(i);
114  }
115  } while (false); // bufs freed here
116 
117  // post processing
118  // NOTE mc64 NOT return inverse permutation
119  for (size_type i(0); i < n; ++i) p[i] = std::abs(p[i]) - 1;
120 
121  // handle symmetric
122  if (IsSymm) {
123  std::copy_n(p.cbegin(), n, q.begin());
124  for (size_type i(0); i < n; ++i) s[i] = std::sqrt(s[i] * t[i]);
125  std::copy_n(s.cbegin(), n, t.begin());
126  } else
127  for (size_type i(0); i < n; ++i) q[i] = i;
128  }
129 };
