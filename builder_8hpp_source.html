79 /// \class HIF
80 /// \tparam ValueType numerical value type, e.g. \a double
81 /// \tparam IndexType index type, e.g. \a int
82 /// \tparam UserDenseFactor Potential user customized dense factor
83 ///
84 /// This is top user interface (C++); it is designed as a preconditioner that
85 /// can be easily plugin other codes. There are two core member functions, 1)
86 /// \ref factorize a multilevel ILU preconditioner and 2) \ref solve the
87 /// preconditioner system. For computing preconditioner, the input can be either
88 /// \ref CCS or \ref CRS, for solving, the input must be \ref Array. However,
89 /// be aware that both CCS/CRS and Array can be used as external data wrappers,
90 /// thus one should not worry about input data duplications.
91 ///
92 /// The following is a simple workflow for Builder.
93 ///
94 /// \code{.cpp}
95 /// #include <hifir.hpp>
96 /// using namespace hif;
97 /// using builder_t = HIF<double, int>;
98 /// using crs_t = builder_t::crs_type;
99 /// int main() {
100 /// const auto A = wrap_const_crs(...);
101 /// builder_t builder;
102 /// builder.factorize(A);
103 /// builder.solve(...);
104 /// }
105 /// \endcode
106 template <class ValueType, class IndexType,
107  template <class> class UserDenseFactor = DefaultDenseSolver>
108 class HIF {
109  public:
110  typedef ValueType value_type; ///< value type
111  typedef Array<value_type> array_type; ///< array type
112  typedef IndexType index_type; ///< index type
113  typedef Array<index_type> iarray_type; ///< index array
114  typedef CRS<value_type, index_type> crs_type; ///< crs type
115  typedef typename crs_type::other_type ccs_type; ///< ccs type
116  // constexpr static SmallScaleType sss_type = SSSType; ///< small scale type
117  typedef Precs<value_type, index_type, UserDenseFactor> precs_type;
118  ///< multilevel preconditioner type
119  typedef typename precs_type::value_type prec_type; ///< single level prec
120  typedef typename prec_type::size_type size_type; ///< size type
121 #ifdef HIF_HIGH_PRECISION_SOLVE
122  typedef typename ValueTypeMixedTrait<value_type>::boost_type boost_value_type;
123 #else
124  typedef value_type boost_value_type;
125 #endif
126  ///< high-precision value type
127  typedef Array<boost_value_type> work_array_type; ///< work array type

132  /// \brief check number of levels
133  /// \note This function takes \f$\mathcal{O}(1)\f$ since C++11
134  inline size_type levels() const {
135  const size_type lvls = _precs.size();
136  if (lvls)
137  return lvls + !_precs.back().dense_solver.empty() +
138  !_precs.back().symm_dense_solver.empty();

150  /// \brief compute the overall nnz of the multilevel preconditioners
151  inline size_type nnz() const {
152  if (empty()) return 0u;
153  size_type n(0);
154  for (auto itr = _precs.cbegin(); itr != _precs.cend(); ++itr)

159  /// \brief get the numerical rank
160  inline size_type rank() const {
161  if (empty()) return 0u;
162  const auto &prec_last = precs().back();
163  return std::min(nrows(), ncols()) -
164  ((prec_last.n - prec_last.m) - prec_last.last_rank());
165  }

197  /// \brief get the stats
198  /// \param[in] entry statistic entry
199  ///
200  /// Currently, the following stats are available:
201  /// - 0: total deferrals
202  /// - 1: dynamic deferrals
203  /// - 2: dynamic deferrals due to bad diagonals
204  /// - 3: dynamic deferrals due to bad conditioning
205  /// - 4: total droppings
206  /// - 5: droppings due to space-controling
207  /// Notice that entry \a 0 subtract entry \a 1 is the static deferrals, which
208  /// are due to zero diagonals that occurs when applying symmetric
209  /// preprocessing.
210  inline size_type stats(const size_type entry) const {
211  static const char *help =
212  "\t0: total deferrals\n"
213  "\t1: dynamic deferrals\n"
214  "\t2: dynamic deferrals due to bad diagonals\n"
215  "\t3: dynamic deferrals due to bad conditioning\n"
216  "\t4: total droppings\n"
217  "\t5: droppings due to space-controling\n";
218  hif_error_if(empty(), "no stats available for empty structure");
219  if (entry > 5u) {
220  // NOTE, we cannot pass the help information as variadic arguments due
221  // to the internal buffer may overflow thus causing segfault.
222  std::stringstream ss;
223  ss << entry << " exceeds maximum statistics entry (5)\nhelp:\n" << help;
224  hif_error(ss.str().c_str());
225  }
226  return _stats[entry];
227  }

249  /// \brief factorize the HIF preconditioner
250  /// \tparam CsType compressed storage input, either \ref CRS or \ref CCS
251  /// \param[in] A input matrix
252  /// \param[in] params control parameters, using the default values
253  /// \param[in] m0 leading block size, if it's zero (default), then the routine
254  /// will assume an asymmetric leading block.
255  /// \sa solve
256  template <class CsType>
257  inline void factorize(const CsType &A, const Params &params = DEFAULT_PARAMS,
258  const size_type m0 = 0u) {
259  const static internal::StdoutStruct Crout_cout;
260  const static internal::DummyStreamer Crout_cout_dummy;
261  using cs_type =
262  typename std::conditional<CsType::ROW_MAJOR, crs_type, ccs_type>::type;
263 
264  static_assert(std::is_same<index_type, typename CsType::index_type>::value,
265  "inconsistent index types");
266 
267  // print introduction
268  if (hif_verbose(INFO, params)) {
269  if (!internal::introduced) {
270  hif_info(internal::intro, HIF_GLOBAL_VERSION, HIF_MAJOR_VERSION,
271  HIF_MINOR_VERSION);
272  internal::introduced = true;
273  }
274  }
275  const bool revert_warn = warn_flag();
276  if (params.verbose == VERBOSE_NONE)
277  (void)warn_flag(0);
278  else
279  warn_flag(1);
280 
281  _nrows = A.nrows();
282  _ncols = A.ncols();
283 
284  DefaultTimer t; // record overall time
285  t.start();
286  if (!empty()) {
287  if (hif_verbose(INFO, params))
288  hif_info("multilevel precs are not empty, wipe previous results first");
289  _precs.clear();
290  // also clear the previous buffer
291  _prec_work.resize(0);
292  }
293  const int schur_threads = mt::get_nthreads(params.threads);
294  // initialize statistics
295  for (size_type i(0); i < sizeof(_stats) / sizeof(size_type); ++i)
296  _stats[i] = 0;
297 
298  cs_type AA;
299  AA.resize(A.nrows(), A.ncols());
300  // check index base
301  if (A.nrows()) {
302  if (A.ind_start()[0] == index_type(0)) {
303  // 0-based, shallow copy
304  AA.ind_start() = A.ind_start();
305  AA.inds() = A.inds();
306  } else if (A.ind_start()[0] == index_type(1)) {
307  // 1-based
308  if (hif_verbose(INFO, params))
309  hif_info("converting to 0-based compressed matrix...");
310  AA.ind_start().resize(A.ind_start().size());
311  for (size_type i(0); i < A.ind_start().size(); ++i)
312  AA.ind_start()[i] = A.ind_start()[i] - 1;
313  AA.inds().resize(A.inds().size());
314  for (size_type i(0); i < A.inds().size(); ++i)
315  AA.inds()[i] = A.inds()[i] - 1;
316  } else
317  hif_error("only {0,1}-based compressed sparse matrices are supported!");
318  }
319  // check value type consistency
320  if (std::is_same<value_type, typename CsType::value_type>::value)
321  AA.vals() = array_type(A.nnz(), (value_type *)A.vals().data(), true);
322  else {
323  if (hif_verbose(INFO, params))
324  hif_info("converting value type precision...");
325  // shallow integer arrays
326  AA.resize(A.nrows(), A.ncols());
327  AA.vals().resize(A.nnz());
328  hif_error_if(AA.vals().status() == DATA_UNDEF,
329  "memory allocation failed");
330  std::copy(A.vals().cbegin(), A.vals().cend(), AA.vals().begin());
331  }
332  // check validity of the input system
333  if (params.check) {
334  if (hif_verbose(INFO, params))
335  hif_info("perform input matrix validity checking");
336  AA.check_validity();
337  }
338  // create size references for dropping
339  iarray_type row_sizes, col_sizes;
340  if (hif_verbose(FAC, params))
341  _factorize_kernel(AA, m0, params, row_sizes, col_sizes, Crout_cout,
342  schur_threads);
343  else
344  _factorize_kernel(AA, m0, params, row_sizes, col_sizes, Crout_cout_dummy,
345  schur_threads);
346  t.finish();
347  if (hif_verbose(INFO, params)) {
348  const size_type Nnz = nnz();
349  hif_info("\ninput nnz(A)=%zd, nnz(precs)=%zd, ratio=%g", A.nnz(), Nnz,
350  (double)Nnz / A.nnz());
351  hif_info("\nmultilevel precs building time (overall) is %gs", t.time());
352  }
353  if (revert_warn) (void)warn_flag(1);
354  }

356  /// \brief factorize the HIF preconditioner with generic interface
357  /// \tparam IsCrs if \a true, then the input compressed structure will be
358  /// assumed to be \ref CRS format
359  /// \tparam IndexType_ integer data type, e.g., \a int
360  /// \tparam ValueType_ value data type, e.g., \a double
361  /// \param[in] n size of system
362  /// \param[in] indptr index starting position array, must be length of \a n+1
363  /// \param[in] indices index value array, must be length of \a indptr[n]
364  /// \param[in] vals value array, same length as that of \a indices
365  /// \param[in] params control parameters, using the default values.
366  /// \param[in] m0 leading block size, if it's zero (default), then the routine
367  /// will assume an asymmetric leading block.
368  ///
369  /// This interface differs from the above one is that it takes plain-old-data
370  /// types as input thus flexible. Notice that the integer and floating data
371  /// types don't need to align with \ref index_type and \ref value_type, which
372  /// aims for mixed-precision computation.
373  template <bool IsCrs, class IndexType_, class ValueType_>
374  inline void factorize(const size_type n, const IndexType_ *indptr,
375  const IndexType_ *indices, const ValueType_ *vals,
376  const Params & params = DEFAULT_PARAMS,
377  const size_type m0 = 0u) {
378  using foreign_crs_type = CRS<ValueType_, IndexType_>;
379  using foreign_ccs_type = typename foreign_crs_type::other_type;
380  using foreign_cs_type = typename std::conditional<IsCrs, foreign_crs_type,
381  foreign_ccs_type>::type;
382  const foreign_cs_type A(n, (IndexType_ *)indptr, (IndexType_ *)indices,
383  (ValueType_ *)vals, true);
384  factorize(A, params, m0);
385  }

387  /// \brief solve \f$\mathbf{x}=\mathbf{M}^{-1}\mathbf{b}\f$
388  /// \tparam RhsType right-hand side type
389  /// \tparam SolType solution type
390  /// \param[in] b right-hand side vector
391  /// \param[out] x solution vector
392  /// \param[in] trans (optional) transpose/Hermitian flag (default is false)
393  /// \param[in] r (optional) dimension for back solve for last level
394  /// efault is its numerical rank
395  template <class RhsType, class SolType>
396  inline void solve(const RhsType &b, SolType &x, const bool trans = false,
397  const size_type r = 0u) const {
398  hif_error_if(empty(), "MILU-Prec is empty!");
399  hif_error_if(b.size() != x.size(), "unmatched sizes");
400  if (_prec_work.empty())
401  _prec_work.resize(
402  compute_prec_work_space(_precs.cbegin(), _precs.cend()));
403  !trans ? prec_solve(_precs.cbegin(), b, r, x, _prec_work)
404  : prec_solve_tran(_precs.cbegin(), b, r, x, _prec_work);
405  if (!trans && nsp)
406  nsp->filter(x.data(), x.size()); // filter null space
407  else if (trans && nsp_tran)
408  nsp_tran->filter(x.data(), x.size()); // filter (left) null space
409  }
410 
411  /// \brief solve \f$\mathbf{X}=\mathbf{M}^{-1}\mathbf{B}\f$ (multiple RHS)
412  /// \tparam RhsValueType right-hand-side (RHS) value type
413  /// \tparam SolValueType solution value type
414  /// \tparam Nrhs number of RHS
415  /// \param[in] b right-hand side vector
416  /// \param[out] x solution vector
417  /// \param[in] r (optional) dimension for back solve for last level
418  /// default is its numerical rank
419  template <class RhsValueType, class SolValueType, size_type Nrhs>
420  inline void solve_mrhs(const Array<std::array<RhsValueType, Nrhs>> &b,
421  Array<std::array<SolValueType, Nrhs>> & x,
422  const size_type r = 0u) const {
423  hif_error_if(empty(), "MILU-Prec is empty!");
424  hif_error_if(b.size() != x.size(), "unmatched sizes");
425  const auto nw = compute_prec_work_space(_precs.cbegin(), _precs.cend());
426  if (_prec_work.empty()) _prec_work.resize(nw * Nrhs);
427  Array<std::array<boost_value_type, Nrhs>> w(
428  nw, (std::array<boost_value_type, Nrhs> *)_prec_work.data(), true);
429  prec_solve_mrhs(_precs.cbegin(), b, r, x, w);
430  hif_warning_if(nsp, "multiple RHS does not support null space filter.");
431  }

433  /// \brief solve with iterative refinement
434  /// \tparam Matrix matrix type
435  /// \tparam RhsType right-hand side type
436  /// \tparam SolType solution type
437  /// \param[in] A matrix operator
438  /// \param[in] b right-hand side vector
439  /// \param[in] N iteration count
440  /// \param[out] x solution vector
441  /// \param[in] trans (optional) transpose/Hermitian flag, default is false
442  /// \param[in] r (optional) dimension for back solve for last level
443  /// default is its numerical rank
444  template <class Matrix, class RhsType, class SolType>
445  inline void hifir(const Matrix &A, const RhsType &b, const size_type N,
446  SolType &x, const bool trans = false,
447  const size_type r = static_cast<size_type>(-1)) const {

449  // user callback
450  _ir.iter_refine(*this, A, b, N, x, r, trans);

453  /// \brief solve with iterative refinement and residual bounds
454  /// \tparam Matrix matrix type
455  /// \tparam RhsType right-hand side type
456  /// \tparam SolType solution type
457  /// \param[in] A matrix operator
458  /// \param[in] b right-hand side vector
459  /// \param[in] N iteration count
460  /// \param[in] betas lower and upper bound of norms of residual
461  /// \param[out] x solution vector
462  /// \param[in] trans (optional) transpose/Hermitian flag, default is false
463  /// \param[in] r (optional) dimension for back solve for last level
464  /// default is its numerical rank

470  SolType &x, const bool trans = false,
471  const size_type r = static_cast<size_type>(-1)) const {
472  // NOTE, do not assume A shares interface of CRS, as it can be

474  return _ir.iter_refine(*this, A, b, N, betas, x, r, trans);
475  }

480  /// \brief compute \f$\mathbf{y}=\mathbf{Mx}\f$
481  /// \tparam RhsType right-hand side type
482  /// \tparam SolType solution type
483  /// \param[in] x right-hand side vector
484  /// \param[out] y solution vector
485  /// \param[in] trans (optional) transpose/Hermitian flag (default is false)
486  /// \param[in] r (optional) dimension for back solve for last level
487  /// default is its numerical rank
488  template <class RhsType, class SolType>
489  inline void mmultiply(const RhsType &x, SolType &y, const bool trans = false,

491  hif_error_if(empty(), "MILU-Prec is empty!");

494  _prec_work.resize(
495  compute_prec_work_space(_precs.cbegin(), _precs.cend()));

500  protected:
501  template <class CsType, class CroutStreamer>
502  inline void _factorize_kernel(const CsType &A, const size_type m0,
503  const Params &opts, iarray_type &row_sizes,
504  iarray_type & col_sizes,
505  const CroutStreamer &Crout_info,
506  const int schur_threads = 1,
507  const bool must_pivot = false) {
508  hif_error_if(A.nrows() != A.ncols(),
509  "Currently only squared systems are supported");
510  size_type m(m0); // init m
511  size_type N; // reference size
512  const size_type cur_level = levels() + 1;
513 
514  // determine the reference size
515  if (opts.N >= 0)
516  N = opts.N;
517  else
518  N = cur_level == 1u ? A.nrows() : _precs.front().n;
519 
520  // check partial symmetry
521  const bool sym = cur_level == 1u && m > 0u;
522  if (!sym) m = A.nrows(); // IMPORTANT! If asymmetric, set m = n
523 
524  CsType S;
525  int pivot_flag = 0;
526  if (opts.is_symm) {
527  S = symm_level_factorize(A, m, N, opts, Crout_info, _precs, row_sizes,
528  _stats, schur_threads);
529  } else {
530  if (opts.pivot == PIVOTING_OFF)
531  // instantiate IsSymm here
532  S = sym ? level_factorize<true>(A, m, N, opts, Crout_info, _precs,
533  row_sizes, col_sizes, _stats,
534  schur_threads)
535  : level_factorize<false>(A, m, N, opts, Crout_info, _precs,
536  row_sizes, col_sizes, _stats,
537  schur_threads);
538  else if (opts.pivot == PIVOTING_ON)
539  S = pivot_level_factorize(A, m, N, opts, Crout_info, _precs, row_sizes,
540  col_sizes, _stats, schur_threads);
541  else {
542  if (must_pivot) {
543  S = pivot_level_factorize(A, m, N, opts, Crout_info, _precs,
544  row_sizes, col_sizes, _stats,
545  schur_threads);
546  pivot_flag = 1; // Always recursively do pivoting
547  } else
548  S = sym ? level_factorize<true>(A, m, N, opts, Crout_info, _precs,
549  row_sizes, col_sizes, _stats,
550  schur_threads, &pivot_flag)
551  : level_factorize<false>(A, m, N, opts, Crout_info, _precs,
552  row_sizes, col_sizes, _stats,
553  schur_threads, &pivot_flag);
554  }
555  }
556
