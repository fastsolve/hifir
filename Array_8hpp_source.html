65 /// \class Array
66 /// \brief Core data structure used in compressed storage
67 /// \tparam T value type
68 ///
69 /// Compared to std::vector, \a Array allows one to wrap external data, which
70 /// is a necessary feature for this package. In addition, we allow shallow
71 /// copy instead of deepcopy used in std::vector.
72 template <class T>
73 class Array {
74  public:

76  typedef T* pointer; ///< corresponding pointer type

148  /// \brief constructor for copying or wrapping external data
149  /// \param[in] n size of array
150  /// \param[in] data external data
151  /// \param[in] wrap flag to indicate wrapping or copying (optional)
152  ///
153  /// If \a wrap is \a true, then a wrapper Array will be constructed, as a
154  /// result, you cannot \ref resize and \ref reserve data.
155  Array(const size_type n, pointer data, bool wrap = false) {
156  _counts = new _RefCount();
157  if (!wrap) {
158  _data = new (std::nothrow) value_type[n];
159  // handle overflow
160  if (!_data) {
161  _size = _cap = 0u;
162  _status = DATA_UNDEF;
163  } else {
164  std::copy_n(data, n, _data);
165  _status = DATA_OWN;
166  }
167  } else {
168  _data = data;
169  _status = DATA_WRAP;
170  }
171  _size = _cap = n;
172  }

174  /// \brief constructor for copying a foreign array, useful in type conversion
175  /// \tparam V foreign value type
176  /// \param other another array
177  template <class V>
178  explicit Array(const Array<V>& other)
179  : _data(new (std::nothrow) value_type[other.size()]),
180  _size(other.size()),
181  _cap(_size),
182  _status(DATA_OWN),
183  _counts(new _RefCount()) {
184  hif_assert(_data, "memory allocation failed");
185  if (!_data) {
186  _size = _cap = 0u;
187  _status = DATA_UNDEF;
188  } else
189  std::copy(other.cbegin(), other.cend(), _data);
190  }

192  /// \brief destructor, handle shallow copy and external data
193  ~Array() {
194  // first trigger reference counter to decrement
195  // Added checking nullptr so that move can work
196  if (_counts) _counts->dec();
197  // handle memory deallocation
198  if (_data && _status == DATA_OWN && _counts && *_counts == 0u)
199  if (!_counts->_invalid) delete[] _data;
200  if (_counts && *_counts == 0u) {
201  // don't forget to free the counter
202  delete _counts;
203  _counts = nullptr;
204  }
205  _data = nullptr;
206  _size = _cap = 0;
207  _status = DATA_UNDEF;
208  }

210  /// \brief constructor for shallow or deep copy
211  /// \note the default copy constructor is shallow
212  /// \param[in] other another Array
213  /// \param[in] clone if \a true, then do deep copy, default is \a false
214  Array(const Array& other, bool clone = false) {
215  if (!clone) {
216  // shallow
217  _counts = other._counts; // reference the counter
218  _data = other._data;
219  _size = other._size;
220  _cap = other._cap;
221  _status = other._status;
222  // increment the counter
223  _counts->inc();
224  } else {
225  // deep copy
226  _counts = new _RefCount();
227  _data = new value_type[other._size];
228  _size = other._size;
229  _cap = _size;
230  _status = DATA_OWN;
231  std::copy_n(other._data, _size, _data);
232  }
233  }

346  /// \brief resize an Array with new size
347  /// \param[in] n new size
348  /// \param[in] presv if \a true (default), then values will be preserved
349  /// \warning This function should not be called for external data
350  /// \sa reserve
351  inline void resize(const size_type n, bool presv = true) {
352  hif_assert(_status != DATA_WRAP, "cannot resize external data");
353  // quick return
354  if (n <= _cap) {
355  _size = n;
356  return;
357  }
358  const size_type cap = _size ? 1.2 * n : n; // 20% more
359  pointer data = new (std::nothrow) value_type[cap];
360  hif_assert(data, "memory allocation failed");
361  if (!data) {
362  // call destructor
363  this->~Array();
364  _counts = new _RefCount();
365  return;
366  }
367  if (presv) std::copy_n(_data, _size, data);
368  // now call destructor
369  this->~Array();
370  // since we own a new database, reset the counter
371  _counts = new _RefCount();
372  _data = data;
373  _size = n;
374  _cap = cap;
375  _status = DATA_OWN;
376  }

378  /// \brief reserve space for Array
379  /// \param[in] n capacity request
380  /// \warning This function should not be called for external data
381  /// \sa resize
382  inline void reserve(const size_type n) {
383  hif_assert(_status != DATA_WRAP, "cannot call reserve for external data");
384  if (_cap >= n) return;
385  pointer data = new (std::nothrow) value_type[n];
386  hif_assert(data, "memory allocation failed");
387  if (!data) {
388  // call destructor
389  this->~Array();
390  _counts = new _RefCount();
391  return;
392  }
393  std::copy_n(_data, _size, data);
394  const size_type size_bak = _size;
395  // now call destructor
396  this->~Array();
397  // since we own a new database, reset the counter
398  _counts = new _RefCount();
399  _data = data;
400  _cap = n;
401  _size = size_bak;
402  _status = DATA_OWN;
403  }

468 /// \brief try steal the array data ownership
469 /// \tparam T value type
470 /// \param[in,out] arr input and output array
471 /// \param[out] data pointer to the internal database
472 /// \param[out] size array size
473 /// \param[out] cap array capacity
474 /// \note If \a arr can't be stole, \a nullptr is assigned to \a data
475 /// \warning Only works for DATA_OWN status
476 /// \warning Manually calling \a delete[] may be needed
477 /// \warning Not designed using in MT
478 template <class T>
479 inline void steal_array_ownership(Array<T>& arr,
480  typename Array<T>::pointer& data,
481  typename Array<T>::size_type& size,
482  typename Array<T>::size_type& cap) {
483  typedef typename Array<T>::size_type size_type;
484  if (arr.status() != DATA_OWN) {
485  data = nullptr;
486  size = cap = 0u;
487  return;
488  }
489  data = arr._data;
490  size = arr._size;
491  cap = arr._cap;
492  arr._data = nullptr;
493  arr._size = size_type(0);
494  arr._cap = size_type(0);
495  arr._status = DATA_UNDEF;
496  // set the counter to invalid
497  arr._counts->_invalid = true;
498 }
