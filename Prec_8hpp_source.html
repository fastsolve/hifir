58 /// \brief A single level preconditioner
59 /// \tparam ValueType value data type, e.g. \a double
60 /// \tparam IndexType index data type, e.g. \a int

67 /// The following demonstrates how to customize a user-defined dense factory

169  /// \brief get number of nonzeros
170  inline size_type nnz() const {
171  size_type nz = m ? L_B.nnz() + U_B.nnz() + m : 0;
172  if (n - m) nz += E.nnz() + F.nnz();

190  /// \brief check if this a last level preconditioner
191  ///
192  /// The idea is first check if the \ref dense_solver is empty or not; be aware
193  /// that there is another situation that we should treat the preconditioner
194  /// is last level---if \ref m is equal to \ref n.
195  ///
196  /// \note Currently, we test m == n, which is fine for squared systems.
197  inline bool is_last_level() const {
198  return !dense_solver.empty() || !symm_dense_solver.empty() || m == n;
199  }

201  /// \brief check the (numerical) rank for last level (if applicable)
202  inline size_type last_rank() const {
203  if (!is_last_level() || m == n) return 0u;
204  if (!dense_solver.empty()) return dense_solver.rank();
205  return symm_dense_solver.rank();

220  /// \brief export all numerical data
221  template <class ExportType>
222  inline void export_sparse_data(
223  typename ExportType::ippointer L_indptr,
224  typename ExportType::ipointer L_indices,
225  typename ExportType::pointer L_vals, typename ExportType::pointer D_vals,
226  typename ExportType::ippointer U_indptr,
227  typename ExportType::ipointer U_indices,
228  typename ExportType::pointer U_vals,
229  typename ExportType::ippointer E_indptr,
230  typename ExportType::ipointer E_indices,
231  typename ExportType::pointer E_vals,
232  typename ExportType::ippointer F_indptr,
233  typename ExportType::ipointer F_indices,
234  typename ExportType::pointer F_vals,
235  typename ValueTypeTrait<typename ExportType::value_type>::value_type
236  *s_vals,
237  typename ValueTypeTrait<typename ExportType::value_type>::value_type
238  * t_vals,
239  typename ExportType::ipointer p_vals,
240  typename ExportType::ipointer pinv_vals,
241  typename ExportType::ipointer q_vals,
242  typename ExportType::ipointer qinv_vals, const bool jit_destroy = false) {
243  // leading block
244  export_compressed_data<ExportType>(L_B, L_indptr, L_indices, L_vals); // L
245  std::copy_n(d_B.cbegin(), m, D_vals); // D
246  if (jit_destroy) array_type().swap(d_B);
247  export_compressed_data<ExportType>(U_B, U_indptr, U_indices, U_vals); // U
248  if (jit_destroy) U_B.destroy();
249  // off-diagonal blocks
250  export_compressed_data<ExportType>(E, E_indptr, E_indices, E_vals); // E
251  if (jit_destroy) E.destroy();
252  export_compressed_data<ExportType>(F, F_indptr, F_indices, F_vals); // F
253  if (jit_destroy) F.destroy();
254  // row/column scaling
255  std::copy_n(s.cbegin(), n, s_vals); // s
256  std::copy_n(t.cbegin(), n, t_vals); // t
257  // row/column permutations
258  std::copy_n(p.cbegin(), n, p_vals); // P
259  std::copy_n(p_inv.cbegin(), n, pinv_vals); // P^T
260  std::copy_n(q.cbegin(), n, q_vals); // Q
261  std::copy_n(q_inv.cbegin(), n, qinv_vals); // Q^T
262  if (jit_destroy) {
263  // destroy all 1D arrays
264  array_type().swap(d_B);
265  sarray_type().swap(s);
266  sarray_type().swap(t);
267  perm_type().swap(p);
268  perm_type().swap(p_inv);
269  perm_type().swap(q);
270  perm_type().swap(q_inv);
271  m = n = 0;
272  }
273  }

275  template <class T>
276  inline void inquire_or_export_dense(T *mat, size_type &nrows,
277  size_type &ncols,
278  const bool destroy = false) {
279  if (!dense_solver.empty()) {
280  if (!mat) {
281  nrows = dense_solver.mat_backup().nrows();
282  ncols = dense_solver.mat_backup().ncols();
283  return;
284  }
285  // assume mat is properly allocated
286  std::copy(dense_solver.mat_backup().array().cbegin(),
287  dense_solver.mat_backup().array().cend(), mat);
288  if (destroy) {
289  array_type().swap(dense_solver.mat_backup().array());
290  array_type().swap(dense_solver.mat().array());
291  }
292  return;
293  }
294  hif_assert(!symm_dense_solver.empty(), "fatal");
295  if (!mat) {
296  nrows = symm_dense_solver.mat_backup().nrows();
297  ncols = symm_dense_solver.mat_backup().ncols();
298  return;
299  }
300  // assume mat is properly allocated
301  std::copy(symm_dense_solver.mat_backup().array().cbegin(),
302  symm_dense_solver.mat_backup().array().cend(), mat);
303  if (destroy) {
304  array_type().swap(symm_dense_solver.mat_backup().array());
305  array_type().swap(symm_dense_solver.mat().array());
306  }
307  }

336 /// \typedef Precs
337 /// \brief multilevel preconditioners
338 /// \tparam ValueType value data type, e.g. \a double
339 /// \tparam IndexType index data type, e.g. \a int
340 /// \tparam IndPtrType index pointer type, default is \a std::ptrdiff_t
341 /// \tparam UserDenseFactor Potential user customized dense factor
342 /// \ingroup slv
343 ///
344 /// We choose to use STL list because adding new node is constant time without
345 /// bothering copying. It's worth noting that querying the size of \a list has
346 /// become constant since C++11. Also, with the constructors in \ref Prec, the
347 /// following way can be used to construct list:
348 ///
349 /// \code{.cpp}
350 /// using precs_t = Precs<double, int>;
351 /// using prec_t = precs_t::value_type;
352 /// precs_t precs;
353 /// precs.emplace_back(m, n, /* rvalue references */);
354 /// \endcode
355 template <class ValueType, class IndexType, class IndPtrType = std::ptrdiff_t,
356  template <class> class UserDenseFactor = DefaultDenseSolver>
357 using Precs =
358  std::list<Prec<ValueType, IndexType, IndPtrType, UserDenseFactor>>;
