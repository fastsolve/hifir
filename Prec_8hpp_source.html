58 /// \class Prec
59 /// \brief A single level preconditioner
60 /// \tparam ValueType value data type, e.g. \a double
61 /// \tparam IndexType index data type, e.g. \a int
62 /// \tparam UserDenseFactor Potential user customized dense factory for the
63 /// final Schur complement, if it is
64 /// \a DefaultDenseSolver (default), the code then uses
65 /// the built-in dense factors.
66 ///
67 /// The following demonstrates how to customize a user-defined dense factory
68 ///
69 /// \code{.cpp}
70 /// #include <hifir.hpp>
71 /// template <class T>
72 /// using MyQRCP = hif::QRCP<T>; // simply wrap built-in QRCP solver

74 /// struct MyDenseSolver : public hif::DefaultDenseSolver<T> {

172  /// \brief get number of nonzeros
173  inline size_type nnz() const {
174  size_type nz = m ? L_B.nnz() + U_B.nnz() + m : 0;
175  if (n - m) nz += E.nnz() + F.nnz();
176  if (!dense_solver.empty() || !symm_dense_solver.empty())
177  return nz + (n - m) * (n - m);
178  return nz;
179  }
180 
181  /// \brief get the number of nonzeros in \a E and \a F components
182  inline size_type nnz_ef() const {
183  if (n - m) return E.nnz() + F.nnz();
184  return 0;
185  }

247  /// \brief check if this a last level preconditioner
248  ///
249  /// The idea is first check if the \ref dense_solver is empty or not; be aware
250  /// that there is another situation that we should treat the preconditioner
251  /// is last level---if \ref m is equal to \ref n.
252  ///
253  /// \note Currently, we test m == n, which is fine for squared systems.
254  inline bool is_last_level() const {
255  return !dense_solver.empty() || !symm_dense_solver.empty() || m == n;
256  }
257 
258  /// \brief check the (numerical) rank for last level (if applicable)
259  inline size_type last_rank() const {
260  if (!is_last_level() || m == n) return 0u;
261  if (!dense_solver.empty()) return dense_solver.rank();
262  return symm_dense_solver.rank();
263  }

278  template <class ExportType, typename T = void>
279  inline typename std::enable_if<!INTERVAL_BASED, T>::type export_sparse_data(
280  typename ExportType::ipointer L_indptr,
281  typename ExportType::ipointer L_indices,
282  typename ExportType::pointer L_vals, typename ExportType::pointer D_vals,
283  typename ExportType::ipointer U_indptr,
284  typename ExportType::ipointer U_indices,
285  typename ExportType::pointer U_vals,
286  typename ExportType::ipointer E_indptr,
287  typename ExportType::ipointer E_indices,
288  typename ExportType::pointer E_vals,
289  typename ExportType::ipointer F_indptr,
290  typename ExportType::ipointer F_indices,
291  typename ExportType::pointer F_vals,
292  typename ValueTypeTrait<typename ExportType::value_type>::value_type
293  *s_vals,
294  typename ValueTypeTrait<typename ExportType::value_type>::value_type
295  * t_vals,
296  typename ExportType::ipointer p_vals,
297  typename ExportType::ipointer pinv_vals,
298  typename ExportType::ipointer q_vals,
299  typename ExportType::ipointer qinv_vals, const bool jit_destroy = false) {
300  // leading block
301  export_compressed_data<ExportType>(L_B, L_indptr, L_indices, L_vals); // L
302  std::copy_n(d_B.cbegin(), m, D_vals); // D
303  if (jit_destroy) array_type().swap(d_B);
304  export_compressed_data<ExportType>(U_B, U_indptr, U_indices, U_vals); // U
305  if (jit_destroy) U_B.destroy();
306  // off-diagonal blocks
307  export_compressed_data<ExportType>(E, E_indptr, E_indices, E_vals); // E
308  if (jit_destroy) E.destroy();
309  export_compressed_data<ExportType>(F, F_indptr, F_indices, F_vals); // F
310  if (jit_destroy) F.destroy();
311  // row/column scaling
312  std::copy_n(s.cbegin(), n, s_vals); // s
313  std::copy_n(t.cbegin(), n, t_vals); // t
314  // row/column permutations
315  std::copy_n(p.cbegin(), n, p_vals); // P
316  std::copy_n(p_inv.cbegin(), n, pinv_vals); // P^T
317  std::copy_n(q.cbegin(), n, q_vals); // Q
318  std::copy_n(q_inv.cbegin(), n, qinv_vals); // Q^T
319  if (jit_destroy) {
320  // destroy all 1D arrays
321  array_type().swap(d_B);
322  sarray_type().swap(s);
323  sarray_type().swap(t);
324  perm_type().swap(p);
325  perm_type().swap(p_inv);
326  perm_type().swap(q);
327  perm_type().swap(q_inv);
328  m = n = 0;
329  }
330  }

349  template <class T>
350  inline void inquire_or_export_dense(T *mat, size_type &nrows,
351  size_type &ncols,
352  const bool destroy = false) {
353  if (!dense_solver.empty()) {
354  if (!mat) {
355  nrows = dense_solver.mat_backup().nrows();
356  ncols = dense_solver.mat_backup().ncols();
357  return;
358  }
359  // assume mat is properly allocated
360  std::copy(dense_solver.mat_backup().array().cbegin(),
361  dense_solver.mat_backup().array().cend(), mat);
362  if (destroy) {
363  array_type().swap(dense_solver.mat_backup().array());
364  array_type().swap(dense_solver.mat().array());
365  }
366  return;
367  }
368  hif_assert(!symm_dense_solver.empty(), "fatal");
369  if (!mat) {
370  nrows = symm_dense_solver.mat_backup().nrows();
371  ncols = symm_dense_solver.mat_backup().ncols();
372  return;
373  }
374  // assume mat is properly allocated
375  std::copy(symm_dense_solver.mat_backup().array().cbegin(),
376  symm_dense_solver.mat_backup().array().cend(), mat);
377  if (destroy) {
378  array_type().swap(symm_dense_solver.mat_backup().array());
379  array_type().swap(symm_dense_solver.mat().array());
380  }
381  }
