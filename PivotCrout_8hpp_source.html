43 /// \class PivotCrout
44 /// \brief Crout update in deferred fashion with pivoting
45 /// \ingroup crout
46 ///
47 /// The idea of this class is to implement Crout kernels that wrap around a
48 /// common factor---step. In this case, the class Crout is as simple (in terms
49 /// of data size) as an integer. It also supports implicitly casting to integer
50 /// as well as increment operations (both suffix and prefix versions). The whole
51 /// disign can be demonstrate by the following C++ pseudocode:
52 ///
53 /// \code{.cpp}
54 /// for (Crout crout; crout < m; ++crout) {
55 /// crout.do_work1(...);
56 /// crout.do_work2(...);
57 /// ...
58 /// }
59 /// \endcode
60 class PivotCrout : public Crout {
61  public:
62  using size_type = Crout::size_type; ///< size type

64  /// \brief compute diagonal entry
65  /// \tparam ScaleArray row/column scaling diagonal matrix type, see \ref Array
66  /// \tparam CcsType ccs format, see \ref CCS
67  /// \tparam PermType permutation matrix type, see \ref Array
68  /// \tparam AugCcsType ccs format, see \ref AugCCS
69  /// \tparam PosArray position array type, see \ref Array
70  /// \tparam DiagType array used for storing diagonal array, see \ref Array
71  /// \tparam AugCrsType crs format, see \ref AugCRS
72  /// \param[in] s row scaling matrix from preprocessing
73  /// \param[in] ccs_A input matrix in ccs scheme
74  /// \param[in] t column scaling matrix from preprocessing
75  /// \param[in] p_inv row inverse permutation (with tail indices)
76  /// \param[in] qk permutated column index (with deferring)
77  /// \param[in] L lower part of decomposition
78  /// \param[in] L_start position array storing the starting locations of \a L
79  /// \param[in] d diagonal entries
80  /// \param[in] U augmented U matrix
81  /// \return diagonal entry value
82  /// \sa compute_ut, compute_l
83  ///
84  /// This routine computes the current diagonal entry of \f$\mathbf{D}\f$
85  /// Mathematically, this routine is to compute:
86  ///
87  /// \f[
88  /// d_{k}=\hat{A}[p_{k},q_{k}]-\mathbf{L}_{k,1:k-1}\mathbf{D}_{k-1}
89  /// \mathbf{U}_{1:k-1,k}
90  /// \f]
91  template <class ScaleArray, class CcsType, class PermType, class AugCcsType,
92  class PosArray, class DiagType, class AugCrsType>
93  inline typename DiagType::value_type compute_dk(
94  const ScaleArray &s, const CcsType &ccs_A, const ScaleArray &t,
95  const PermType &p_inv, const size_type qk, const AugCcsType &L,
96  const PosArray &L_start, const DiagType &d, const AugCrsType &U) const {
97  using value_type = typename DiagType::value_type;
98  using index_type = typename AugCrsType::index_type;
99 
100  value_type dk(0);
101  // load diagonal from the coefficient matrix A
102  do {
103  const size_type defer_thres = deferred_step();
104  // qk is c index
105  auto v_itr = ccs_A.val_cbegin(qk);
106  auto i_itr = ccs_A.row_ind_cbegin(qk);
107  const auto t_qk = t[qk];
108  for (auto last = ccs_A.row_ind_cend(qk); i_itr != last;
109  ++i_itr, ++v_itr) {
110  const auto A_idx = *i_itr;
111  const size_type idx = p_inv[A_idx];
112  // push to the sparse vector only if its in range _step+1:n
113  if (idx == defer_thres) {
114  dk = s[A_idx] * *v_itr * t_qk; // load diagonal entry
115  break;
116  }
117  }
118  } while (false);
119 
120  // compute the LDU part
121  if (_step) {
122  // get the deferred column handle
123  index_type aug_id = U.start_col_id(deferred_step());
124  while (!U.is_nil(aug_id)) {
125  // get the row index
126  const size_type row_idx = U.row_idx(aug_id);
127  hif_assert(row_idx < _step,
128  "compute_ut row index %zd should not exceed step %zd for U",
129  row_idx, _step);
130  hif_assert(row_idx < L_start.size(), "%zd exceeds the L_start size",
131  row_idx);
132  // compute d*U
133  const auto du = d[row_idx] * U.val_from_col_id(aug_id);
134  // get the starting position from L_start
135  auto L_v_itr = L.val_cbegin(row_idx) + L_start[row_idx];
136  auto L_i_itr = L.row_ind_cbegin(row_idx) + L_start[row_idx];
137  if (L_i_itr != L.row_ind_cend(row_idx) &&
138  size_type(*L_i_itr) == deferred_step())
139  dk -= du * *L_v_itr; // diagonal part
140  // advance augmented handle
141  aug_id = U.next_col_id(aug_id);
142  } // while
143  }
144  return dk;
145  }

147  /// \brief dual of the \ref compute_l, i.e. computing the row of U
148  /// \tparam ScaleArray row scaling diagonal matrix type, see \ref Array
149  /// \tparam CrsType crs format, see \ref CRS
150  /// \tparam PermType permutation matrix type, see \ref Array
151  /// \tparam AugCcsType ccs format, see \ref AugCCS
152  /// \tparam DiagType diagonal matrix type, see \ref Array
153  /// \tparam AugCrsType crs format, see \ref AugCRS
154  /// \tparam PosArray position array type, see \ref Array
155  /// \tparam SpVecType work array for current row vector, see \ref SparseVector
156  /// \param[in] s row scaling matrix from preprocessing
157  /// \param[in] crs_A input matrix in crs format
158  /// \param[in] t column scaling matrix from preprocessing
159  /// \param[in] pk permuted row index (with deferring)
160  /// \param[in] q_inv inverse column permutation (with tails indices)
161  /// \param[in] L lower part
162  /// \param[in] d diagonal vector
163  /// \param[in] U upper part
164  /// \param[in] U_start leading row positions of current \ref _step
165  /// \param[in,out] ut upon input, it contains preloaded row in A via
166  /// load_arow; upon output, it contains the factor of
167  /// \f$\mathbf{u}_k^T\f$.
168  /// \sa compute_l
169  ///
170  /// This routine computes the current row vector of \f$\mathbf{U}\f$
171  /// (w/o diagonal scaling). Mathematically, this routine is to compute:
172  ///
173  /// \f[
174  /// \mathbf{u}_{k}^{T}=
175  /// \hat{\mathbf{A}}[p_{k},\mathbf{q}_{k+1:n}]-
176  /// \mathbf{L}_{k,1:k-1}\mathbf{D}_{k-1}
177  /// \mathbf{U}_{1:k-1,k+1:n}
178  /// \f]
179  ///
180  /// It's worth noting that, conceptally, the formula above is nothing but
181  /// a vector matrix operation. However, standard implementation won't give
182  /// good performance (especially with consideration of cache performance),
183  /// this is because \f$\mathbf{L}\f$ is stored in column major whereas
184  /// row major for \f$\mathbf{U}\f$. Therefore, the actual implementation
185  /// is in the fashion that loops through \f$\mathbf{U}\f$ in row major
186  /// while keeping \f$\mathbf{L}\f$ as much static as possible.
187  ///
188  /// Regarding computation cost, it takes
189  /// \f$\mathcal{O}(\cup_j \textrm{nnz}(\mathbf{U}_{j,:}))\f$, where
190  /// \f$l_{kj}\neq 0\f$ for computing the vector matrix operation.
191  template <class ScaleArray, class CrsType, class PermType, class AugCcsType,
192  class DiagType, class AugCrsType, class PosArray, class SpVecType>
193  void compute_ut(const ScaleArray &s, const CrsType &crs_A,
194  const ScaleArray &t, const size_type pk,
195  const PermType &q_inv, const AugCcsType &L, const DiagType &d,
196  const AugCrsType &U, const PosArray &U_start,
197  SpVecType &ut) const {
198  // compilation checking
199  using index_type = typename PosArray::value_type;
200 
201  // reset sparse buffer
202  ut.reset_counter();
203 
204  // first load the A row
205  _load_arow(s, crs_A, t, pk, q_inv, ut);
206 
207  // if not first step
208  if (_step) {
209  // get the starting row ID with deferring
210  index_type aug_id = L.start_row_id(deferred_step());
211  while (!L.is_nil(aug_id)) {
212  // get the column index
213  const size_type col_idx = L.col_idx(aug_id);
214  hif_assert(
215  col_idx < _step,
216  "compute_ut column index %zd should not exceed step %zd for L",
217  col_idx, _step);
218  hif_assert(col_idx < U_start.size(), "%zd exceeds the U_start size",
219  col_idx);
220  // compute L*d
221  const auto ld = L.val_from_row_id(aug_id) * d[col_idx];
222  // get the starting position from U_start
223  auto U_v_itr = U.val_cbegin(col_idx) + U_start[col_idx];
224  auto U_i_itr = U.col_ind_cbegin(col_idx) + U_start[col_idx],
225  U_i_last = U.col_ind_cend(col_idx);
226  if (U_i_itr != U_i_last && size_type(*U_i_itr) == deferred_step()) {
227  // advance to offsets
228  ++U_i_itr;
229  ++U_v_itr;
230  }
231  // for loop to go thru all entries in U
232  for (; U_i_itr != U_i_last; ++U_i_itr, ++U_v_itr) {
233  const auto idx = *U_i_itr;
234  hif_assert(idx > deferred_step(),
235  "U index %zd in computing ut should greater than step "
236  "%zd(defers:%zd)",
237  idx, _step, _defers);
238  if (ut.push_back(idx, _step))
239  ut.vals()[idx] = -ld * *U_v_itr;
240  else
241  ut.vals()[idx] -= ld * *U_v_itr;
242  }
243  // advance augmented handle
244  aug_id = L.next_row_id(aug_id);
245  } // while
246  }
247  }

249  /// \brief compute the column vector of L of current step
250  /// \tparam ScaleArray row/column scaling diagonal matrix type, see \ref Array
251  /// \tparam CcsType ccs format, see \ref CCS
252  /// \tparam PermType permutation matrix type, see \ref Array
253  /// \tparam AugCcsType ccs format, see \ref AugCCS
254  /// \tparam PosArray position array type, see \ref Array
255  /// \tparam DiagType array used for storing diagonal array, see \ref Array
256  /// \tparam AugCrsType crs format, see \ref AugCRS
257  /// \tparam SpVecType sparse vector for storing l, see \ref SparseVector
258  /// \param[in] s row scaling matrix from preprocessing
259  /// \param[in] ccs_A input matrix in ccs scheme
260  /// \param[in] t column scaling matrix from preprocessing
261  /// \param[in] p_inv row inverse permutation (with tail indices)
262  /// \param[in] qk permutated column index (with deferring)
263  /// \param[in] L lower part of decomposition
264  /// \param[in] L_start position array storing the starting locations of \a L
265  /// \param[in] d diagonal entries
266  /// \param[in] U augmented U matrix
267  /// \param[in,out] l upon input, it contains preloaded column in A via
268  /// load_acol; upon output, it contains the factor of
269  /// \f$\mathbf{\ell}_k\f$.
270  /// \sa compute_ut
271  ///
272  /// This routine computes the current column vector of \f$\mathbf{L}\f$
273  /// (w/o diagonal scaling). Mathematically, this routine is to compute:
274  ///
275  /// \f[
276  /// \mathbf{\ell}_{k}=
277  /// \hat{\mathbf{A}}[\mathbf{p}_{k+1:n},q_{k}]-
278  /// \mathbf{L}_{k+1:n,1:k-1}\mathbf{D}_{k-1}
279  /// \mathbf{U}_{1:k-1,k}
280  /// \f]
281  ///
282  /// It's worth noting that, conceptally, the formula above is nothing but
283  /// a matrix vector operation. However, standard implementation won't give
284  /// good performance (especially with consideration of cache performance),
285  /// this is because \f$\mathbf{L}\f$ is stored in column major whereas
286  /// row major for \f$\mathbf{U}\f$. Therefore, the actual implementation
287  /// is in the fashion that loops through \f$\mathbf{L}\f$ in column major
288  /// while keeping \f$\mathbf{U}\f$ as much static as possible.
289  ///
290  /// Regarding the complexity cost, it takes
291  /// \f$\mathcal{O}(\cup_i \textrm{nnz}(\mathbf{L}_{:,i}))\f$, where
292  /// \f$u_{ik}\neq 0\f$ for computing the matrix vector operation.
293  template <class ScaleArray, class CcsType, class PermType, class AugCcsType,
294  class PosArray, class DiagType, class AugCrsType, class SpVecType>
295  void compute_l(const ScaleArray &s, const CcsType &ccs_A, const ScaleArray &t,
296  const PermType &p_inv, const size_type qk, const AugCcsType &L,
297  const PosArray &L_start, const DiagType &d,
298  const AugCrsType &U, SpVecType &l) const {
299  // compilation checking
300  using index_type = typename PosArray::value_type;
301  static constexpr bool IS_SYMM = false; // never symmetric for pivoting code
302 
303  // clear sparse counter
304  l.reset_counter();
305 
306  // load A column
307  _load_acol<IS_SYMM>(s, ccs_A, t, p_inv, qk, size_type(0), l);
308 
309  // if not first step
310  if (_step) {
311  // get the deferred column handle
312  index_type aug_id = U.start_col_id(deferred_step());
313  while (!U.is_nil(aug_id)) {
314  // get the row index
315  const size_type row_idx = U.row_idx(aug_id);
316  hif_assert(row_idx < _step,
317  "compute_ut row index %zd should not exceed step %zd for U",
318  row_idx, _step);
319  hif_assert(row_idx < L_start.size(), "%zd exceeds the L_start size",
320  row_idx);
321  // compute d*U
322  const auto du = d[row_idx] * U.val_from_col_id(aug_id);
323  // get the starting position from L_start
324  auto L_v_itr = L.val_cbegin(row_idx) + L_start[row_idx];
325  auto L_i_itr = L.row_ind_cbegin(row_idx) + L_start[row_idx],
326  L_i_last = L.row_ind_cend(row_idx);
327  if (L_i_itr != L_i_last && size_type(*L_i_itr) == deferred_step()) {
328  ++L_i_itr;
329  ++L_v_itr;
330  }
331  for (; L_i_itr != L_i_last; ++L_i_itr, ++L_v_itr) {
332  // convert to c index
333  const auto idx = *L_i_itr;
334  hif_assert(idx > deferred_step(),
335  "L index %zd in computing l should greater than step "
336  "%zd(defers:%zd)",
337  idx, _step, _defers);
338  // compute this entry, if index does not exist, assign new value to
339  // -L*d*u, else, -= L*d*u
340  if (l.push_back(idx, _step))
341  l.vals()[idx] = -du * *L_v_itr;
342  else
343  l.vals()[idx] -= *L_v_itr * du;
344  }
345  // advance augmented handle
346  aug_id = U.next_col_id(aug_id);
347  } // while
348  }
349  }

351  /// \brief compress L and U and update their corresponding starting positions
352  /// \tparam CsType either \ref AugCRS or \ref AugCCS
353  /// \tparam PosArray position array type, see \ref Array
354  /// \param[in,out] T strictly lower (L) or upper (U) matrices
355  /// \param[in,out] start local positions
356  /// \note This routine essentially compresses deferred_step to _step
357  /// \note All other routines should be called before this one, and
358  /// deferred_step therein are conceptually the current Crout step
359  template <class CsType, class PosArray>
360  inline void update_compress(CsType &T, PosArray &start) const {
361  using index_type = typename PosArray::value_type;
362 
363  // get the starting augmented ID
364  index_type aug_id = T.start_aug_id(deferred_step());
365  while (!T.is_nil(aug_id)) {
366  // get corresponding primary index, i.e., row index in CRS while column
367  // index for CCS
368  const index_type primary_idx = T.primary_idx(aug_id);
369  auto itr = T.ind_begin(primary_idx) + start[primary_idx];
370  *itr -= _defers; // compress
371  ++start[primary_idx]; // increment next position
372  // advance augment handle
373  aug_id = T.next_aug_id(aug_id);
374  }
375  // need to update newly added entry at the end
376  start[_step] = 0;
377  T.secondary_start()[_step] = T.secondary_start()[deferred_step()];
378  T.secondary_end()[_step] = T.secondary_end()[deferred_step()];
379  T.secondary_counts()[_step] = T.secondary_counts()[deferred_step()];
380  }

382  /// \brief estimate the inverse norm with augmented ds
383  /// \tparam AugCsType Augmented compressed storage, see \ref AugCRS or
384  /// \ref AugCCS
385  /// \tparam KappaArray Inverse norm array type, see \ref Array
386  /// \param[in] T Augmented compressed storage input
387  /// \param[in,out] kappa history of inverse norms
388  /// \param[in] entry (optional) concatenate \a entry to T_{k-1} for computing
389  /// the absolute conditioning of leading block of T, default
390  /// value is k, i.e., \a _step
391  template <class AugCsType, class KappaArray>
392  inline bool update_kappa(const AugCsType &T, KappaArray &kappa,
393  const size_type entry = 0u) {
394  using value_type = typename AugCsType::value_type;
395  using index_type = typename AugCsType::index_type;
396  constexpr static bool one = true, neg_one = false;
397 
398  if (!_step) {
399  kappa[0] = value_type(1);
400  return one;
401  }
402 
403  // we need to loop through all entries in row step
404  value_type sum(0);
405 
406  // start augmented ID
407  index_type aug_id = T.start_aug_id(entry ? entry : deferred_step());
408  while (!T.is_nil(aug_id)) {
409  const index_type primary_idx = T.primary_idx(aug_id);
410  hif_assert((size_type)primary_idx < kappa.size(),
411  "%zd exceeds the solution size", (size_type)primary_idx);
412  hif_assert((size_type)primary_idx < _step,
413  "the matrix U should only contain the strict upper case");
414  sum += kappa[primary_idx] * T.val_from_aug_id(aug_id);
415  // advance augment handle
416  aug_id = T.next_aug_id(aug_id);
417  }
418  const value_type k1 = value_type(1) - sum, k2 = value_type(-1) - sum;
419  if (std::abs(k1) < std::abs(k2)) {
420  kappa[_step] = k2;
421  return neg_one;
422  }
423  kappa[_step] = k1;
424  return one;
425  }

437  /// \brief interchagen with pivot for pivoting with Crout LU
438  /// \tparam AugCsType Augmented compressed data types, see \ref AugCRS or
439  /// \ref AugCCS
440  /// \tparam PermType Permutation array type, see \ref BiPermMatrix
441  /// \tparam InvPermType Inverse permutation array type
442  /// \tparam DiagArrayType Diagonal array type, see \ref Array
443  /// \tparam SpVecType Array type for storing current row or column of A
444  /// \param[in] pivot Pivoting position
445  /// \param[in,out] T Either lower or upper factors from Crout LU and step k
446  /// \param[in,out] p Permutation array
447  /// \param[in,out] p_inv Inverse permutation mapping of \a p
448  /// \param[in,out] d diagonal array
449  /// \param[in,out] a_k k-th row or column of A (scaled and permuted)
450  template <class AugCsType, class PermType, class InvPermType,
451  class DiagArrayType, class SpVecType>
452  inline void interchange(const size_type pivot, AugCsType &T, PermType &p,
453  InvPermType &p_inv, DiagArrayType &d,
454  SpVecType &a_k) const {
455  // swap permutation
456  std::swap(p[deferred_step()], p[pivot]);
457  std::swap(p_inv[p[deferred_step()]], p_inv[p[pivot]]);
458  // interchanage in the L or U factors
459  T.interchange_entries(deferred_step(), pivot);
460  // swap current diagonal with the pivot entry in k-th row or column of A
461  std::swap(d[deferred_step()], a_k[pivot]);
462  }

464  /// \brief Apply inverse-based rook pivoting
465  /// \tparam ScaleArray Row/column scaling array, see \ref Array
466  /// \tparam PermType Permutation array type
467  /// \tparam PermType2 Inverse permutation array type
468  /// \tparam CcsType Ccs storage type, see \ref CCS
469  /// \tparam CrsType Crs storage type, see \ref CRS
470  /// \tparam CroutInfoStreamer Crout update information streamer
471  /// \tparam PosArray Integer array type for storing position information
472  /// \tparam PivotUnaryOpL Unary predictor for finding pivot in L
473  /// \tparam PivotUnaryOpU Unary predictor for finding pivot in U
474  /// \tparam AugCcsType Augmented ccs storage type, see \ref AugCCS
475  /// \tparam DiagType Diagonal array type, see \ref Array
476  /// \tparam AugCrsType Augmented crs storage type, see \ref AugCRS
477  /// \tparam SpVecType Sparse vector workspace, see \ref SparseVector
478  /// \param[in] s row scaling vector
479  /// \param[in] t column scaling vector
480  /// \param[in] A_ccs input coefficient matrix in CCS
481  /// \param[in] A_crs input coefficient matrix in CRS
482  /// \param[in] m leading dimension size
483  /// \param[in] gamma thresholding for rook pivoting
484  /// \param[in] max_steps maximum number of pivoting steps
485  /// \param[in] Crout_info information streamer (empty for non-verbose mode)
486  /// \param[in] L_start starting positions in columns for \a L
487  /// \param[in] U_start starting positions in rows for \a U
488  /// \param[in] op_l unary operator for finding a pivot in L
489  /// \param[in] op_ut unary operator for finding a pivot in U
490  /// \param[in,out] p row permutation array, will be updated due to pivoting
491  /// \param[in,out] p_inv inverse of row permutation, will be updated
492  /// \param[in,out] q column permutation array, will be updated due to pivoting
493  /// \param[in,out] q_inv inverse of column permutation, will be updated
494  /// \param[in,out] L lower factor, will be permuted if necessary
495  /// \param[in,out] d diagonal factor (vector), will be updated for current
496  /// step \param[in,out] U upper factor, will be permuted if ncessary
497  /// \param[out] l sparse vector work space for computing \f$\mathbf{\ell}_k\f$
498  /// \param[out] ut sparse vector work space for computing \f$\mathbf{u}_k^T\f$
499  /// \return number of pivots in row and column swappings.
500  /// \sa interchange
501  ///
502  /// Regarding \a op_l and \a op_u, we find a pivot in L if and only if it
503  /// satisfies op_l condition, i.e., op_l(pivot_l)==true.
504  template <class ScaleArray, class PermType, class PermType2, class CcsType,

540  });

559  } else if (pivot_step > 0)

561  } else if (pivot_step > 0)
562  break;

566  //-----------------

569  // important! we need to reset dense flags in ut
570  if (pivot_step) ut.restore_cur_state();
571  compute_ut(s, A_crs, t, p[k], q_inv, L, d, U, U_start, ut);
572  // sort indices based on magnitude
573  ut.sort_indices([&](const index_type a, const index_type b) {
574  return std::abs(ut[a]) > std::abs(ut[b]);
575  });
576  // find pivot
577  const auto pivot_row = ut.find_if(op_ut);
578  // compute diagonal if necessary
579  if (l_pivot)

581  if (pivot_row != std::numeric_limits<size_type>::max()) {
582  hif_assert(pivot_row > k && pivot_row < m,
583  "wrong pivot range (pvt=%zd), (k,m)=(%zd,%zd)", pivot_row, k,
584  m);
585  // d may be updated
586  if (std::abs(d[k]) <
587  gamma * std::max(std::abs(ut[pivot_row]), std::abs(d[k]))) {
588  Crout_info(
589  " interchanging cols current=%zd (deferrals=%zd) and pivot=%zd",
590  k, _defers, size_type(pivot_row));
591  // row pivoting occurs
592  ++row_pivots;
593  // swap
594  interchange(pivot_row, U, q, q_inv, d, ut);
595  } else
596  break;
597  } else
598  break;
599  }
600  if (pivot_step > max_steps)
601  hif_warning(" could not satisfy pivoting requirement in %d iterations",
602  max_steps);
603  // Crout_info(" could not satisfy pivoting requirement in %d iterations",
604  // max_steps);
605  return std::make_pair(col_pivots, row_pivots);
606  }
