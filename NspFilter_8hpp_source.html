42 /// \class NspFilter
43 /// \brief filter attached to \ref HIF for filtering null space components
44 /// \ingroup itr
45 ///
46 /// Flexible design of null space components filter. The default mode is
47 /// constant filter, i.e., there is a constant mode in the system. Two ways to
48 /// customize the filter, 1) override the \ref NspFilter::user_filter function
49 /// or 2) pass in a callback function.
50 ///
51 /// For both 1) and 2), the interface is \a void(void*,std::size_t,char) where
52 /// the first parameter is in/out array whose length is specified by the second
53 /// parameter. The last parameter indicates the data types, which follow the
54 /// BLAS conventions, i.e., 'd' for double, 's' for single, etc.
55 ///
56 /// As to the user call back function, it is wrapped inside \a std::function,
57 /// which allows easy binding external user data. For more, see \a std::bind.
58 ///
59 /// In most cases, a (partial) constant mode is enough. If user-customized
60 /// behaviors are needed, then it will be easy to work with user callback
61 /// in C++ and override \ref NspFilter::user_filter in other programming
62 /// environments, e.g., Python.
63 class NspFilter {
64  public:
65  using user_cb_type =
66  std::function<void(void *, const std::size_t, const char)>;
67  ///< user call back type
68  enum {
69  CONSTANT = 0, ///< constant null space, can applied on a certain range
70  USER_OR, ///< user override virtual function
71  USER_CB, ///< user callback function
72  };
73 
74  /// \brief default constructor
75  /// \param[in] start (optional) starting position for constant null space
76  /// \param[in] end (optional) ending position for constant null space
77  explicit NspFilter(const std::size_t start = 0,
78  const std::size_t end = static_cast<std::size_t>(-1))
79  : _type(CONSTANT) {
80  _cst_rg[0] = start;
81  _cst_rg[1] = end;
82  }
83 
84  /// \brief constructor with user callback type
85  /// \param[in] f external call back
86  /// \note allow implicit construction
87  NspFilter(const user_cb_type &f) : _type(USER_CB), _user_f(f) {
88  _cst_rg[0] = 0;
89  _cst_rg[1] = static_cast<std::size_t>(-1);
90  }
91 
92  /// \brief constructor with rvalue reference of callback type
93  /// \param[in] f external call back
94  /// \note \a f will become invalid state
95  /// \note allow implicit construction
96  NspFilter(user_cb_type &&f) : _type(USER_CB), _user_f(std::move(f)) {
97  _cst_rg[0] = 0;
98  _cst_rg[1] = static_cast<std::size_t>(-1);
99  }

120  /// \param[in] start (optional) starting range default is 0
121  /// \param[in] end (optional) ending position, default is max(size_t)

126  _cst_rg[1] = end;
127  _type = CONSTANT;
128  }

130  /// \brief set null space callback
131  inline void set_nsp_callback(const user_cb_type &f) {
132  _user_f = f;
133  if (!_user_f) hif_error("empty user callback was not attached!");
134  _type = USER_CB;
135  }

137  /// \brief filtering out null space component from \a x
138  template <class T>
139  inline void filter(T *x, const std::size_t n) const {
140  switch (_type) {
141  case CONSTANT:
142  _const_filter(x, n);
143  break;
144  case USER_OR:
145  user_filter((void *)x, n, ValueTypeTrait<T>::signature);
146  break;
147  default:
148  if (!_user_f) hif_error("user callback was not attached!");
149  _user_f((void *)x, n, ValueTypeTrait<T>::signature);
150  break;
151  }
152  }
153 
154  /// \brief overridable routine for filtering
155  virtual void user_filter(void *, const std::size_t, const char) const {
156  hif_error("user routine was not overriden!");
157  }
158 
159  protected:
160  template <class T>
161  inline void _const_filter(T *x, const std::size_t n) const {
162  const std::size_t start = _cst_rg[0];
163  const std::size_t end =
164  (_cst_rg[1] == static_cast<std::size_t>(-1) || _cst_rg[1] < _cst_rg[0])
165  ? n
166  : _cst_rg[1];
167  hif_error_if(end < start || end > n,
168  "wrong range (start,end,n)=(%zd,%zd,%zd)", start, end, n);
169  if (start == end) return;
170  // compute I-one*one^t/n, which is equiv as shifting
171  const T shift_val =
172  std::accumulate(x + start, x + end, T(0)) / T(end - start);
173  for (std::size_t i(start); i < end; ++i) x[i] -= shift_val;
174  }
175 
176  protected:
177  int _type; ///< type
178  std::size_t _cst_rg[2]; ///< constant range
179  user_cb_type _user_f; ///< user call back
180 };
