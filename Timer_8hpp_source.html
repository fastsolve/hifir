107 /// \class Timer
108 /// \brief C++11 standard timer implementation
109 /// \tparam Clock ClockType, default is TIMER_SYSTEM_CLOCK
110 /// \tparam RepType time period representation type, default is \a double
111 /// \note Using the timer in a local environment is preferred
112 /// \ingroup util
113 template <ClockType Clock = TIMER_SYSTEM_CLOCK, class RepType = double>
114 class Timer {
115  public:
116  constexpr static ClockType CLOCK = Clock; ///< clock type
117  typedef typename internal::ClockTrait<CLOCK>::clock_type clock_type;
118  ///< std clock type
119  typedef typename internal::TimePointTrait<CLOCK>::time_point time_point;
120  ///< time point
121  typedef RepType default_rep_type; ///< representation type
122 
123  /// \brief start timer
124  /// \sa finish
125  inline void start() { _start = clock_type::now(); }
126 
127  /// \brief finish timer
128  /// \sa start
129  inline void finish() { _end = clock_type::now(); }
130 
131  /// \brief get time report
132  /// \tparam _RepType representation type
133  /// \tparam Unit report unit type, default is TIMER_SECONDS
134  /// \return a time period representation in RepType and Unit
135  template <class _RepType, ClockUnit Unit = TIMER_SECONDS>
136  inline _RepType time() const {
137  using unit_t = typename internal::ClockUnitTrait<Unit>::unit_type;
138  std::chrono::duration<RepType, unit_t> period = _end - _start;
139  return period.count();
140  }
141 
142  /// \brief get time report in seconds
143  inline default_rep_type time() const { return time<default_rep_type>(); }
144 
145  /// \brief get time report in milliseconds

147  return time<default_rep_type, TIMER_MILLISECONDS>();
148  }
