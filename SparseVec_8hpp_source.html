42 /// \class IndexValueArray
43 /// \tparam ValueType value type, e.g. \a double, \a float, etc
44 /// \tparam IndexType index type, e.g. \a int
45 /// \ingroup ds
46 /// \note We use \a std::vector for internal data management
47 ///
48 /// This data structure stores two essential arrays: 1) a dense array of
49 /// application "values" and 2) a sparse array of the activated indices
50 /// in the value array. By dense, we mean an array that supports random
51 /// access; by sparse, we mean an array that only contains what we need.
52 /// Notice that this data structure is mainly used in sparse matrix
53 /// operations for efficiency, where we just need to manipulate the indices
54 /// without worry about touching the values. Be aware the it's may be
55 /// necessary that the sparse array have the same length (in terms of memory)
56 /// as that of the dense array due to the fact that we cannot estimate the
57 /// sparse size precisely.
58 ///
59 /// For the efficiency purpose, we explicitly create an size counter,
60 /// \ref _counts, to avoid calling the \a std::vector::push_back, which is
61 /// quite expansive.
62 template <class ValueType, class IndexType>
63 class IndexValueArray {
64  public:
65  typedef ValueType value_type; ///< value type
66  typedef IndexType index_type; ///< index type
67  typedef std::vector<value_type> array_type; ///< value array
68  typedef std::vector<index_type> iarray_type; ///< index array
69  typedef typename array_type::size_type size_type; ///< size
70  typedef IndexValueArray this_type; ///< handy type wrapper
71  typedef typename ValueTypeTrait<value_type>::value_type scalar_type;
72  ///< scalar (real) type
73 
74  /// \brief default constructor
75  IndexValueArray() : _vals(), _inds(), _counts(0u) {}
76 
77  /// \brief constructor with dense and (optionally) sparse sizes
78  /// \param[in] d_n dense size
79  /// \param[in] s_n sparse size, if == 0, then use \a d_n
80  explicit IndexValueArray(const size_type d_n, const size_type s_n = 0u)
81  : _vals(d_n), _inds(s_n ? s_n : d_n), _counts(0u) {}
82 
83  // allow moving methods
84  IndexValueArray(this_type &&) = default;
85  this_type &operator=(this_type &&) = default;
86 
87  // copying methods are banned
88  IndexValueArray(const this_type &) = delete;
89  this_type &operator=(const this_type &) = delete;
90 
91  /// \brief resize the buffer
92  /// \param[in] d_n dense size
93  /// \param[in] s_n sparse size, if == 0 (default), then use \a d_n
94  inline void resize(const size_type d_n, const size_type s_n = 0u) {
95  _vals.resize(d_n);
96  _inds.resize(s_n ? s_n : d_n);
97  }
98 
99  /// \brief get the number of counts
100  inline size_type size() const { return _counts; }
101 
102  /// \brief reset counter
103  inline void reset_counter() { _counts = 0u; }
104 
105  /// \brief check if empty or not
106  inline bool empty() const { return _counts == 0u; }
107 
108  /// \brief sort the indices
109  /// \note Complexity: \f$\mathcal{O}(n\log n)\f$, \f$n\f$ is \ref _counts
110  inline void sort_indices() {
111  std::sort(_inds.begin(), _inds.begin() + _counts);
112  }
113 
114  /// \brief sort the indices given a unary operator
115  /// \tparam UnaryOp unary operator type
116  template <class UnaryOp>
117  inline void sort_indices(const UnaryOp &op) {
118  std::sort(_inds.begin(), _inds.begin() + _counts, op);
119  }
120 
121  /// \brief push back an index
122  /// \param[in] i index
123  inline void push_back(const size_type i) {
124  hif_assert(i < _vals.size(), "%zd exceeds the value array size", i);
125  hif_assert(_inds.size(), "empty array, did you call resize?");
126  _inds[_counts++] = i;
127  }
128 
129  /// \brief get the index
130  /// \param[in] i local index in range of _counts
131  inline size_type idx(const size_type i) const {
132  hif_assert(i < _counts, "%zd exceeds index array bound %zd", i, _counts);
133  return _inds[i];
134  }
135 
136  /// \brief get the value
137  /// \param[in] i local index in range of _counts
138  inline value_type val(const size_type i) const { return _vals[idx(i)]; }
139 
140  /// \brief get the 1 norm of the vector
141  inline scalar_type norm1() const {
142  scalar_type tmp(0);
143  for (size_type i(0); i < _counts; ++i) tmp += std::abs(val(i));
144  return tmp;
145  }
146 
147  /// \brief get the 2 norm squared of the vector
148  inline scalar_type norm2_sq() const {
149  scalar_type tmp(0);
150  for (size_type i(0); i < _counts; ++i)
151  tmp += real(val(i) * conjugate(val(i)));
152  return tmp;
153  }
154 
155  /// \brief get the 2 norm of the vector
156  inline value_type norm2() const { return std::sqrt(norm2_sq()); }
157 
158  /// \brief get the maximum magnitude value, only works wit real number
159  inline typename ValueTypeTrait<value_type>::value_type max_mag() const {
160  using v_t = typename ValueTypeTrait<value_type>::value_type;
161  v_t v(0);
162  for (size_type i(0); i < _counts; ++i) v = std::max(v, std::abs(val(i)));
163  return v;
164  }
165 
166  /// \brief get the index corresponding maximum magnitude
167  inline size_type imax_mag() const {
168  if (!_counts) return std::numeric_limits<size_type>::max();
169  return *std::max_element(_inds.cbegin(), _inds.cbegin() + _counts,
170  [&](const index_type i, const index_type j) {
171  return std::abs(_vals[i]) < std::abs(_vals[j]);
172  });
173  }
174 
175  /// \brief get the index corresponding maximum magnitude with index bound
176  template <class UnaryOp>
177  inline size_type imax_mag(const UnaryOp &op) const {
178  // if empty, return max value of size_t
179  if (!_counts) return std::numeric_limits<size_type>::max();
180  // find a starting position, where it is within the bound, i.e., typically
181  // the size of leading block
182  auto pos = std::find_if(_inds.cbegin(), _inds.cbegin() + _counts, op);
183  // if all tail region, return max value of size_t
184  if (pos == _inds.cbegin() + _counts)
185  return std::numeric_limits<size_type>::max();
186  size_type j(*pos);
187  for (size_type i(0); i < _counts; ++i) {
188  if (!op(_inds[i])) continue;
189  if (std::abs(_vals[j]) < std::abs(_vals[_inds[i]])) j = _inds[i];
190  }
191  return j;
192  }
193 
194  /// \brief find the first entry given a unary operator
195  template <class UnaryOp>
196  inline size_type find_if(const UnaryOp &op) const {
197  if (!_counts) return std::numeric_limits<size_type>::max();
198  auto pos = std::find_if(_inds.cbegin(), _inds.cbegin() + _counts, op);
199  // if all tail region, return max value of size_t
200  if (pos == _inds.cbegin() + _counts)
201  return std::numeric_limits<size_type>::max();
202  return *pos;
203  }
204 
205  /// \brief inplace scaling
206  /// \param[in] alpha scaling factor, i.e., v=alpha*v
207  inline void scale(const value_type alpha) {
208  for (size_type i(0); i < _counts; ++i) _vals[_inds[i]] *= alpha;
209  }
210 
211  /// \brief operator access
212  /// \param[in] i index in range of dense size, one-based aware
213  inline value_type &operator[](const size_type i) {
214  hif_assert(i < _vals.size(), "%zd exceeds value size bound", i);
215  return _vals[i];
216  }
217 
218  /// \brief operator access, constant version
219  /// \param[in] i index in range of dense size, one-based aware
220  inline const value_type &operator[](const size_type i) const {
221  hif_assert(i < _vals.size(), "%zd exceeds value size bound", i);
222  return _vals[i];
223  }
224 
225  // utils
226  inline array_type & vals() { return _vals; }
227  inline iarray_type & inds() { return _inds; }
228  inline const array_type & vals() const { return _vals; }
229  inline const iarray_type &inds() const { return _inds; }
230 
231  protected:
232  array_type _vals; ///< values
233  iarray_type _inds; ///< indices
234  size_type _counts; ///< current counts
235 };

237 /// \class SparseVector
238 /// \tparam ValueType value type, e.g. \a double
239 /// \tparam IndexType index type, e.g. \a int
240 /// \ingroup ds
241 ///
242 /// This class is mainly used in Crout update. The total memory cost is linear
243 /// with respect to the matrix system size and computation cost is bounded
244 /// by \f$\mathcal{O}(\textrm{lnnz}\log \textrm{lnnz}\f$, where
245 /// \f$\textrm{lnnz}\f$ is the local number of nonzeros.
246 template <class ValueType, class IndexType>
247 class SparseVector : public IndexValueArray<ValueType, IndexType> {
248  typedef IndexValueArray<ValueType, IndexType> _base; ///< base
249 
250  public:
251  typedef typename _base::value_type value_type; ///< value
252  typedef typename _base::index_type index_type; ///< index
253  typedef typename _base::size_type size_type; ///< size
254  typedef typename _base::array_type array_type; ///< value array
255  typedef typename _base::iarray_type iarray_type; ///< index array
256  typedef SparseVector this_type; ///< handy type wrapper
257 
258  private:
259  constexpr static index_type _EMPTY = static_cast<index_type>(-1);
260  ///< empty flag
261 
262  public:
263  /// \brief default constructor
264  SparseVector() = default;
265 
266  /// \brief constructor with dense and sparse sizes
267  /// \param[in] d_n dense size
268  /// \param[in] s_n sparse index size, if == 0 (default), then use \a d_n
269  explicit SparseVector(const size_type d_n, const size_type s_n = 0u)
270  : _base(d_n, s_n),
271  _dense_tags(d_n, static_cast<index_type>(-1)),
272  _sparse_tags(_inds.size(), false) {}
273 
274  // ban copy methods
275  SparseVector(const this_type &) = delete;
276  this_type &operator=(const this_type &) = delete;
277 
278  // allow move
279  SparseVector(this_type &&) = default;
280  this_type &operator=(this_type &&) = default;
281 
282  /// \brief resize sparse vector
283  /// \param[in] d_n dense size
284  /// \param[in] s_n sparse size, if == 0 (default), using \a d_n
285  /// \note Overloading the base version, i.e. \ref _base::resize
286  inline void resize(const size_type d_n, const size_type s_n = 0u) {
287  constexpr static auto _empty = _EMPTY;
288  _base::resize(d_n, s_n);
289  _dense_tags.resize(d_n);
290  _sparse_tags.resize(_inds.size());
291  // NOTE that we enforce to reset ALL tags to default/empty stage
292  std::fill(_dense_tags.begin(), _dense_tags.end(), _empty);
293  std::fill(_sparse_tags.begin(), _sparse_tags.end(), false);
294  }
295 
296  /// \brief mark an index to be dropped node
297  /// \param[in] i i-th entry in _inds (C-based)
298  inline void mark_delete(const size_type i) {
299  hif_assert(i < _counts, "%zd exceeds size bound %zd", i, _counts);
300  _sparse_tags[i] = true;
301  }
302 
303  /// \brief compress indices
304  inline void compress_indices() {
305  size_type i = 0u;
306  for (auto j = i; j < _counts; ++j) {
307  if (!_sparse_tags[j])
308  _inds[i++] = _inds[j];
309  else
310  _sparse_tags[j] = false; // NOTE that the flag reset here
311  }
312  _counts = i;
313  }
314 
315  /// \brief push back an index at a specific Crout update step
316  /// \param[in] i index
317  /// \param[in] step Crout update step
318  /// \return if \a true, then a new value has been pushed back
319  /// \note The tag array is for unsure the union of the indices
320  inline bool push_back(const index_type i, const size_type step) {
321  hif_assert((size_type)i < _dense_tags.size(), "%zd exceeds the dense size",
322  (size_type)i);
323  if (_dense_tags[i] != static_cast<index_type>(step)) {
324  _base::push_back(i);
325  _dense_tags[i] = step;
326  return true; // got a new value
327  }
328  return false; // not a new value
329  }
330 
331  /// \brief reset current state
332  inline void restore_cur_state() {
333  const size_type n = _counts;
334  for (size_type i(0); i < n; ++i) _dense_tags[_base::idx(i)] = _EMPTY;
335  }
336 
337  protected:
338  using _base::_counts; ///< bring in base counts
339  using _base::_inds; ///< bring in base value array
340  using _base::_vals; ///< bring in base index array
341  iarray_type _dense_tags; ///< dense tag for union
342  std::vector<bool> _sparse_tags; ///< sparse binary tag for drop
343 };
