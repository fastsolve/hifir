40 /// \brief apply numerical dropping based on dropping tolerance
41 /// \tparam KappaType data type for inverse norm
42 /// \tparam SpVecType sparse vector type, see \ref SparseVector
43 /// \param[in] tau dropping threshold parameter
44 /// \param[in] kappa inverse condition number of either L or U
45 /// \param[in,out] v sparse vector
46 /// \sa apply_space_dropping
47 /// \ingroup inv
48 template <class KappaType, class SpVecType>
49 inline void apply_num_dropping(const double tau, const KappaType kappa,
50  SpVecType &v) {
51  using size_type = typename SpVecType::size_type;
52 
53  const KappaType coeff = tau / kappa;
54 
55  if (coeff <= 0.0) return;
56 
57  const size_type n = v.size();
58  for (size_type i = 0u; i < n; ++i)
59  if (std::abs(v.val(i)) <= coeff) v.mark_delete(i);
60  v.compress_indices(); // NOTE sparse flags are reset here
61 }

63 /// \brief apply space limitation dropping
64 /// \tparam SpVecType sparse vector type, see \ref SparseVector
65 /// \param[in] nnz reference (local) number of nonzeros, i.e. input matrix
66 /// \param[in] alpha filling limiter
67 /// \param[in,out] v sparse vector
68 /// \param[in] start_size starting size, default is 0
69 /// \sa apply_num_dropping
70 /// \ingroup scl
71 template <class SpVecType>
72 inline void apply_space_dropping(
73  const typename SpVecType::size_type nnz, const double alpha, SpVecType &v,
74  const typename SpVecType::size_type start_size = 0u) {
75  using size_type = typename SpVecType::size_type;
76  using index_type = typename SpVecType::index_type;
77  using extractor = internal::SpVInternalExtractor<SpVecType>;
78 
79  if (alpha <= 0.0) return;
80 
81  if (v.size()) {
82  size_type N1 = std::ceil(alpha * nnz);
83  if (start_size >= N1) N1 = start_size + 1;
84  const size_type N = N1 - start_size;
85  hif_assert(N != 0u, "zero number of limitation!");
86  const size_type n = v.size();
87  if (n > N) {
88  // we need to extract the N values with largest mag of values, in other
89  // words, the rest any entry in the rest n2-N is smaller the extracted
90  // entries in terms of mag
91 
92  auto & inds = v.inds(); // std::vector
93  const auto &vals = v.vals(); // std::vector
94  auto first = inds.begin(), last = first + n;
95 
96  if (N == 1u)
97  // special case, not sure if nth_element can have nth == first
98  std::iter_swap(first,
99  std::max_element(
100  first, last,
101  [&](const index_type i, const index_type j) -> bool {
102  return std::abs(vals[i]) < std::abs(vals[j]);
103  }));
104  else
105  // using c++ built-in selection/partition to find the n largest
106  // values. Note that it's most likely that the internal implementation
107  // is using introselect, where the average complexity is O(n2), worst
108  // cast O(n2 log n2)
109  //
110  // explain of the routine (basically the lambda CMP):
111  // after this routine returns, the following relation is satisfied:
112  // for each j in [first, first+N) and i in [first+N,last)
113  // the lambda CMP(i,j)==false,
114  // in other words, any entry in the first part is an upper bound for the
115  // second part with the following impl
116  std::nth_element(first, first + N - 1, last,
117  [&](const index_type i, const index_type j) -> bool {
118  return std::abs(vals[i]) > std::abs(vals[j]);
119  });
120 
121  // directly modify the internal counter O(1)
122  static_cast<extractor &>(v).counts() = N;
123  }
124  }
125 }
