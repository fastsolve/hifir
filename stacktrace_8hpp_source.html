33 // see https://panthema.net/2008/0901-stacktrace-demangled/
34 
35 /// \def LOAD_STACKTRACE(__ss, __limit)
36 /// \brief Load stack trace on error aborts.
37 /// \ingroup util
38 ///
39 /// Currently, this macro only works on *nix systems. It will load C++ symbols
40 /// nicely to \a ostream \a __ss, thus can be later transferred to, e.g.,
41 /// HIF_STDERR
42 
43 #if defined(__GNUC__) && defined(__unix__)
44 # include <cxxabi.h>
45 # include <execinfo.h>
46 # define LOAD_STACKTRACE(__ss, __limit)
47  do {
48  __ss << "stack trace:\n";
49  void * addrs[__limit + 1];
50  const int addr_len = backtrace(addrs, sizeof(addrs) / sizeof(void *));
51  if (addr_len == 0) {
52  __ss << " <empty, possibly corrupt>\n";
53  return;
54  }
55  char ** symbol_list = backtrace_symbols(addrs, addr_len);
56  std::size_t func_name_size = 512u;
57  char * func_name = (char *)std::malloc(func_name_size);
58  for (int i = 1; i < addr_len; ++i) {
59  char *begin_name(nullptr), *begin_offset(nullptr),
60  *end_offset(nullptr);
61  for (char *p = symbol_list[i]; *p; ++p) {
62  if (*p == '(')
63  begin_name = p;
64  else if (*p == '+')
65  begin_offset = p;
66  else if (*p == ')' && begin_offset) {
67  end_offset = p;
68  break;
69  }
70  }
71  if (begin_name && begin_offset && end_offset &&
72  begin_name < begin_offset) {
73  *begin_name++ = '\0';
74  *begin_offset++ = '\0';
75  *end_offset = '\0';
76  int status;
77  char *ret = abi::__cxa_demangle(begin_name, func_name,
78  &func_name_size, &status);
79  if (status == 0) {
80  func_name = ret;
81  __ss << '[' << i << "] " << symbol_list[i] << ':' << func_name
82  << '+' << begin_offset << '\n';
83  } else
84  __ss << '[' << i << "] " << symbol_list[i] << ':' << begin_name
85  << "()+" << begin_offset << '\n';
86  } else
87  __ss << '[' << i << "] " << symbol_list[i] << '\n';
88  }
89  std::free(func_name);
90  std::free(symbol_list);
91  } while (false)
92 #else
93 # define LOAD_STACKTRACE(__ss, __limit)
94  do {
95  __ss << "stack trace is not available on the platform.\n";
96  } while (false)
97 #endif // __GNUC__
98 
99 #endif // _HIF_UTILS_STACKTRACE_HPP
