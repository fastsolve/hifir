48 /// \brief Determine total number of intervals given a compressed storage
49 /// \tparam MaxInterval Maximum interval upper bound
50 /// \tparam CsType Compressed storage, i.e., \ref CRS or \ref CCS
51 /// \tparam IndexArray Index array type, i.e., \ref Array
52 /// \param[in] A Input compressed matrix
53 /// \param[out] itrv_start output interval starting position array
54 template <std::size_t MaxInterval, class CsType, class IndexArray>
55 inline void determine_nitrvs(const CsType &A, IndexArray &itrv_start) {
56  using size_type = typename CsType::size_type;
57  // resize
58  itrv_start.resize(A.ind_start().size());
59  if (!itrv_start.size()) return;
60 
61  // the first element should be 1
62  itrv_start.front() = 0;
63 
64  const auto n = itrv_start.size() - 1;
65 
66  for (size_type i(0); i < n; ++i) {
67  if (!A.nnz_in_primary(i)) {
68  // empty row/column
69  itrv_start[i + 1] = itrv_start[i];
70  continue;
71  }
72  auto iter = A.ind_cbegin(i) + 1, iter_last = A.ind_cend(i);
73  size_type nitrvs(1); // NOTE: always start with 1
74  size_type len(1);
75  while (iter != iter_last) {
76  if (*(iter - 1) + 1 != *iter || len > MaxInterval) {
77  ++nitrvs; // increment number of intervals
78  len = 0; // reset length
79  }
80  ++iter;
81  ++len;
82  }
83  itrv_start[i + 1] = itrv_start[i] + nitrvs;
84  }
85 }

87 /// \brief Build the interval arrays
88 /// \tparam CsType Compressed storage, i.e., \ref CRS or \ref CCS
89 /// \tparam IndexArray Index array type, i.e., \ref Array
90 /// \tparam IntervalArray Interval array type, i.e., \ref Array
91 /// \param[in] A Input compressed matrix
92 /// \param[in] itrv_start Input interval startingposition array
93 /// \param[out] ind_start Output starting index of each interval
94 /// \param[out] lens Output lengths of each interval, as to \a ind_start
95 template <class CsType, class IndexArray, class IntervalArray>
96 inline void build_intervals(const CsType &A, const IndexArray &itrv_start,
97  IndexArray &ind_start, IntervalArray &lens) {
98  using size_type = typename CsType::size_type;
99 
100  const size_type total_nitrvs = itrv_start.back();
101  ind_start.resize(total_nitrvs);
102  lens.resize(total_nitrvs);
103 
104  const auto n = itrv_start.size() - 1;
105 
106  auto ind_iter = ind_start.begin();
107  auto len_iter = lens.begin();
108 
109  for (size_type i(0); i < n; ++i) {
110  if (!A.nnz_in_primary(i)) continue;
111  auto iter = A.ind_cbegin(i), iter_last = A.ind_cend(i);
112  *ind_iter = *iter++; // increment iter here
113  *len_iter = 1;
114  while (iter != iter_last) {
115  // XXX: !len:-> overflow back to zero
116  if (*(iter - 1) + 1 != *iter || !*len_iter) {
117  // get a breaking point
118  *++ind_iter = *iter;
119  if (!*len_iter) --(*len_iter);
120  *++len_iter = 0;
121  }
122  ++iter;
123  ++(*len_iter);
124  }
125  ++ind_iter;
126  ++len_iter;
127  hif_assert(ind_iter == ind_start.begin() + itrv_start[i + 1], "fatal, %zd",
128  size_type((ind_start.begin() + itrv_start[i + 1]) - ind_iter));
129  hif_assert(len_iter == lens.begin() + itrv_start[i + 1], "fatal, %zd",
130  size_type((lens.begin() + itrv_start[i + 1]) - len_iter));
131  }
132 }

134 /// \brief Analyze the cost ratio of interval to classical storages
135 /// \tparam INDEX_SIZE Index size
136 /// \tparam VALUE_SIZE Value type size
137 /// \tparam INTERVAL_SIZE Interval type size
138 /// \param[in] n Number of rows/columns
139 /// \param[in] nnz Total number of nonzeros
140 /// \param[in] nitrvs Total number of itervals
141 /// \return double
142 /// \note If the result is greater than 1, it means we are wasting memory by
143 /// using interval storage
144 template <std::size_t INDEX_SIZE, std::size_t VALUE_SIZE,
145  std::size_t INTERVAL_SIZE>
146 inline double analyze_storage_cost_ratio(const std::size_t n,
147  const std::size_t nnz,
148  const std::size_t nitrvs) {
149  // cost of classical
150  const double classical_size =
151  (double)nnz * (INDEX_SIZE + VALUE_SIZE) + (double)(n + 1) * INDEX_SIZE;
152  // cost of interval
153  const double interval_size = (double)nnz * VALUE_SIZE +
154  (double)nitrvs * (INDEX_SIZE + INTERVAL_SIZE) +
155  2.0 * (n + 1) * INDEX_SIZE;
156  return classical_size != 0.0 ? interval_size / classical_size : 0.0;
157 }

273  /// \brief matrix vector multiplication (low-level API)
274  /// \tparam Vx other value type for \a x
275  /// \tparam Vy other value type for \a y
276  /// \param[in] x input array pointer
277  /// \param[in] istart index start
278  /// \param[in] len local length
279  /// \param[out] y output array pointer
280  /// \warning User's responsibility to maintain valid pointers
281  template <class Vx, class Vy>
282  inline void multiply_nt_low(const Vx *x, const size_type istart,
283  const size_type len, Vy *y) const {
284  if (_ref)
285  _ref->multiply_nt_low(x, istart, len, y);
286  else {
287  const auto n = istart + len;
288  for (size_type i(istart); i < n; ++i) {
289  auto val_i = _vals.cbegin() + _row_start[i];
290  auto len_iter = _col_len.cbegin() + _itrv_start[i];
291  auto last = _col_ind_start.cbegin() + _itrv_start[i + 1];
292  typename std::conditional<(sizeof(Vy) > sizeof(value_type)), Vy,
293  value_type>::type tmp(0);
294  // loop thru all intervals
295  for (auto iter = _col_ind_start.cbegin() + _itrv_start[i]; iter != last;
296  ++iter, ++len_iter) {
297  const auto idx = *iter;
298  const interval_type m = *len_iter;
299  // for each interval, loop thru the consecutive range
300  for (interval_type j(0); j < m; ++j, ++val_i)
301  tmp += *val_i * x[idx + j];
302  }
303  y[i] = tmp;
304  }
305  }
306  }

352  /// \brief matrix transpose vector multiplication with different type
353  /// \tparam Vx other value type for \a x
354  /// \tparam Vy other value type for \a y
355  /// \param[in] x input array pointer
356  /// \param[out] y output array pointer
357  /// \warning User's responsibility to maintain valid pointers
358  template <class Vx, class Vy>
359  inline void multiply_t_low(const Vx *x, Vy *y) const {
360  if (!_nrows) return;

362  _ref->multiply_t_low(x, y);
363  else {
364  std::fill_n(y, _ncols, Vy(0));
365  for (size_type i(0); i < _nrows; ++i) {
366  const auto temp = x[i];
367  auto val_i = _vals.cbegin() + _row_start[i];
368  auto len_iter = _col_len.cbegin() + _itrv_start[i];
369  auto last = _col_ind_start.cbegin() + _itrv_start[i + 1];
370  // loop thru all intervals
371  for (auto iter = _col_ind_start.cbegin() + _itrv_start[i]; iter != last;
372  ++iter, ++len_iter) {
373  const auto idx = *iter;
374  const interval_type m = *len_iter;
375  // for each interval, loop thru the consecutive range
376  for (interval_type j(0); j < m; ++j, ++val_i)
377  y[idx + j] += *val_i * temp;
378  }
379  }
380  }

409  /// \brief Assume as a strict lower matrix and solve with forward sub
410  /// \tparam RhsType Rhs type
411  /// \param[in,out] y Input rhs, output solution
412  /// \note Advanced use for preconditioner solve.
413  /// \sa solve_as_strict_upper
414  template <class RhsType>
415  inline void solve_as_strict_lower(RhsType &y) const {
416  if (_ref) {
417  _ref->solve_as_strict_lower(y);
418  return;
419  }
420  // retrieve value type from rhs
421  using value_type_ = typename std::remove_reference<decltype(y[0])>::type;
422  for (size_type j(1); j < _nrows; ++j) {
423  auto v_itr = _vals.cbegin() + _row_start[j];
424  typename std::conditional<(sizeof(value_type_) < sizeof(value_type)),
425  value_type, value_type_>::type tmp(0);
426  auto len_iter = _col_len.cbegin() + _itrv_start[j];
427  auto last = _col_ind_start.cbegin() + _itrv_start[j + 1];
428  // loop thru all intervals
429  for (auto iter = _col_ind_start.cbegin() + _itrv_start[j]; iter != last;
430  ++iter, ++len_iter) {
431  const auto idx = *iter;
432  const interval_type m = *len_iter;
433  // for each interval, loop thru the consecutive range
434  for (interval_type i(0); i < m; ++i, ++v_itr)
435  tmp += *v_itr * y[idx + i];
436  }
437  y[j] -= tmp;
438  }
439  }
440 
441  /// \brief Assume as a strict upper matrix and solve with backward sub
442  /// \tparam RhsType Rhs type
443  /// \param[in,out] y Input rhs, output solution
444  /// \note Advanced use for preconditioner solve.
445  /// \sa solve_as_strict_lower
446  template <class RhsType>
447  inline void solve_as_strict_upper(RhsType &y) const {
448  if (_ref) {
449  _ref->solve_as_strict_upper(y);
450  return;
451  }
452  // retrieve value type from rhs
453  using value_type_ = typename std::remove_reference<decltype(y[0])>::type;
454  hif_assert(_nrows, "cannot be empty");
455  for (size_type j = _nrows - 1; j != 0u; --j) {
456  const size_type j1 = j - 1;
457  auto v_itr = _vals.cbegin() + _row_start[j1];
458  typename std::conditional<(sizeof(value_type_) < sizeof(value_type)),
459  value_type, value_type_>::type tmp(0);
460  auto len_iter = _col_len.cbegin() + _itrv_start[j1];
461  auto last = _col_ind_start.cbegin() + _itrv_start[j];
462  // loop thru all intervals
463  for (auto iter = _col_ind_start.cbegin() + _itrv_start[j1]; iter != last;
464  ++iter, ++len_iter) {
465  const auto idx = *iter;
466  const interval_type m = *len_iter;
467  // for each interval, loop thru the consecutive range
468  for (interval_type i(0); i < m; ++i, ++v_itr)
469  tmp += *v_itr * y[idx + i];
470  }
471  y[j1] -= tmp;
472  }
473  }

587  /// \brief matrix vector multiplication with different value type
588  /// \tparam Vx other value type for \a x
589  /// \tparam Vy other value type for \a y
590  /// \param[in] x input array pointer
591  /// \param[out] y output array pointer
592  /// \warning User's responsibilty to ensure valid pointers
593  template <class Vx, class Vy>
594  inline void multiply_nt_low(const Vx *x, Vy *y) const {
595  if (!_ncols) return;
596  if (_ref)
597  _ref->multiply_nt_low(x, y);
598  else {
599  std::fill_n(y, _nrows, Vy(0));
600  for (size_type i(0); i < _ncols; ++i) {
601  const auto temp = x[i];
602  auto val_i = _vals.cbegin() + _col_start[i];
603  auto len_iter = _row_len.cbegin() + _itrv_start[i];
604  auto last = _row_ind_start.cbegin() + _itrv_start[i + 1];
605  // loop thru all intervals
606  for (auto iter = _row_ind_start.cbegin() + _itrv_start[i]; iter != last;
607  ++iter, ++len_iter) {
608  const auto idx = *iter;
609  const interval_type m = *len_iter;
610  // for each interval, loop thru the consecutive range
611  for (interval_type j(0); j < m; ++j, ++val_i)
612  y[idx + j] += *val_i * temp;
613  }
614  }
615  }
616  }

687  /// \brief Assume as a strict lower matrix and solve with forward sub
688  /// \tparam RhsType Rhs type
689  /// \param[in,out] y Input rhs, output solution
690  /// \note Advanced use for preconditioner solve.
691  /// \sa solve_as_strict_upper
692  template <class RhsType>
693  inline void solve_as_strict_lower(RhsType &y) const {
694  if (_ref) {
695  _ref->solve_as_strict_lower(y);
696  return;
697  }
698  for (size_type j(0); j < _ncols; ++j) {
699  const auto y_j = y[j];
700  auto v_itr = _vals.cbegin() + _col_start[j];
701  auto len_iter = _row_len.cbegin() + _itrv_start[j];
702  auto last = _row_ind_start.cbegin() + _itrv_start[j + 1];
703  // loop thru all intervals
704  for (auto iter = _row_ind_start.cbegin() + _itrv_start[j]; iter != last;
705  ++iter, ++len_iter) {
706  const auto idx = *iter;
707  const interval_type m = *len_iter;
708  // for each interval, loop thru the consecutive range
709  for (interval_type i(0); i < m; ++i, ++v_itr)
710  y[idx + i] -= *v_itr * y_j;
711  }
712  }
713  }

715  /// \brief Assume as a strict upper matrix and solve with backward sub
716  /// \tparam RhsType Rhs type
717  /// \param[in,out] y Input rhs, output solution
718  /// \note Advanced use for preconditioner solve.
719  /// \sa solve_as_strict_lower
720  template <class RhsType>
721  inline void solve_as_strict_upper(RhsType &y) const {
722  if (_ref) {
723  _ref->solve_as_strict_upper(y);
724  return;
725  }
726  using rev_iterator = std::reverse_iterator<const_i_iterator>;
727  using rev_v_iterator = std::reverse_iterator<const_v_iterator>;
728  using rev_itrv_iterator =
729  std::reverse_iterator<typename Array<interval_type>::const_iterator>;
730 
731  for (size_type j(_ncols - 1); j != 0u; --j) {
732  const auto y_j = y[j];
733  auto v_itr = rev_v_iterator(_vals.cbegin() + _col_start[j + 1]);
734  auto len_iter = rev_itrv_iterator(_row_len.cbegin() + _itrv_start[j + 1]);
735  auto last = rev_iterator(_row_ind_start.cbegin() + _itrv_start[j]);
736  // loop through all intervals in reversed order
737  for (auto iter =
738  rev_iterator(_row_ind_start.cbegin() + _itrv_start[j + 1]);
739  iter != last; ++iter, ++len_iter) {
740  const auto idx = *iter - 1;
741  const interval_type m = *len_iter;
742  for (interval_type i(m); i != 0u; --i, ++v_itr)
743  y[idx + i] -= *v_itr * y_j;
744  }
745  }
746  }
