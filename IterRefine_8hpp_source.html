39 /// \class IterRefine
40 /// \tparam ValueType value type used in iterative refinement, e.g., \a double
41 /// \brief Iterative refinement operator
42 /// \ingroup ir
43 /// \note \a ValueType does not need to agree with that used in \a MType
44 template <class ValueType>
45 class IterRefine {
46  public:
47  using value_type = ValueType; ///< value type
48  using array_type = Array<value_type>; ///< array type
49  using size_type = typename array_type::size_type; ///< size type
50 
51  /// \brief default constructor

54  /// \brief Stationary iteration used as refinement
55  /// \tparam MType preconditioner, see \ref HIF
56  /// \tparam Matrix matrix type, see \ref CRS or \ref CCS
57  /// \param[in] M HIF preconditioner
58  /// \param[in] A input matrix
59  /// \param[in] b right-hand side vector
60  /// \param[in] N number of iterations
61  /// \param[out] x solution of Jacobi after \a N iterations
62  /// \param[in] last_dim (optional) dimension for back solve for last level
63  /// default is its numerical rank in \a M
64  /// \param[in] tran (optional) transpose/Herimitian flag, default is false
65  ///
66  /// This function implements the abstracted Jacobi processes as follows
67  ///
68  /// \f{eqnarray*}{
69  /// \mathbf{Ax}&=&\mathbf{b} \\
70  /// (\mathbf{M+A-M})\mathbf{x}&=&\mathbf{b} \\
71  /// \mathbf{Mx}&=&\mathbf{b}-(\mathbf{A-M})\mathbf{x} \\
72  /// \mathbf{x}&=&\mathbf{M}^{-1}(\mathbf{r}+\mathbf{Mx}) \\
73  /// \mathbf{x}_{k+1}&=&\mathbf{M}^{-1}\mathbf{r}+\mathbf{x}_k
74  /// \f}
75  template <class MType, class Matrix, class IArrayType, class OArrayType>
76  inline void iter_refine(const MType &M, const Matrix &A, const IArrayType &b,
77  const size_type N, OArrayType &x,
78  const size_type last_dim = static_cast<size_type>(-1),
79  const bool tran = false) const {
80  if (N <= 1) {
81  // if iteration is less than 2, then use original interface
82  M.solve(b, x, tran, last_dim);
83  return;
84  }
85  // now, allocate space
86  _init(M.ncols());
87  const size_type n(b.size()); // dimension
88  std::fill(x.begin(), x.end(), value_type(0)); // init to zero
89  for (size_type i(0); i < N; ++i) {
90  std::copy(x.cbegin(), x.cend(), _xk.begin()); // copy rhs to x
91  if (i) {
92  // starting 2nd iteration
93  if (!tran)
94  mt::multiply_nt(A, _xk, x);
95  else
96  multiply(A, _xk, x, true);
97  for (size_type j(0); j < n; ++j) x[j] = b[j] - x[j]; // residual
98  } else
99  std::copy_n(b.cbegin(), n, x.begin());
100  M.solve(x, _r, tran, last_dim); // compute inv(M)*x=r
101  for (size_type j(0); j < n; ++j) x[j] = _r[j] + _xk[j]; // update
102  }
103  }

105  /// \brief Stationary iteration used as refinement
106  /// \tparam MType preconditioner, see \ref HIF
107  /// \tparam Matrix matrix type, see \ref CRS or \ref CCS
108  /// \param[in] M HIF preconditioner
109  /// \param[in] A input matrix
110  /// \param[in] b right-hand side vector
111  /// \param[in] N number of iterations
112  /// \param[in] betas lower and upper bound of residual norms
113  /// \param[out] x solution of Jacobi after \a N iterations
114  /// \param[in] last_dim (optional) dimension for back solve for last level
115  /// default is its numerical rank in \a M
116  /// \param[in] tran (optional) transpose/Herimitian flag, default is false
117  /// \return Number of refinements and flag. If flag == 0, then it converges;
118  /// if flag > 0, then it diverges; otherwise, it reaches maxit bound.
119  template <class MType, class Matrix, class IArrayType, class OArrayType>
120  inline std::pair<size_type, int> iter_refine(
121  const MType &M, const Matrix &A, const IArrayType &b, const size_type N,
122  const double *betas, OArrayType &x,
123  const size_type last_dim = static_cast<size_type>(-1),
124  const bool tran = false) const {
125  if (N <= 1) {
126  // if iteration is less than 2, then use original interface
127  M.solve(b, x, tran, last_dim);
128  return std::make_pair(size_type(1), -1);
129  }
130  // compute norm of RHS
131  const double bnorm = norm2(b);
132  if (bnorm == 0.0) {
133  std::fill(x.begin(), x.end(), value_type(0));
134  return std::make_pair(size_type(0), 0);
135  }
136  // now, allocate space
137  _init(M.ncols());
138  const size_type n(b.size()); // dimension
139  std::fill(x.begin(), x.end(), value_type(0)); // init to zero
140  std::copy_n(b.cbegin(), n, _r.begin());
141  size_type iters(0);
142  int flag(0);
143  for (;;) {
144  M.solve(_r, _xk, tran, last_dim); // compute xk=inv(M)*x
145  for (size_type j(0); j < n; ++j) x[j] += _xk[j];
146  if (++iters >= N) {
147  flag = -1;
148  break;
149  }
150  if (!tran)
151  mt::multiply_nt(A, x, _xk);
152  else
153  multiply(A, x, _xk, true);
154  for (size_type j(0); j < n; ++j) _r[j] = b[j] - _xk[j];
155  const double res = norm2(_r) / bnorm;
156  if (res <= betas[0]) break;
157  if (res > betas[1]) {
158  flag = 1;
159  break;
160  }
161  }
162  return std::make_pair(iters, flag);
163  }

169  }

171  protected:

179  }
180 
181  protected:
182  mutable array_type _xk; ///< previous solution
183  mutable array_type _r; ///< residual
184 };
