188  /// \brief check the validity of a matrix
189  /// \tparam IsRow is row major, mainly for error message
190  /// \param[in] other_size size in secondary entry, column size for CRS and
191  /// row size for CCS
192  template <bool IsRow>
193  inline void check_validity(const size_type other_size) const {
194  static const char *pname = IsRow ? "row" : "column";
195  static const char *oname = IsRow ? "column" : "row";
196 
197  if (status() != DATA_UNDEF) {
198  hif_error_if(_ind_start.size() < _psize + 1,
199  "invalid %s size and %s start array length", pname, pname);
200  hif_error_if(_ind_start.front() != 0,
201  "the leading entry in %s start should be zero", pname);
202  if (nnz() > _indices.size() || nnz() > _vals.size())
203  hif_error(
204  "inconsistent between nnz (%s_start(end)-%s_start(first),%zd) and "
205  "indices/values length %zd/%zd",
206  pname, pname, nnz(), _indices.size(), _vals.size());
207  // check each entry
208  for (size_type i = 0u; i < _psize; ++i) {
209  hif_error_if(!std::is_sorted(_ind_cbegin(i), _ind_cend(i),
210  [](const index_type l,
211  const index_type r) { return l <= r; }),
212  "%s %zd (C-based) is not sorted", pname, i);
213  for (auto itr = _ind_cbegin(i), last = _ind_cend(i); itr != last;
214  ++itr) {
215  hif_error_if(size_type(*itr) >= other_size,
216  "%zd (C-based) exceeds %s bound %zd in %s %zd",
217  size_type(*itr), oname, other_size, pname, i);
218  }
219  }
220  }

262  /// \brief push back a \b SORTED index list to _indices
263  /// \tparam Iter iterator type
264  /// \tparam ValueArray dense value array
265  /// \param[in] ii ii-th entry in primary direction
266  /// \param[in] first starting iterator
267  /// \param[in] last ending iterator
268  /// \param[in] v dense value array, the value can be queried from indices
269  /// \warning The indices are assumed to be sorted
270  template <class Iter, class ValueArray>
271  inline void _push_back_primary(const size_type ii, Iter first, Iter last,
272  const ValueArray &v) {
273  hif_assert(ii + 1u == _ind_start.size(),
274  "inconsistent pushing back at entry %zd", ii);
275  (void)ii;
276  // first push back the list
277  hif_assert(_indices.size() == _vals.size(), "fatal error");
278  _indices.push_back(first, last);
279  const auto nnz = std::distance(first, last);
280  const size_type start = _vals.size();
281  hif_assert(nnz >= 0, "reversed iterators");
282  _vals.resize(start + nnz);
283  hif_assert(_indices.size() == _vals.size(), "fatal error");
284  const size_type n = _vals.size();
285  for (auto i = start; i < n; ++i, ++first) {
286  hif_assert(size_type(*first) < v.size(), "%zd exceeds the length of v",
287  size_type(*first));
288  _vals[i] = v[*first];
289  }
290  // finally, push back ind start
291  _ind_start.push_back(_ind_start.back() + nnz);
292  }

294  /// \brief push back two \b SORTED index lists to _indices
295  /// \tparam Iter1 iterator type 1
296  /// \tparam ValueArray1 dense value array 1
297  /// \tparam Iter2 iterator type 2
298  /// \tparam ValueArray2 dense value array 2
299  /// \param[in] ii ii-th entry in primary direction
300  /// \param[in] first1 starting iterator
301  /// \param[in] last1 ending iterator
302  /// \param[in] v1 dense value array, the value can be queried from indices
303  /// \param[in] first2 starting iterator
304  /// \param[in] last2 ending iterator
305  /// \param[in] v2 dense value array, the value can be queried from indices
306  /// \warning The indices are assumed to be sorted
307  template <class Iter1, class ValueArray1, class Iter2, class ValueArray2>
308  inline void _push_back_primary(const size_type ii, Iter1 first1, Iter1 last1,
309  const ValueArray1 &v1, Iter2 first2,
310  Iter2 last2, const ValueArray2 &v2) {
311  hif_assert(ii + 1u == _ind_start.size(),
312  "inconsistent pushing back at entry %zd", ii);
313  (void)ii;
314  // first push back the list
315  hif_assert(_indices.size() == _vals.size(), "fatal error");
316  _indices.push_back(first1, last1);
317  _indices.push_back(first2, last2);
318  const auto nnz1 = std::distance(first1, last1);
319  const auto nnz2 = std::distance(first2, last2);
320  size_type start = _vals.size();
321  hif_assert(nnz1 >= 0 && nnz2 >= 0, "reversed iterators");
322  // first range
323  _vals.resize(start + nnz1);
324  const size_type n1 = _vals.size();
325  for (; start < n1; ++start, ++first1) {
326  hif_assert(size_type(*first1) < v1.size(), "%zd exceeds the length of v1",
327  size_type(*first1));
328  _vals[start] = v1[*first1];
329  }
330  // second range
331  _vals.resize(start + nnz2);
332  const size_type n2 = _vals.size();
333  for (; start < n2; ++start, ++first2) {
334  hif_assert(size_type(*first2) < v2.size(), "%zd exceeds the length of v2",
335  size_type(*first2));
336  _vals[start] = v2[*first2];
337  }
338  // finally, push back ind start
339  _ind_start.push_back(_ind_start.back() + nnz1 + nnz2);
340  }

368  /// \brief split against secondary size
369  /// \tparam IsSecond if \a true, then extract the second part
370  /// \param[in] m size split against to
371  /// \param[in] start position start array
372  /// \param[out] indptr index start array
373  /// \param[out] indices index value array
374  /// \param[out] vals values
375  template <bool IsSecond>
376  inline void _split(const size_type m, const indptr_type *start,
377  iparray_type &indptr, iarray_type &indices,
378  array_type &vals) const {
379  if (!_ind_start.size()) return;
380  indptr.resize(_ind_start.size());
381  hif_error_if(indptr.status() == DATA_UNDEF, "memory allocation failed");
382  auto ind_first = _indices.cbegin();
383  size_type nnz(0);
384  for (size_type i(0); i < _psize; ++i) {
385  auto itr = ind_first + start[i];
386  nnz += IsSecond ? _ind_cend(i) - itr : itr - _ind_cbegin(i);
387  }
388  // if we have an empty matrix
389  if (!nnz) {
390  std::fill(indptr.begin(), indptr.end(), indptr_type(0));
391  return;
392  }
393  indices.resize(nnz);
394  hif_error_if(indices.status() == DATA_UNDEF, "memory allocation failed");
395  vals.resize(nnz);
396  hif_error_if(vals.status() == DATA_UNDEF, "memory allocation failed");
397 
398  auto i_itr = indices.begin();
399  auto v_itr = vals.begin();
400 
401  auto val_first = _vals.cbegin();
402 
403  // now set the indptr front
404  indptr.front() = 0;
405 
406  for (size_type i = 0; i < _psize; ++i) {
407  auto split_ind_itr = start[i] + ind_first;
408  auto split_val_itr = start[i] + val_first;
409  auto bak_itr = i_itr;
410  if (IsSecond) {
411  i_itr = std::copy(split_ind_itr, _ind_cend(i), i_itr);
412  for (auto itr = bak_itr; itr != i_itr; ++itr) *itr -= m;
413  v_itr = std::copy(split_val_itr, val_cend(i), v_itr);
414  } else {
415  i_itr = std::copy(_ind_cbegin(i), split_ind_itr, i_itr);
416  v_itr = std::copy(val_cbegin(i), split_val_itr, v_itr);
417  }
418  indptr[i + 1] = indptr[i] + (i_itr - bak_itr);
419  }
420  }

446  /// \brief split and store to the counterpart type
447  /// \tparam IsSecond if \a true, then extract the second part
448  /// \param[in] m size split against to
449  /// \param[in] n overall size of the secondary direction
450  /// \param[in] start position start array
451  /// \param[out] indptr index start array
452  /// \param[out] indices index value array
453  /// \param[out] vals values
454  template <bool IsSecond>
455  inline void _split_dual(const size_type m, const size_type n,
456  const indptr_type *start, iparray_type &indptr,
457  iarray_type &indices, array_type &vals) const {
458  if (!_ind_start.size()) return;
459  indptr.resize((IsSecond ? n - m : m) + 1);
460  hif_error_if(indptr.status() == DATA_UNDEF, "memory allocation failed");
461  // initialize as zeros
462  std::fill(indptr.begin(), indptr.end(), indptr_type(0));
463  auto ind_first = _indices.cbegin();
464  for (size_type i(0); i < _psize; ++i) {
465  // store the position first
466  auto find_itr = ind_first + start[i];
467  if (IsSecond) {
468  auto Itr = _ind_cend(i);
469  for (auto itr = find_itr; itr != Itr; ++itr) ++indptr[*itr - m + 1];
470  } else
471  for (auto itr = _ind_cbegin(i); itr != find_itr; ++itr)
472  ++indptr[*itr + 1];
473  }
474  const size_type N2 = indptr.size() - 1;
475  for (size_type i(0); i < N2; ++i) indptr[i + 1] += indptr[i];
476 
477  if (!indptr[N2]) {
478  std::fill(indptr.begin(), indptr.end(), indptr_type(0));
479  return;
480  }
481 
482  indices.resize(indptr[N2]);
483  hif_error_if(indices.status() == DATA_UNDEF, "memory allocation failed");
484  vals.resize(indptr[N2]);
485  hif_error_if(vals.status() == DATA_UNDEF, "memory allocation failed");
486 
487  auto val_first = _vals.cbegin();
488 
489  for (size_type i = 0; i < _psize; ++i) {
490  auto split_ind_itr = start[i] + ind_first;
491  if (IsSecond) {
492  auto v_itr = start[i] + val_first;
493  auto Itr = _ind_cend(i);
494  for (auto itr = split_ind_itr; itr != Itr; ++itr, ++v_itr) {
495  const auto idx = *itr - m;
496  indices[indptr[idx]] = i;
497  vals[indptr[idx]] = *v_itr;
498  ++indptr[idx];
499  }
500  } else {
501  auto v_itr = val_cbegin(i);
502  for (auto itr = _ind_cbegin(i); itr != split_ind_itr; ++itr, ++v_itr) {
503  const auto idx = *itr;
504  indices[indptr[idx]] = i;
505  vals[indptr[idx]] = *v_itr;
506  ++indptr[idx];
507  }
508  }
509  }
510 
511  hif_assert(indptr[N2] == indptr[N2 - 1], "fatal");
512 
513  indptr_type tmp(0);
514  for (size_type i(0); i < N2; ++i) std::swap(indptr[i], tmp);
515  }

543  /// \param[out] indptr index pointer array
544  /// \param[out] indices index array
545  /// \param[out] vals numerical value array
546  template <bool DoValue, bool OnlyLeading>
547  inline void _compute_perm(const iarray_type &p, const iarray_type &qt,
548  const size_type leading, iparray_type &indptr,
549  iarray_type &indices, array_type &vals) const {
550  const size_type m(p.size()), n(qt.size());
551  hif_error_if(m != _psize, "permutation out of bound");
552  array_type buf;
553  const size_type psize = OnlyLeading ? leading : m;
554  if (DoValue) {
555  buf.resize(OnlyLeading ? leading : n);
556  hif_error_if(buf.status() == DATA_UNDEF, "memory allocation failed");
557  }
558  indptr.resize(psize + 1);
559  hif_error_if(indptr.status() == DATA_UNDEF, "memory allocation failed");
560  // compute an upper bound of nnz, efficiently
561  size_type nnz(0);
562  for (size_type i(0); i < psize; ++i) nnz += _nnz(p[i]);
563  if (!nnz) {
564  std::fill(indptr.begin(), indptr.end(), indptr_type(0));
565  return;
566  }
567  indptr.front() = 0;
568  indices.resize(nnz);
569  hif_error_if(indices.status() == DATA_UNDEF, "memory allocation failed");
570  if (DoValue) {
571  vals.resize(nnz);
572  hif_error_if(vals.status() == DATA_UNDEF, "memory allocation failed");
573  }
574  auto i_itr = indices.begin();
575  auto v_itr = vals.begin();
576  for (size_type i(0); i < psize; ++i) {
577  auto last = _ind_cend(p[i]);
578  auto V_itr = val_cbegin(p[i]);
579  auto itr_bak = i_itr;
580  for (auto itr = _ind_cbegin(p[i]); itr != last; ++itr, V_itr += DoValue) {
581  if (!OnlyLeading) {
582  *i_itr++ = qt[*itr];
583  if (DoValue) buf[qt[*itr]] = *V_itr;
584  } else {
585  const size_type inv_idx = qt[*itr];
586  if (inv_idx < psize) {
587  *i_itr++ = inv_idx;
588  if (DoValue) buf[inv_idx] = *V_itr;
589  }
590  }
591  }
592  // sort
593  indptr[i + 1] = indptr[i] + (i_itr - itr_bak);
594  std::sort(itr_bak, i_itr);
595  if (DoValue)
596  for (auto itr = itr_bak; itr != i_itr; ++itr, ++v_itr)
597  *v_itr = buf[*itr];
598  }
599  indices.resize(i_itr - indices.begin());
600  if (DoValue) vals.resize(v_itr - vals.begin());
601  }

603  /// \brief extract leading block structure
604  /// \param[in] m leading block size
605  /// \param[out] indptr index pointer array
606  /// \param[out] indices index array
607  inline void _extract_leading_struct(const size_type m, iparray_type &indptr,
608  iarray_type &indices) const {
609  indptr.resize(m + 1);
610  hif_error_if(indptr.status() == DATA_UNDEF, "memory allocation failed");
611  indptr.front() = 0;
612  size_type nz(0);
613  for (size_type i(0); i < m; ++i) {
614  auto last = find_sorted(ind_cbegin(i), ind_cend(i), m).second;
615  indptr[i + 1] = last - _indices.cbegin();
616  nz += (last - ind_cbegin(i));
617  }
618  indices.resize(nz);
619  if (nz) {
620  hif_error_if(indices.status() == DATA_UNDEF, "memory allocation failed");
621  indptr.front() = 0;
622  auto itr = indices.begin();
623  for (size_type i(0); i < m; ++i) {
624  auto last = _indices.cbegin() + indptr[i + 1];
625  itr = std::copy(ind_cbegin(i), last, itr);
626  indptr[i + 1] = indptr[i] + (last - ind_cbegin(i));
627  }
628  }
629  }

654  /// \brief filter out small values
655  /// \param[in] n number of columns for CRS and ros for CCS
656  /// \param[in] eps eliminating threshold, default is 0
657  /// \param[in] shrink if \a false (default), then do not shrink memory
658  /// \return total elminated entries
659  inline size_type _prune(const size_type n, scalar_type eps = 0,
660  const bool shrink = false) {
661  if (_psize) {
662  const size_type psize = _psize;
663  eps = eps < 0.0 ? 0.0 : eps;
664  const size_type nz = nnz();
665  size_type j(0);
666  auto v_itr_first = _vals.cbegin();
667  auto i_itr_first = _indices.cbegin();
668  size_type pos(0);
669  // work space for determine thresholds for the orthogonal direction, i.e.,
670  // column for CRS and row for CCS. n+1 to potentially handle 1-based
671  // index since this maybe called by the user directly, and the user
672  // may use 1-based index.
673  Array<scalar_type> buf(n + 1);
674  std::fill_n(buf.begin(), n, scalar_type(0));
675  // determine the thresholds for the orthogonal direction
676  for (size_type i(0); i < nz; ++i)
677  buf[_indices[i]] = eps * std::max(buf[_indices[i]], std::abs(_vals[i]));
678  for (size_type i(0); i < psize; ++i) {
679  const size_type j_bak(j);
680  auto v_itr = v_itr_first + pos;
681  auto i_itr = i_itr_first + pos;
682  const auto lnnz = _ind_start[i + 1] - pos;
683  if (lnnz) {
684  auto last = v_itr + lnnz;
685  const scalar_type thres =
686  eps *
687  std::abs(*std::max_element(
688  v_itr, last, [](const value_type l, const value_type r) {
689  return std::abs(l) < std::abs(r);
690  }));
691  auto i_itr2 = i_itr;
692  for (auto itr = v_itr; itr != last; ++itr, ++i_itr2)
693  if (!(std::abs(*itr) <= std::min(thres, buf[*i_itr2]))) {
694  _indices[j] = *i_itr2;
695  _vals[j++] = *itr;
696  }
697  }
698  // update position
699  pos = _ind_start[i + 1];
700  _ind_start[i + 1] = _ind_start[i] + (j - j_bak);
701  }
702  const size_type elms = nz - j;
703  if (elms && shrink) {
704  do {
705  array_type vals2(nnz());
706  hif_error_if(nnz() && vals2.status() == DATA_UNDEF,
707  "memory allocation failed");
708  std::copy_n(v_itr_first, nnz(), vals2.begin());
709  _vals.swap(vals2);
710  } while (false);
711  do {
712  iarray_type indices2(nnz());
713  hif_error_if(nnz() && indices2.status() == DATA_UNDEF,
714  "memory allocation failed");
715  std::copy_n(i_itr_first, nnz(), indices2.begin());
716  _indices.swap(indices2);
717  } while (false);
718  } else if (elms) {
719  _vals.resize(nnz());
720  _indices.resize(nnz());
721  }
722  return elms;
723  } else
724  return 0;
725  }
726 
727  protected:
728  iparray_type _ind_start; ///< index pointer array, size of n+1
729  iarray_type _indices; ///< index array, size of nnz
730  array_type _vals; ///< numerical data array, size of nnz
731  size_type _psize; ///< primary size
732 };

734 /// \brief convert from one type to another type, e.g. ccs->crs
735 /// \tparam ValueArray numerical value array type
736 /// \tparam IndexArray index array type
737 /// \tparam IndPtrArray index pointer array
738 /// \param[in] i_ind_start input index start array
739 /// \param[in] i_indices input index array
740 /// \param[in] i_vals input value array
741 /// \param[out] o_ind_start output index start array
742 /// \param[out] o_indices output index array
743 /// \param[out] o_vals output values
744 /// \warning o_indices should be initialized properly with value 0 or 1
745 /// \warning Both o_{indices,vals} should be allocated properly
746 /// \note Work with both Array and \a std::vector
747 /// \ingroup ds
748 template <class ValueArray, class IndexArray, class IndPtrArray>
749 inline void convert_storage(const IndPtrArray &i_ind_start,
750  const IndexArray & i_indices,
751  const ValueArray &i_vals, IndPtrArray &o_ind_start,
752  IndexArray &o_indices, ValueArray &o_vals) {
753  typedef typename ValueArray::size_type size_type;
754  typedef typename IndPtrArray::value_type indptr_type;
755 
756  // both o_{indices,vals} are uninitialized arrays with size of input nnz
757  hif_error_if(i_indices.size() != (size_type)i_ind_start.back(),
758  "nnz %zd does not match that in start array %zd",
759  i_indices.size(), i_ind_start.back());
760  hif_error_if(i_indices.size() != i_vals.size(),
761  "nnz sizes (%zd,%zd) do not match between indices and vals",
762  i_indices.size(), i_vals.size());
763  hif_error_if(i_indices.size() != o_indices.size(),
764  "input nnz %zd does not match of that the output (%zd)",
765  i_indices.size(), o_indices.size());
766  hif_error_if(o_indices.size() != o_vals.size(),
767  "nnz sizes (%zd,%zd) do not match between indices and vals",
768  o_indices.size(), o_indices.size());
769 
770  const size_type o_n = o_ind_start.size() - 1u;
771  const size_type i_n = i_ind_start.size() - 1u;
772 
773  // step 1, counts nnz per secondary direction, O(nnz)
774  for (const auto p : i_indices) {
775  hif_assert((size_type)p < o_n, "%zd exceeds the bound %zd", (size_type)p,
776  o_n);
777  ++o_ind_start[p + 1];
778  }
779 
780  // step 2, build o_ind_start, O(n)
781  for (size_type i = 0u; i < o_n; ++i) o_ind_start[i + 1] += o_ind_start[i];
782  hif_assert(o_ind_start.back() == i_ind_start.back(), "fatal issue");
783 
784  // step 3, build output indices and values, O(nnz)
785  auto i_iiter = i_indices.cbegin(); // index iterator
786  auto i_viter = i_vals.cbegin(); // value iterator
787  for (size_type i = 0u; i < i_n; ++i) {
788  const auto lnnz = i_ind_start[i + 1] - i_ind_start[i];
789  const auto lend = i_iiter + lnnz;
790  for (; i_iiter != lend; ++i_iiter, ++i_viter) {
791  const auto j = *i_iiter;
792  // get position
793  const auto jj = o_ind_start[j];
794  o_indices[jj] = i;
795  o_vals[jj] = *i_viter;
796  // increment the counter
797  ++o_ind_start[j];
798  }
799  }
800  hif_assert(o_ind_start[o_n] == o_ind_start[o_n - 1], "fatal issue");
801 
802  // final step, revert back to previous stage for output index start, O(n)
803  indptr_type temp(0);
804  for (size_type i = 0u; i < o_n; ++i) std::swap(temp, o_ind_start[i]);
805 
806  // NOTE we can create a buffer of size n, so that the last step will become
807  // optional
808 }

1053  /// \brief scale by a diagonal matrix from right
1054  /// \tparam DiagArray diagonal array type
1055  /// \param[in] t diagonal matrix multiplying from right-hand side
1056  /// \sa scale_diag_left
1057  ///
1058  /// Mathematically, this member function is to perform
1059  /// \f$\mathbf{A} \textrm{diag}(\mathbf{t})\f$; the overall complexity
1060  /// is in order of \f$\mathcal{O}(nnz)\f$
1061  template <class DiagArray>
1062  inline void scale_diag_right(const DiagArray &t) {
1063  hif_error_if(_ncols > t.size(), "column sizes do not match (%zd,%zd)",
1064  _ncols, t.size());
1065  auto v_itr = vals().begin();
