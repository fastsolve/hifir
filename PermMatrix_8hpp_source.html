43 /// \class PermMatrix
44 /// \tparam IndexType index type used, e.g. \a int
45 /// \brief permutation matrix defined as the P(i) entry in original matrix is
46 /// the i-th entry in the permutated one
47 template <class IndexType>
48 class PermMatrix {
49  public:
50  typedef IndexType index_type; ///< index type
51  typedef Array<index_type> iarray_type; ///< index array
52  typedef typename iarray_type::size_type size_type; ///< size
53  typedef typename iarray_type::reference reference; ///< reference
54  typedef typename iarray_type::const_reference const_reference;
55  ///< constant reference
56 
57  /// \brief default constructor
58  PermMatrix() = default;
59 
60  /// \brief constructor with own data
61  /// \param[in] n size of permutation vector
62  explicit PermMatrix(const size_type n) : _p(n) {}
63 
64  /// \brief clone of wrap external vector
65  /// \param[in] p external permutation vector
66  /// \param[in] clone if \a false (default), do shallow copy
67  explicit PermMatrix(const iarray_type &p, bool clone = false)
68  : _p(p, clone) {}
69 
70  /// \brief deep- or shallow copy constructor
71  /// \param[in] other another permutation vector
72  /// \param[in] clone if \a false (default), do shallow copy
73  PermMatrix(const PermMatrix &other, bool clone = false)
74  : _p(other._p, clone) {}
75 
76  // default stuffs
77  PermMatrix(PermMatrix &&) = default;
78  PermMatrix &operator=(const PermMatrix &) = default;
79  PermMatrix &operator=(PermMatrix &&) = default;
80 
81  /// \brief resize the permutation vector
82  /// \param[in] n size
83  inline void resize(const size_type n) { _p.resize(n); }
84 
85  /// \brief get the size of permutation vector
86  inline size_type size() const { return _p.size(); }
87 
88  /// \brief swap entries i and j
89  /// \param[in] i entry i
90  /// \param[in] j entry j
91  inline void interchange(const size_type i, const size_type j) {
92  std::swap(_p[i], _p[j]);
93  }
94 
95  /// \brief make an identity permutation
96  inline void make_eye() {
97  const size_type n = _p.size();
98  for (size_type i = 0u; i < n; ++i) _p[i] = i;
99  }
100 
101  /// \brief check if identity mapping
102  inline bool is_eye() const {
103  return size() ? _p.front() == 0 && std::is_sorted(_p.cbegin(), _p.cend())
104  : true;
105  }
106 
107  /// \brief get the permutation entry
108  /// \param[in] i i-th permutation entry
109  inline reference operator[](const size_type i) { return _p[i]; }
110 
111  /// \brief get the permutation entry, constant version
112  /// \param[in] i i-th permutation entry
113  inline const_reference operator[](const size_type i) const { return _p[i]; }
114 
115  /// \brief get the underlaying data
116  inline iarray_type &operator()() { return _p; }
117 
118  /// \brief get the underlaying data, constant reference
119  inline const iarray_type &operator()() const { return _p; }
120 
121  protected:
122  iarray_type _p; ///< underlying permutation array
123 };

186  /// \brief make identity permutation and its inverse mapping
187  inline void make_eye() {
188  hif_assert(_p.size() == _p_inv.size(), "nonmatching sizes");
189  _base::make_eye();
190  std::copy(_p.cbegin(), _p.cend(), _p_inv.begin());
191  }
192 
193  /// \brief build the inverse mapping
194  /// \note Assume that we have a permutation vector that has been setup
195  /// \note Complexity is linear, i.e. \f$\mathcal{O}(n)\f$, to the size
196  inline void build_inv() {
197  hif_assert(_p.size() == _p_inv.size(), "nonmatching sizes");
198  const size_type n = _p.size();
199  for (size_type i = 0u; i < n; ++i) _p_inv[_p[i]] = i;
200  }
201 
202  /// \brief swap two entries and their corresponding inverse mappings
203  /// \param[in] i i-th entry
204  /// \param[in] j j-th entry
205  inline void interchange(const size_type i, const size_type j) {
206  _base::interchange(i, j);
207  std::swap(_p_inv[_p[i]], _p_inv[_p[j]]);
208  }
209 
210  /// \brief swap two inverse entries
211  /// \param[in] i i-th entry
212  /// \param[in] j j-th entry
213  inline void interchange_inv(const size_type i, const size_type j) {
214  std::swap(_p_inv[i], _p_inv[j]);
215  std::swap(_p[_p_inv[i]], _p[_p_inv[j]]);
216  }
217 
218  /// \brief get the underlaying data
219  inline iarray_type &inv() { return _p_inv; }
220 
221  /// \brief get the underlaying data, constant reference
222  inline const iarray_type &inv() const { return _p_inv; }
223 
224  protected:
225  using _base::_p;
226  iarray_type _p_inv; ///< inverse mapping array
227 };
