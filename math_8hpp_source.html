76 /// \brief compute the dot product
77 /// \tparam ArrayType array type, see, for instance, \ref Array
78 /// \tparam Iteratable iteratable type for second array
79 /// \param[in] v1 first array input
80 /// \param[in] v2 second array input
81 /// \return dot product of \a v1 and \a v2
82 template <class ArrayType, class Iteratable>
83 inline typename ArrayType::value_type inner(const ArrayType & v1,
84  const Iteratable &v2) {
85  using value_type = typename ArrayType::value_type;
86  value_type tmp(0);
87  const auto n = v1.size();
88  for (auto i = 0ul; i < n; ++i) tmp += conjugate(v1[i]) * v2[i];
89  return tmp;

107 /// \brief compute the Euclidean norm of a given vector
108 /// \tparam ArrayType array type, see, for instance, \ref Array
109 /// \param[in] v array input
110 template <class ArrayType>
111 inline typename ValueTypeTrait<typename ArrayType::value_type>::value_type
112 norm2(const ArrayType &v) {
113  // get the scalar type
114  using value_type = typename ArrayType::value_type;
115  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
116 
117  scalar_type tmp(0);
118  const auto n = v.size();
119  if (!n) return scalar_type(0);
120  // get the max mag
121  const scalar_type max_mag = abs(*std::max_element(
122  v.cbegin(), v.cend(), [](const value_type &l, const value_type &r) {
123  return abs(l) < abs(r);
124  }));
125 
126  if (max_mag == Const<scalar_type>::ZERO)
127  for (auto i = 0ul; i < n; ++i) tmp += abs(v[i]);
128  else {
129  const auto alpha = scalar_type(1) / max_mag;
130  value_type a;
131  for (auto i = 0ul; i < n; ++i) {
132  a = v[i] * alpha;
133  tmp += real(conjugate(a) * a);
134  }
135  tmp = max_mag * std::sqrt(tmp);
136  }
137  return tmp;
138 }

140 /// \brief normalize an array
141 /// \tparam ArrayType array type, see, for instance, \ref Array
142 /// \param[in,out] v array input, upon output, it will be normalized
143 template <class ArrayType>
144 inline void normalize(ArrayType &v) {
145  // get the scalar type
146  using scalar_type =
147  typename ValueTypeTrait<typename ArrayType::value_type>::value_type;
148  const auto inv_nrm = scalar_type(1) / norm2(v);
149  for (auto iter = v.begin(); iter != v.end(); ++iter) *iter *= inv_nrm;
150 }

156 /// \param[out] w iterator output
