69 /// \brief write a \ref CRS matrix
70 /// \tparam CsType Crs type, see \ref CRS
71 /// \param[in] filename file name
72 /// \param[in] A matrix for writing to file
73 /// \sa read_bin
74 /// \note This function requires HDF5
75 template <class CsType>
76 inline void write_bin(const char *filename, const CsType &A) {
77 #ifdef HIF_HAS_HDF5
78  using value_type = typename CsType::value_type;
79  using index_type = typename CsType::index_type;
80  using indptr_type = typename CsType::indptr_type;
81  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
82 
83 # ifndef H5_NO_NAMESPACE
84  using namespace H5;
85 # endif
86 
87  static_assert(std::is_floating_point<scalar_type>::value,
88  "must be scalar type");
89  static_assert(std::is_integral<index_type>::value &&
90  std::is_integral<indptr_type>::value,
91  "must be integer types");
92  static_assert(sizeof(index_type) >= 4ul && sizeof(indptr_type) >= 4u,
93  "must be at least 32bit integers");
94 
95  std::uint64_t info[4];
96  info[0] = A.nrows(); // nrows
97  info[1] = A.ncols(); // ncols
98  info[2] = CsType::ROW_MAJOR;
99  info[3] = std::is_floating_point<value_type>::value;
100 
101  try {
102  Exception::dontPrint();
103  H5File f(H5std_string(filename), H5F_ACC_TRUNC);
104  // write information
105  hsize_t n = 4;
106  DataSpace infosp(1, &n);
107  auto ds_info = f.createDataSet(H5std_string("sparse_info"),
108  PredType::NATIVE_UINT64, infosp);
109  ds_info.write((const void *)info, PredType::NATIVE_UINT64);
110  // write ind_start
111  const void *ind_start = (const void *)A.ind_start().data();
112  n = A.ind_start().size();
113  DataSpace ind_start_sp(1, &n);
114  if (std::is_same<indptr_type, long long>::value) {
115  const auto &type = PredType::NATIVE_LLONG;
116  auto ds_indptr =
117  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
118  ds_indptr.write(ind_start, type);
119  } else if (std::is_same<indptr_type, unsigned long long>::value) {
120  const auto &type = PredType::NATIVE_ULLONG;
121  auto ds_indptr =
122  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
123  ds_indptr.write(ind_start, type);
124  } else if (std::is_same<indptr_type, long>::value) {
125  const auto &type = PredType::NATIVE_LONG;
126  auto ds_indptr =
127  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
128  ds_indptr.write(ind_start, type);
129  } else if (std::is_same<indptr_type, unsigned long>::value) {
130  const auto &type = PredType::NATIVE_ULONG;
131  auto ds_indptr =
132  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
133  ds_indptr.write(ind_start, type);
134  } else if (std::is_same<indptr_type, int>::value) {
135  const auto &type = PredType::NATIVE_INT;
136  auto ds_indptr =
137  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
138  ds_indptr.write(ind_start, type);
139  } else if (std::is_same<indptr_type, unsigned int>::value) {
140  const auto &type = PredType::NATIVE_UINT;
141  auto ds_indptr =
142  f.createDataSet(H5std_string("ind_start"), type, ind_start_sp);
143  ds_indptr.write(ind_start, type);
144  } else
145  hif_error("unsupported integer type");
146  // write indices
147  n = A.inds().size();
148  DataSpace nnz_sp(1, &n);
149  const void *inds = (const void *)A.inds().data();
150  if (std::is_same<index_type, long long>::value) {
151  const auto &type = PredType::NATIVE_LLONG;
152  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
153  ds_inds.write(inds, type);
154  } else if (std::is_same<index_type, unsigned long long>::value) {
155  const auto &type = PredType::NATIVE_ULLONG;
156  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
157  ds_inds.write(inds, type);
158  } else if (std::is_same<index_type, long>::value) {
159  const auto &type = PredType::NATIVE_LONG;
160  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
161  ds_inds.write(inds, type);
162  } else if (std::is_same<index_type, unsigned long>::value) {
163  const auto &type = PredType::NATIVE_ULONG;
164  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
165  ds_inds.write(inds, type);
166  } else if (std::is_same<index_type, int>::value) {
167  const auto &type = PredType::NATIVE_INT;
168  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
169  ds_inds.write(inds, type);
170  } else if (std::is_same<index_type, unsigned int>::value) {
171  const auto &type = PredType::NATIVE_UINT;
172  auto ds_inds = f.createDataSet(H5std_string("indices"), type, nnz_sp);
173  ds_inds.write(inds, type);
174  } else
175  hif_error("unsupported integer type");
176  // write values
177  const void *vals = (const void *)A.vals().data();
178  n = A.vals().size() * (info[3] == 0u ? 2 : 1);
179  DataSpace nnz_sp2(1, &n);
180  if (std::is_same<scalar_type, double>::value) {
181  const auto &type = PredType::NATIVE_DOUBLE;
182  auto ds_vals = f.createDataSet(H5std_string("vals"), type, nnz_sp2);
183  ds_vals.write(vals, type);
184  } else if (std::is_same<scalar_type, float>::value) {
185  const auto &type = PredType::NATIVE_FLOAT;
186  auto ds_vals = f.createDataSet(H5std_string("vals"), type, nnz_sp2);
187  ds_vals.write(vals, type);
188  } else {
189  const auto &type = PredType::NATIVE_LDOUBLE;
190  auto ds_vals = f.createDataSet(H5std_string("vals"), type, nnz_sp2);
191  ds_vals.write(vals, type);
192  }
193  } catch (const Exception &h5e) {
194  hif_error("HDF5 returned error from func:%s with message: %s",
195  h5e.getCFuncName(), h5e.getCDetailMsg());
196  } catch (...) {
197  throw;
198  }
199 #else
200  (void)filename;
201  (void)A;
202  hif_error("write_bin requires HDF5");
203 #endif
204 }

206 /// \brief read data from file to a \ref CRS matrix
207 /// \tparam IndPtrArray indptr array
208 /// \tparam IndexArray index array type
209 /// \tparam ValueArray value array type
210 /// \param[in] filename file name
211 /// \param[out] info sparse matrix information
212 /// \param[out] ind_start indptr array
213 /// \param[out] indices index array
214 /// \param[out] vals value array
215 /// \sa write_bin
216 /// \note This function requires HDF5
217 template <class IndPtrArray, class IndexArray, class ValueArray>
218 inline void read_bin(const char *filename, std::uint64_t info[],
219  IndPtrArray &ind_start, IndexArray &indices,
220  ValueArray &vals) {
221 #ifdef HIF_HAS_HDF5
222  using value_type = typename ValueArray::value_type;
223  using index_type = typename IndexArray::value_type;
224  using indptr_type = typename IndPtrArray::value_type;
225  using scalar_type = typename ValueTypeTrait<value_type>::value_type;
226 
227 # ifndef H5_NO_NAMESPACE
228  using namespace H5;
229 # endif
230 
231  try {
232  Exception::dontPrint();
233  H5File f(H5std_string(filename), H5F_ACC_RDONLY);
234  f.openDataSet(H5std_string("sparse_info"))
235  .read((void *)info, PredType::NATIVE_UINT64);
236  if ((std::is_floating_point<value_type>::value && (info[3] == 0u)) ||
237  (!std::is_floating_point<value_type>::value && (info[3] == 1u)))
238  hif_error("cannot mixed real and complex in read_bin");
239  if (info[2])
240  ind_start.resize(info[0] + 1);
241  else
242  ind_start.resize(info[1] + 1);
243  auto ds_indptr = f.openDataSet(H5std_string("ind_start"));
244  if (sizeof(indptr_type) == ds_indptr.getDataType().getSize())
245  ds_indptr.read((void *)ind_start.data(), ds_indptr.getDataType());
246  else if (ds_indptr.getDataType().getSize() == 4) {
247  Array<int> buf(ind_start.size());
248  ds_indptr.read((void *)buf.data(), ds_indptr.getDataType());
249  std::copy(buf.cbegin(), buf.cend(), ind_start.begin());
250  } else {
251  Array<long long> buf(ind_start.size());
252  ds_indptr.read((void *)buf.data(), ds_indptr.getDataType());
253  std::copy(buf.cbegin(), buf.cend(), ind_start.begin());
254  }
255  auto ds_inds = f.openDataSet(H5std_string("indices"));
256  auto n = ds_inds.getSpace().getSelectNpoints();
257  indices.resize(n);
258  if (sizeof(index_type) == ds_inds.getDataType().getSize())
259  ds_inds.read((void *)indices.data(), ds_inds.getDataType());
260  else if (ds_inds.getDataType().getSize() == 4) {
261  Array<int> buf(indices.size());
262  ds_inds.read((void *)buf.data(), ds_inds.getDataType());
263  std::copy(buf.cbegin(), buf.cend(), indices.begin());
264  } else {
265  Array<long long> buf(indices.size());
266  ds_inds.read((void *)buf.data(), ds_inds.getDataType());
267  std::copy(buf.cbegin(), buf.cend(), indices.begin());
268  }
269  auto ds_vals = f.openDataSet(H5std_string("vals"));
270  n = ds_vals.getSpace().getSelectNpoints();
271  if (info[3] == 0u)
272  vals.resize(n / 2); // complex
273  else
274  vals.resize(n);
275  const int fac = info[3] == 0u ? 2 : 1;
276  if (sizeof(value_type) == ds_vals.getDataType().getSize() * fac)
277  ds_vals.read((void *)vals.data(), ds_vals.getDataType());
278  else if (ds_vals.getDataType().getSize() == 4) {
279  Array<float> buf(n);
280  ds_vals.read((void *)buf.data(), ds_vals.getDataType());
281  std::copy_n(buf.data(), n, (scalar_type *)vals.data());
282  } else if (ds_vals.getDataType().getSize() == 8) {
283  Array<double> buf(n);
284  ds_vals.read((void *)buf.data(), ds_vals.getDataType());
285  std::copy_n(buf.data(), n, (scalar_type *)vals.data());
286  } else {
287  Array<long double> buf(n);
288  ds_vals.read((void *)buf.data(), ds_vals.getDataType());
289  std::copy_n(buf.data(), n, (scalar_type *)vals.data());
290  }
291  } catch (const Exception &h5e) {
292  hif_error("HDF5 returned error from func:%s with message: %s",
293  h5e.getCFuncName(), h5e.getCDetailMsg());
294  } catch (...) {
295  throw;
296  }
297 #else
298  (void)filename;
299  (void)info;
300  (void)ind_start;
301  (void)indices;
302  (void)vals;
303  hif_error("read_bin requires HDF5");
304 #endif
305 }

309 /// \brief write to ASCII file
310 /// \tparam IsRowMajor flag to indicate \ref CRS or \ref CCS
311 /// \tparam IndexArray index array type, see \ref Array
312 /// \tparam ValueArray value array type, see \ref Array
313 /// \param[in] fname file name for output
314 /// \param[in] ind_start index start array
315 /// \param[in] other_size column size for \ref CRS, row size for \ref CCS
316 /// \param[in] indices index array
317 /// \param[in] vals value array
318 /// \param[in] m leading symmetric block size
319 template <bool IsRowMajor, class IndexArray, class ValueArray>
320 inline void write_ascii(const char *fname, const IndexArray &ind_start,
321  const typename IndexArray::size_type other_size,
322  const IndexArray &indices, const ValueArray &vals,
323  const typename IndexArray::size_type m) {
324  constexpr static char is_row = IsRowMajor ? 'R' : 'C';
325  constexpr static char is_c = 'C';
326  using value_type = typename ValueArray::value_type;
327  constexpr static int data_size = sizeof(value_type);
328  constexpr static char dtype = std::is_floating_point<value_type>::value
329  ? (data_size == 8 ? 'D' : 'S')
330  : (data_size == 16 ? 'Z' : 'C');
331  constexpr static int prec = dtype == 'D' || dtype == 'Z' ? 16 : 8;
332 
333  const static char info[4] = {is_row, is_c, dtype, '\0'};
334 
335  if (!ind_start.size()) hif_error("cannot write matrix with empty ind_start");
336 
337  decltype(other_size) nrows(IsRowMajor ? ind_start.size() - 1 : other_size),
338  ncols(IsRowMajor ? other_size : ind_start.size() - 1),
339  nnz(indices.size());
340 
341  hif_error_if(vals.size() != nnz, "inconsistent nnz");
342  hif_error_if(nnz != decltype(nnz)(ind_start.back() - ind_start.front()),
343  "inconsistent nnz");
344 
345  std::ofstream f(fname);
346  hif_error_if(!f.is_open(), "cannot open file %s.", fname);
347 
348  f << info << '\n';
349  f << nrows << ' ' << ncols << ' ' << nnz << ' ' << m << '\n';
350 
351  for (const auto v : ind_start) f << v << '\n';
352  for (const auto v : indices) f << v << '\n';
353  f.setf(std::ios::scientific);
354  f.precision(prec);
355  for (const auto v : vals) f << v << '\n';
356 
357  f.close();
358 }

360 /// \brief query information of an ASCII file
361 /// \param[in] filename binary filename
362 /// \return tuple of CRS/CCS order, C-index, double/float, is real, nrows,
363 /// ncols, nnz, m
364 /// \ingroup itr
365 /// \sa query_info_bin
366 inline std::tuple<bool, bool, bool, bool, std::uint64_t, std::uint64_t,
367  std::uint64_t, std::uint64_t>
368 query_info_ascii(const char *filename) {
369  const static char dtypes[5] = {'D', 'S', 'Z', 'C', '\0'};
370 
371  std::ifstream f(filename);
372  hif_error_if(!f.is_open(), "cannot open file %s.", filename);
373 
374  bool is_row, is_c;
375  char d;
376  std::size_t nrows, ncols, nnz, m;
377 
378  std::string buf;
379  for (;;) {
380  buf.clear();
381  std::getline(f, buf);
382  if (!buf.size() || buf[0] == '#') continue;
383  break;
384  }
385 
386  // buf should contain the information
387  if (buf.size() != 3u) hif_error("not a valid native hif ascii file");
388  if (buf[0] != 'R' && buf[0] != 'C')
389  hif_error("the first char should be either R or C (row or column major)");
390 
391  // skip comments
392  {
393  int i = 0;
394  for (; i < 4; ++i)
395  if (buf.back() == dtypes[i]) break;
396  if (i == 4) hif_error("unknown data type");
397  }
398 
399  is_row = buf.front() == 'R';
400  is_c = buf[1] == 'C';
401  d = buf.back();
402 
403  const bool is_double = d == 'D' || d == 'Z';
404  const bool is_real = d == 'D' || d == 'S';
405 
406  // read sizes
407  f >> nrows >> ncols >> nnz >> m;
408 
409  return std::make_tuple(is_row, is_c, is_double, is_real, nrows, ncols, nnz,
410  m);
411 }

413 /// \brief read data from ASCII file
414 /// \tparam IndexArray index array type, see \ref Array
415 /// \tparam ValueArray value array type, see \ref Array
416 /// \param[in] fname file name for output
417 /// \param[in] ind_start index start array
418 /// \param[in] indices index array
419 /// \param[in] vals value array
420 template <class IndexArray, class ValueArray>
421 inline std::tuple<bool, bool, char, std::size_t, std::size_t, std::size_t,
422  std::size_t>
423 read_ascii(const char *fname, IndexArray &ind_start, IndexArray &indices,
424  ValueArray &vals) {
425  using value_type = typename ValueArray::value_type;
426  constexpr static bool is_real = std::is_floating_point<value_type>::value;
427  const static char dtypes[5] = {'D', 'S', 'Z', 'C', '\0'};
428 
429  std::ifstream f(fname);
430  hif_error_if(!f.is_open(), "cannot open file %s.", fname);
431 
432  bool is_row, is_c;
433  char d;
434  std::size_t nrows, ncols, nnz, m;
435 
436  std::string buf;
437  for (;;) {
438  buf.clear();
439  std::getline(f, buf);
440  if (!buf.size() || buf[0] == '#') continue;
441  break;
442  }
443 
444  // buf should contain the information
445  if (buf.size() != 3u) hif_error("not a valid native hif ascii file");
446  if (buf[0] != 'R' && buf[0] != 'C')
447  hif_error("the first char should be either R or C (row or column major)");
448  if (buf[1] != 'C' && buf[1] != 'F')
449  hif_error(
450  "the second char should be either C or F (c index or fortran index)");
451 
452  // skip comments
453  {
454  int i = 0;
455  for (; i < 4; ++i)
456  if (buf.back() == dtypes[i]) break;
457  if (i == 4) hif_error("unknown data type");
458  }
459 
460  is_row = buf.front() == 'R';
461  is_c = buf[1] == 'C';
462  d = buf.back();
463 
464  if (is_real) {
465  hif_error_if(d == 'Z' || d == 'C',
466  "cannot load complex data to real array");
467  } else {
468  hif_error_if(d == 'D' || d == 'S',
469  "cannot load real data to complex array");
470  }
471 
472  // read sizes
473  f >> nrows >> ncols >> nnz >> m;
474 
475 # ifdef HIF_DEBUG
476  hif_info("file %s has size attributes: %zd, %zd, %zd, %zd", fname, nrows,
477  ncols, nnz, m);
478 # endif
479 
480  const std::size_t primary_size = is_row ? nrows + 1 : ncols + 1;
481  ind_start.resize(primary_size);
482  hif_error_if(ind_start.status() == DATA_UNDEF, "memory allocation failed");
483  indices.resize(nnz);
484  vals.resize(nnz);
485  hif_error_if(indices.status() == DATA_UNDEF || vals.status() == DATA_UNDEF,
486  "memory allocation failed");
487 
488  for (std::size_t i = 0u; i < primary_size; ++i) f >> ind_start[i];
489  for (std::size_t i = 0u; i < nnz; ++i) f >> indices[i];
490  for (std::size_t i = 0u; i < nnz; ++i) f >> vals[i];
491 
492  if (!is_c) {
493  for (auto &v : ind_start) --v;
494  for (auto &v : indices) --v;
495  }
496 
497  f.close();
498 
499  return std::make_tuple(is_row, is_c, d, nrows, ncols, nnz, m);
500 }

546 template <int TypeID, bool IsReal, class IndexType, class ValueType>

567  std::size_t index(0);
568  int r, c;
569  double vr, vi;
570  for (std::size_t i(0); i < nnz; ++i) {
571  const int flag = IsReal ? std::fscanf(f, "%d %d %lg", &r, &c, &vr)
572  : std::fscanf(f, "%d %d %lg %lg", &r, &c, &vr, &vi);
573  if (flag != 4 - IsReal) {
574  std::fclose(f);
575  hif_error("insufficient objects scanned from fscanf");
576  }
577  rows[index] = r - 1; // to 0-based index
578  cols[index] = c - 1; // to 0-based index
579  set_real(vr, vals[index]);
580  set_imag(vi, vals[index]);
581  ++index;
582  if (TypeID == GENERAL || r == c) continue;
583  // not general and not along the diagonal
584  rows[index] = c - 1; // to 0-based index
585  cols[index] = r - 1; // to 0-based index
586  if (TypeID == SYMM)
587  vals[index] = vals[index - 1];
588  else if (TypeID == HERM)
589  vals[index] = conjugate(vals[index - 1]);
590  else
591  vals[index] = -vals[index - 1];
592  ++index;
593  }
594  if (index != nnz) {
595  rows.resize(index);
596  cols.resize(index);
597  vals.resize(index);
598  }
599 }

655  storage_scheme_str(storage_scheme);

657  to_lower(data_type_str);
658  to_lower(storage_scheme_str);
659 
660  if (crd_str != std::string("coordinate") && crd_str != std::string("array")) {
661  std::fclose(f);
662  hif_error("unknown crd %s", crd_str.c_str());
663  }
664  is_sparse = crd_str != std::string("array");
665 
666  if (data_type_str != std::string("real") &&
667  data_type_str != std::string("complex")) {
668  std::fclose(f);
669  hif_error("HIFIR unsupported data type %s", data_type_str.c_str());
670  }
671  is_real = data_type_str != std::string("complex");
672 
673  if (storage_scheme_str == std::string("general"))
674  type_id = 0;
675  else if (storage_scheme_str == std::string("symmetric"))
676  type_id = 1;
677  else if (storage_scheme_str == std::string("hermitian"))
678  type_id = 2;
679  else if (storage_scheme_str == std::string("skew-symmetric"))
680  type_id = 3;
681  else {
682  std::fclose(f);
683  hif_error("unknown storage_scheme %s", storage_scheme_str.c_str());
